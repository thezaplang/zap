222void Binder::applyImports(ModuleState &module, bool allowIncomplete) {
223 std::map<std::string, std::string> namespaceOwners;
224
225 for (const auto &import : module.info->imports) {
226 if (!import.moduleAlias.empty() && import.targetModuleIds.size() != 1) {
227 if (!allowIncomplete) {
228 error(import.span, "Module alias imports are only allowed when the "
229 "path resolves to a single module.");
230 }
231 continue;
232 }
233
234 for (const auto &targetId : import.targetModuleIds) {
235 auto targetIt = modules_.find(targetId);
236 if (targetIt == modules_.end()) {
237 continue;
238 }
239
240 auto &target = targetIt->second;
241 auto alias = import.moduleAlias.empty() ? target.info->moduleName
242 : import.moduleAlias;
243
244 auto existingOwner = namespaceOwners.find(alias);
245 if (existingOwner != namespaceOwners.end() &&
246 existingOwner->second != targetId) {
247 if (!allowIncomplete) {
248 error(import.span, "Import namespace '" + alias +
249 "' is ambiguous because multiple files share "
250 "that module name.");
251 }
252 continue;
253 }
254
255 if (!module.scope->lookupLocal(alias)) {
256 module.scope->declare(alias, target.symbol);
257 namespaceOwners[alias] = targetId;
258 } else if (module.scope->lookupLocal(alias) != target.symbol) {
259 if (!allowIncomplete) {
260 error(import.span, "Cannot import module '" + alias +
261 "' because that name is already declared in "
262 "the current file.");
263 }
264 continue;
265 }
266
267 if (import.bindings.empty()) {
268 bool isImplicitPreludeImport =
269 import.rawPath == "std/prelude" && import.moduleAlias.empty();
270 if (isImplicitPreludeImport) {
271 for (const auto &exported : target.symbol->exports) {
272 if (!module.scope->lookupLocal(exported.first)) {
273 module.scope->declare(exported.first, exported.second);
274 }
275 }
276 }
277
278 if (import.visibility == Visibility::Public) {
279 module.symbol->exports[alias] = target.symbol;
280 for (const auto &exported : target.symbol->exports) {
281 module.symbol->exports[exported.first] = exported.second;
282 }
283 }
284 continue;
285 }
286
287 if (import.targetModuleIds.size() != 1) {
288 if (!allowIncomplete) {
289 error(import.span, "Selective imports are only allowed when the path "
290 "resolves to a single module.");
291 }
292 continue;
293 }
294
295 for (const auto &binding : import.bindings) {
296 auto exportedIt = target.symbol->exports.find(binding.sourceName);
297 if (exportedIt == target.symbol->exports.end()) {
298 if (allowIncomplete) {
299 continue;
300 }
301 auto memberIt = target.symbol->members.find(binding.sourceName);
302 if (memberIt != target.symbol->members.end() &&
303 memberIt->second->visibility != Visibility::Public) {
304 error(import.span, "Member '" + binding.sourceName +
305 "' of module '" + alias + "' is private.");
306 } else {
307 error(import.span, "Module '" + alias + "' has no public member '" +
308 binding.sourceName + "'.");
309 }
310 continue;
311 }
312
313 auto existing = module.scope->lookupLocal(binding.localName);
314 if (existing && existing != exportedIt->second) {
315 if (!allowIncomplete) {
316 error(import.span, "Imported name '" + binding.localName +
317 "' conflicts with an existing declaration "
318 "in the current file.");
319 }
320 continue;
321 }
322 if (!existing) {
323 module.scope->declare(binding.localName, exportedIt->second);
324 }
325 if (import.visibility == Visibility::Public) {
326 module.symbol->exports[binding.localName] = exportedIt->second;
327 }
328 }
329 }

334 if (module.finalImportsApplied) {
335 return;
336 }
337 if (module.valuesPreparationInProgress) {
338 return;
339 }
340
341 module.valuesPreparationInProgress = true;
342
343 for (const auto &import : module.info->imports) {
344 for (const auto &targetId : import.targetModuleIds) {
345 auto targetIt = modules_.find(targetId);
346 if (targetIt != modules_.end()) {

349 }

675 --externTypeContextDepth_;
676
677 auto linkName = extDecl->name_;
678 bool isStdFsModule =
679 module.info->moduleName == "fs" &&
680 module.info->sourceName.find("/std/fs.zp") != std::string::npos;
681 bool isStdIoModule =
682 module.info->moduleName == "io" &&
683 module.info->sourceName.find("/std/io.zp") != std::string::npos;
684 if (isStdFsModule && extDecl->name_ == "mkdir") {
685 linkName = "zap_fs_mkdir";
686 } else if (isStdFsModule && extDecl->name_ == "remove") {
687 linkName = "zap_fs_remove";
688 } else if (isStdFsModule && extDecl->name_ == "rename") {
689 linkName = "zap_fs_rename";
690 } else if (isStdIoModule && extDecl->name_ == "printf") {
691 linkName = "zap_printf";
692 } else if (isStdIoModule && extDecl->name_ == "printfln") {
693 linkName = "zap_printfln";
694 }
695
696 auto symbol = std::make_shared<FunctionSymbol>(
697 extDecl->name_, std::move(params), std::move(retType), linkName,
698 module.info->moduleName, extDecl->visibility_, false,
699 extDecl->isCVariadic_);
700 validateAndApplyFunctionAttributes(*extDecl, symbol, true);
701 if (symbol->hasNoMangle ||
702 (symbol->hasExternC && symbol->externAbi == "C")) {
703 symbol->linkName = symbol->name;
704 }
705 auto existing = module.scope->lookupLocal(extDecl->name_);
706 if (findFunctionBySignature(existing, *symbol)) {
707 error(extDecl->span,
708 "External function '" + extDecl->name_ + "' already declared.");
709 continue;
710 }
711 auto overloads = module.scope->declareFunction(extDecl->name_, symbol);
712 if (!overloads) {
713 error(extDecl->span, "External function '" + extDecl->name_ +
714 "' conflicts with an existing declaration.");
715 continue;
716 }
717 declaredFunctionSymbols_[extDecl] = symbol;
718 module.symbol->members[extDecl->name_] = overloads;
719 if (extDecl->visibility_ == Visibility::Public) {
720 auto exportIt = module.symbol->exports.find(extDecl->name_);
721 std::shared_ptr<OverloadSetSymbol> exportSet;
722 if (exportIt == module.symbol->exports.end()) {
723 exportSet = std::make_shared<OverloadSetSymbol>(
724 extDecl->name_, module.info->moduleName, Visibility::Public);
725 module.symbol->exports[extDecl->name_] = exportSet;
726 } else {
727 exportSet =
728 std::dynamic_pointer_cast<OverloadSetSymbol>(exportIt->second);
729 }
730 if (!exportSet) {
731 error(extDecl->span, "External function '" + extDecl->name_ +
732 "' conflicts with an exported declaration.");
733 } else {
734 exportSet->addOverload(symbol);
735 }
736 }
737 } else if (auto varDecl = dynamic_cast<VarDecl *>(child.get())) {
738 if (!varDecl->isGlobal_) {
739 continue;
740 }
741 auto type = mapType(*varDecl->type_);
742 if (!type) {
743 error(varDecl->span,
744 "Unknown type: " + varDecl->type_->qualifiedName());
745 type = std::make_shared<zir::PrimitiveType>(zir::TypeKind::Void);
746 }
747 auto linkName = varDecl->isExternal_
748 ? varDecl->name_
749 : mangleName(module.info->linkPath.empty()
750 ? module.info->moduleId
751 : module.info->linkPath,
752 varDecl->name_);
753 auto symbol = std::make_shared<VariableSymbol>(
754 varDecl->name_, type, false, false, linkName, module.info->moduleName,
755 varDecl->visibility_);
756 symbol->is_external = varDecl->isExternal_;
757 if (!module.scope->declare(varDecl->name_, symbol)) {
758 error(varDecl->span,
759 "Variable '" + varDecl->name_ + "' already declared.");
760 }
761 module.symbol->members[varDecl->name_] = symbol;
762 if (varDecl->visibility_ == Visibility::Public) {
763 module.symbol->exports[varDecl->name_] = symbol;
764 }
765 } else if (auto constDecl = dynamic_cast<ConstDecl *>(child.get())) {
766 auto type = mapType(*constDecl->type_);
767 if (!type) {
768 error(constDecl->span,
769 "Unknown type: " + constDecl->type_->qualifiedName());
770 type = std::make_shared<zir::PrimitiveType>(zir::TypeKind::Void);
771 }
772 auto symbol = std::make_shared<VariableSymbol>(
773 constDecl->name_, type, true, false,
774 mangleName(module.info->linkPath.empty() ? module.info->moduleId
775 : module.info->linkPath,
776 constDecl->name_),

783 if (constDecl->visibility_ == Visibility::Public) {
784 module.symbol->exports[constDecl->name_] = symbol;

786 }

801void Binder::visit(ClassDecl &node) {
802 auto symbol =
803 std::dynamic_pointer_cast<TypeSymbol>(currentScope_->lookup(node.name_));
804 if (!symbol || !symbol->isClass || !symbol->genericParameterNames.empty()) {
805 return;
806 }
807
808 auto classType = std::static_pointer_cast<zir::ClassType>(symbol->type);
809 auto &classInfo = classInfos_[classType->getName()];
810 currentClassStack_.push_back(classType->getName());
811
812 if (node.baseType_) {
813 bool hasOwnCtor = false;
814 bool hasOwnDtor = false;
815 for (const auto &method : node.methods_) {
816 hasOwnCtor = hasOwnCtor || method->name_ == "init";
817 hasOwnDtor = hasOwnDtor || method->name_ == "deinit";
818 }
819 auto baseType = mapType(*node.baseType_);
820 if (!baseType || baseType->getKind() != zir::TypeKind::Class) {
821 error(node.baseType_->span, "Base type must be a class.");
822 } else {
823 auto baseClass = std::static_pointer_cast<zir::ClassType>(baseType);
824 classType->setBase(baseClass);
825 auto baseIt = classInfos_.find(baseClass->getName());
826 if (baseIt != classInfos_.end()) {
827 if (!hasOwnCtor) {
828 classInfo.constructor = baseIt->second.constructor;
829 }
830 if (!hasOwnDtor) {
831 classInfo.destructor = baseIt->second.destructor;
832 }
833 classInfo.fields = baseIt->second.fields;
834 classInfo.methods.insert(baseIt->second.methods.begin(),
835 baseIt->second.methods.end());
836 for (const auto &field : baseClass->getFields()) {
837 classType->addField(field.name, field.type, field.visibility);
838 }
839 }
840 }
841 }
842
843 for (const auto &field : node.fields_) {
844 auto fieldType = mapType(*field->type);
845 if (!fieldType) {
846 error(field->span, "Unknown type: " + field->type->qualifiedName());
847 fieldType = std::make_shared<zir::PrimitiveType>(zir::TypeKind::Void);
848 }
849 classType->addField(field->name, fieldType,
850 static_cast<int>(field->visibility_));
851 classInfo.fields[field->name] = std::make_shared<VariableSymbol>(
852 field->name, fieldType, false, false, field->name,
853 modules_[currentModuleId_].info->moduleName, field->visibility_);
854 }
855
856 auto boundRecord = std::make_unique<BoundRecordDeclaration>();
857 boundRecord->type = classType;
858 boundRoot_->records.push_back(std::move(boundRecord));
859
860 for (const auto &method : node.methods_) {
861 method->accept(*this);
862 }
863
864 currentClassStack_.pop_back();
865}

867void Binder::visit(FunDecl &node) {
868 auto symbolIt = declaredFunctionSymbols_.find(&node);
869 auto symbol =
870 symbolIt == declaredFunctionSymbols_.end() ? nullptr : symbolIt->second;
871 if (!symbol) {
872 return;
873 }
874
875 if (!symbol->genericParameterNames.empty() &&
876 !symbol->isGenericInstantiation) {
877 return;
878 }
879
880 pushScope();
881 auto oldFunction = currentFunction_;
882 currentFunction_ = symbol;
883 int oldUnsafeDepth = unsafeDepth_;
884 if (node.isUnsafe_) {
885 ++unsafeDepth_;
886 }
887
888 for (const auto &param : symbol->parameters) {
889 if (!currentScope_->declare(param->name, param)) {
890 error(node.span, "Parameter '" + param->name + "' already declared.");
891 }
892 }
893
894 auto boundBody = bindBody(node.body_.get(), false);
895
896 popScope();
897 currentFunction_ = oldFunction;
898 unsafeDepth_ = oldUnsafeDepth;
899
900 bool hasReturn = blockAlwaysReturns(boundBody.get());
901
902 if (!hasReturn && symbol->linkName == "main" &&
903 symbol->returnType->isInteger()) {
904 auto intType = std::make_shared<zir::PrimitiveType>(zir::TypeKind::Int);
905 auto lit = std::make_unique<BoundLiteral>("0", intType);
906 boundBody->statements.push_back(
907 std::make_unique<BoundReturnStatement>(std::move(lit)));
908 hasReturn = true;
909 }
910
911 if (!hasReturn && symbol->returnType->getKind() != zir::TypeKind::Void) {
912 auto kind = symbol->returnType->getKind();
913 if (symbol->returnType->isInteger() || kind == zir::TypeKind::Float ||
914 kind == zir::TypeKind::Bool) {
915 std::string litVal = "0";
916 if (kind == zir::TypeKind::Float)
917 litVal = "0.0";
918 else if (kind == zir::TypeKind::Bool)
919 litVal = "false";
920 auto lit = std::make_unique<BoundLiteral>(litVal, symbol->returnType);
921 boundBody->statements.push_back(
922 std::make_unique<BoundReturnStatement>(std::move(lit)));
923 }
924
925 _diag.report(node.span, zap::DiagnosticLevel::Warning,
926 "Function '" + node.name_ +
927 "' has non-void return type but no return on some paths.");
928 }
929
930 boundRoot_->functions.push_back(
931 std::make_unique<BoundFunctionDeclaration>(symbol, std::move(boundBody)));
932}
