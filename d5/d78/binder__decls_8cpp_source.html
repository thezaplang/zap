292 std::map<std::string, std::string> namespaceOwners;
293
294 for (const auto &import : module.info->imports) {
295 if (!import.moduleAlias.empty() && import.targetModuleIds.size() != 1) {
296 if (!allowIncomplete) {
297 error(import.span, "Module alias imports are only allowed when the "
298 "path resolves to a single module.");
299 }
300 continue;
301 }
302
303 for (const auto &targetId : import.targetModuleIds) {
304 auto targetIt = modules_.find(targetId);
305 if (targetIt == modules_.end()) {
306 continue;
307 }
308
309 auto &target = targetIt->second;
310 auto alias = import.moduleAlias.empty() ? target.info->moduleName
311 : import.moduleAlias;
312
313 auto existingOwner = namespaceOwners.find(alias);
314 if (existingOwner != namespaceOwners.end() &&
315 existingOwner->second != targetId) {
316 if (!allowIncomplete) {
317 error(import.span, "Import namespace '" + alias +
318 "' is ambiguous because multiple files share "
319 "that module name.");
320 }
321 continue;
322 }
323
324 if (!module.scope->lookupLocal(alias)) {
325 module.scope->declare(alias, target.symbol);
326 namespaceOwners[alias] = targetId;
327 } else if (module.scope->lookupLocal(alias) != target.symbol) {
328 if (!allowIncomplete) {
329 error(import.span, "Cannot import module '" + alias +
330 "' because that name is already declared in "
331 "the current file.");
332 }
333 continue;
334 }
335
336 if (import.bindings.empty()) {
337 bool isImplicitStdImport =
338 import.rawPath == "std/prelude" && import.moduleAlias.empty();
339 if (isImplicitStdImport) {
340 for (const auto &exported : target.symbol->exports) {
341 if (!module.scope->lookupLocal(exported.first)) {
342 module.scope->declare(exported.first, exported.second);
343 }
344 }
345 }
346
347 if (import.visibility == Visibility::Public) {
348 module.symbol->exports[alias] = target.symbol;
349 for (const auto &exported : target.symbol->exports) {
350 module.symbol->exports[exported.first] = exported.second;
351 }
352 }
353 continue;
354 }
355
356 if (import.targetModuleIds.size() != 1) {
357 if (!allowIncomplete) {
358 error(import.span, "Selective imports are only allowed when the path "
359 "resolves to a single module.");
360 }
361 continue;
362 }
363
364 for (const auto &binding : import.bindings) {
365 auto exportedIt = target.symbol->exports.find(binding.sourceName);
366 if (exportedIt == target.symbol->exports.end()) {
367 if (allowIncomplete) {
368 continue;
369 }
370 auto memberIt = target.symbol->members.find(binding.sourceName);
371 if (memberIt != target.symbol->members.end() &&
372 memberIt->second->visibility != Visibility::Public) {
373 error(import.span, "Member '" + binding.sourceName +
374 "' of module '" + alias + "' is private.");
375 } else {
376 error(import.span, "Module '" + alias + "' has no public member '" +
377 binding.sourceName + "'.");
378 }
379 continue;
380 }
381
382 auto existing = module.scope->lookupLocal(binding.localName);
383 if (existing && existing != exportedIt->second) {
384 if (!allowIncomplete) {
385 error(import.span, "Imported name '" + binding.localName +
386 "' conflicts with an existing declaration "
387 "in the current file.");
388 }
389 continue;
390 }
391 if (!existing) {
392 module.scope->declare(binding.localName, exportedIt->second);
393 }
394 if (import.visibility == Visibility::Public) {
395 module.symbol->exports[binding.localName] = exportedIt->second;
396 }
397 }
398 }
