60 case OpCode::Sub:

62 break;
63 case OpCode::Mul:
64 opStr = "mul";

66 case OpCode::SDiv:

72 case OpCode::SRem:

74 break;
75 case OpCode::URem:
76 opStr = "urem";
77 break;
78 default:
79 opStr = "binary";
80 break;
81 }
82 return result->getName() + " = " + opStr + " " + lhs->getTypeName() + " " +
83 lhs->getName() + ", " + rhs->getName();
