66 case OpCode::Sub:
67 opStr = "sub";

69 case OpCode::Mul:
70 opStr = "mul";
71 break;
72 case OpCode::SDiv:
73 opStr = "sdiv";

75 case OpCode::UDiv:

78 case OpCode::SRem:
79 opStr = "srem";

81 case OpCode::URem:
82 opStr = "urem";
83 break;
84 case OpCode::Shl:
85 opStr = "shl";
86 break;
87 case OpCode::LShr:
88 opStr = "lshr";
89 break;
90 case OpCode::AShr:
91 opStr = "ashr";
92 break;
93 case OpCode::BitAnd:
94 opStr = "and";
95 break;
96 case OpCode::BitOr:
97 opStr = "or";
98 break;
