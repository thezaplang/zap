34std::string computeLogicalModulePath(const std::filesystem::path &canonicalPath) {
35 auto stdRoot =
36 std::filesystem::weakly_canonical(std::filesystem::path(ZAPC_STDLIB_DIR));
37 auto cwdRoot = std::filesystem::weakly_canonical(std::filesystem::current_path());
38
39 auto buildRelative = [&](const std::filesystem::path &root,
40 const std::string &prefix = "")
41 -> std::optional<std::string> {
42 auto rootText = root.generic_string();
43 auto pathText = canonicalPath.generic_string();
44 if (pathText == rootText || pathText.rfind(rootText + "/", 0) != 0) {
45 return std::nullopt;
46 }
47
48 auto rel = std::filesystem::relative(canonicalPath, root);
49 auto stripped = stripSourceExtension(rel);
50 if (prefix.empty()) {
51 return stripped;
52 }
53 return prefix + "/" + stripped;
54 };
55
56 if (auto logical = buildRelative(stdRoot, "std")) {
57 return *logical;
58 }
59 if (auto logical = buildRelative(cwdRoot)) {
60 return *logical;
61 }
62 return stripSourceExtension(canonicalPath.filename());
63}

65bool readSourceFile(const std::filesystem::path &path, std::string &content) {
66 std::ifstream file(path, std::ios::binary | std::ios::ate);
67 if (!file) {
68 driver::reportError("couldn't open the provided file: ", path,
69 "\nreason: ", strerror(errno));
70 return true;
71 }
72
73 auto size = file.tellg();
74 content.assign(static_cast<size_t>(std::max<std::streamsize>(size, 0)), '\0');
75
76 if (size == 0) {
77 err() << "warning: provided file is empty: " << path << '\n';
78 } else {

86bool resolveImportTargets(const std::filesystem::path &modulePath,
87 const ImportNode &importNode,
88 std::vector<std::filesystem::path> &targets) {
89 std::filesystem::path resolvedPath;
90 if (importNode.path.rfind("std/", 0) == 0) {
91 resolvedPath = std::filesystem::path(ZAPC_STDLIB_DIR) /
92 importNode.path.substr(std::string("std/").size());
93 } else {
94 resolvedPath = modulePath.parent_path() / importNode.path;
95 }
96 resolvedPath = resolvedPath.lexically_normal();
97
98 auto tryFile = [&](const std::filesystem::path &path) -> bool {
99 if (std::filesystem::exists(path) && std::filesystem::is_regular_file(path)) {
100 targets.push_back(std::filesystem::weakly_canonical(path));
101 return true;
102 }
103 return false;
104 };
105
106 if (tryFile(resolvedPath)) {
107 return false;
108 }
109
110 if (resolvedPath.extension() != ".zp" && tryFile(resolvedPath.string() + ".zp")) {
111 return false;
112 }
113
114 if (std::filesystem::exists(resolvedPath) &&
115 std::filesystem::is_directory(resolvedPath)) {
116 for (const auto &entry : std::filesystem::directory_iterator(resolvedPath)) {
117 if (!entry.is_regular_file()) {
118 continue;
119 }
120 if (entry.path().extension() == ".zp") {
121 targets.push_back(std::filesystem::weakly_canonical(entry.path()));
122 }
123 }
124 std::sort(targets.begin(), targets.end());
125 return false;
126 }
127
128 driver::reportError("import path not found: ", resolvedPath);
129 return true;
130}

132bool loadModuleGraph(
133 const std::filesystem::path &entryPath,
134 std::map<std::string, std::unique_ptr<sema::ModuleInfo>> &modules,
135 std::set<std::string> &visiting) {
136 auto canonicalPath = std::filesystem::weakly_canonical(entryPath);
137 auto moduleId = canonicalPath.string();
138
139 if (modules.find(moduleId) != modules.end()) {
140 return false;
141 }
142 if (visiting.count(moduleId)) {
143 driver::reportError("cyclic import detected involving ", canonicalPath);
144 return true;
145 }
146
147 visiting.insert(moduleId);
148
149 std::string source;
150 if (readSourceFile(canonicalPath, source)) {
151 return true;
152 }
153
154 DiagnosticEngine diagnostics(source, canonicalPath.string());
155 Lexer lex(diagnostics);
156 auto tokens = lex.tokenize(source);
157 Parser parser(tokens, diagnostics);
158 auto ast = parser.parse();
159
160 if (diagnostics.hadErrors() || !ast) {
161 diagnostics.printText(err());
162 return true;
163 }
164
165 auto module = std::make_unique<sema::ModuleInfo>();
166 module->moduleId = moduleId;
167 module->moduleName = canonicalPath.stem().string();
168 module->linkPath = computeLogicalModulePath(canonicalPath);
169 module->sourceName = canonicalPath.string();
170 module->root = std::move(ast);
171
172 for (const auto &child : module->root->children) {
173 auto importNode = dynamic_cast<ImportNode *>(child.get());
174 if (!importNode) {
175 continue;
176 }
177
178 sema::ResolvedImport resolved;
179 resolved.rawPath = importNode->path;
180 resolved.moduleAlias = importNode->moduleAlias;
181 resolved.visibility = importNode->visibility_;
182 resolved.span = importNode->span;
183 for (const auto &binding : importNode->bindings) {
184 resolved.bindings.push_back({binding.sourceName, binding.localName});
185 }
186
187 std::vector<std::filesystem::path> importTargets;
188 if (resolveImportTargets(canonicalPath, *importNode, importTargets)) {
189 return true;
190 }
191
192 for (const auto &target : importTargets) {
193 resolved.targetModuleIds.push_back(target.string());
194 }
195
196 module->imports.push_back(std::move(resolved));
197 }
198
199 for (const auto &import : module->imports) {
200 for (const auto &targetId : import.targetModuleIds) {
201 if (loadModuleGraph(targetId, modules, visiting)) {
202 return true;
203 }
204 }
205 }
206
207 visiting.erase(moduleId);
208 modules[moduleId] = std::move(module);
209 return false;
210}
