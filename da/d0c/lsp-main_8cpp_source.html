133std::optional<std::filesystem::path> uriToPath(std::string_view uri) {
134 constexpr std::string_view prefix = "file://";
135 if (uri.rfind(prefix, 0) != 0) {
136 return std::nullopt;
137 }
138
139 std::string decoded;
140 decoded.reserve(uri.size() - prefix.size());
141 for (size_t i = prefix.size(); i < uri.size(); ++i) {
142 char ch = uri[i];
143 if (ch == '%' && i + 2 < uri.size()) {
144 auto hex = [](char value) -> int {
145 if (value >= '0' && value <= '9')
146 return value - '0';
147 if (value >= 'a' && value <= 'f')
148 return 10 + value - 'a';
149 if (value >= 'A' && value <= 'F')
150 return 10 + value - 'A';
151 return -1;
152 };
153 int hi = hex(uri[i + 1]);
154 int lo = hex(uri[i + 2]);
155 if (hi >= 0 && lo >= 0) {
156 decoded.push_back(static_cast<char>((hi << 4) | lo));
157 i += 2;
158 continue;
159 }
160 }
161 decoded.push_back(ch);
162 }
163
164#ifdef _WIN32
165 if (decoded.size() >= 3 && decoded[0] == '/' &&
166 std::isalpha(static_cast<unsigned char>(decoded[1])) &&
167 decoded[2] == ':') {
168 decoded.erase(decoded.begin());
169 }
170#endif
171
172 return std::filesystem::path(decoded).lexically_normal();
173}

936std::optional<CallContext> callContextAtOffset(const std::string &source,
937 size_t offset) {
938 if (source.empty()) {
939 return std::nullopt;
940 }
941
942 size_t pos = std::min(offset, source.size());
943 int nestedParens = 0;
944 int nestedBrackets = 0;
945 int nestedBraces = 0;
946 int64_t activeParameter = 0;
947 bool foundOpenParen = false;
948 size_t openParen = 0;
949
950 while (pos > 0) {
951 --pos;
952 char ch = source[pos];
953 if (ch == ')') {
954 ++nestedParens;
955 } else if (ch == '(') {
956 if (nestedParens == 0 && nestedBrackets == 0 && nestedBraces == 0) {
957 openParen = pos;
958 foundOpenParen = true;
959 break;
960 }
961 if (nestedParens > 0) {
962 --nestedParens;
963 }
964 } else if (ch == ']') {
965 ++nestedBrackets;
966 } else if (ch == '[') {
967 if (nestedBrackets > 0) {
968 --nestedBrackets;
969 }
970 } else if (ch == '}') {
971 ++nestedBraces;
972 } else if (ch == '{') {
973 if (nestedBraces > 0) {
974 --nestedBraces;
975 }
976 } else if (ch == ',' && nestedParens == 0 && nestedBrackets == 0 &&
977 nestedBraces == 0) {
978 ++activeParameter;
979 }
980 }
981
982 if (!foundOpenParen) {
983 return std::nullopt;
984 }
985
986 size_t end = openParen;
987 while (end > 0 && std::isspace(static_cast<unsigned char>(source[end - 1]))) {
988 --end;
989 }
990 size_t start = end;
991 while (start > 0) {
992 char ch = source[start - 1];
993 if (isIdentifierChar(ch) || ch == '.' ||
994 std::isspace(static_cast<unsigned char>(ch))) {
995 --start;
996 continue;
997 }
998 break;
999 }
1000
1001 std::string callee = source.substr(start, end - start);
1002 callee.erase(
1003 std::remove_if(callee.begin(), callee.end(),
1004 [](unsigned char ch) { return std::isspace(ch); }),
1005 callee.end());
1006 if (callee.empty()) {
1007 return std::nullopt;
1008 }
1009
1010 return CallContext{std::move(callee), activeParameter};
1011}

1013std::optional<CallContext> callContextNearOffset(const std::string &source,
1014 size_t offset) {
1015 if (auto call = callContextAtOffset(source, offset)) {
1016 return call;
1017 }
1018
1019 if (source.empty()) {
1020 return std::nullopt;
1021 }
1022
1023 size_t pos = std::min(offset, source.size() - 1);
1024 if (isIdentifierChar(source[pos])) {
1025 size_t end = pos + 1;
1026 while (end < source.size() && isIdentifierChar(source[end])) {
1027 ++end;
1028 }
1029 while (end < source.size() &&
1030 std::isspace(static_cast<unsigned char>(source[end]))) {
1031 ++end;
1032 }
1033 if (end < source.size() && source[end] == '(') {
1034 return callContextAtOffset(source, end + 1);
1035 }
1036 }
1037
1038 if (pos < source.size() && source[pos] == '(') {
1039 return callContextAtOffset(source, pos + 1);
1040 }
1041
1042 return std::nullopt;
1043}

1494 auto uriIt = uriByCanonicalPath_.find(key);
1495 if (uriIt != uriByCanonicalPath_.end()) {

1509 bool loadModuleGraph(
1510 const std::filesystem::path &modulePath,
1511 std::map<std::string, std::unique_ptr<sema::ModuleInfo>> &modules,
1512 std::set<std::string> &visiting, AnalysisResult &result,
1513 const std::string &entryUri, bool allowEntryErrors = false) const {
1514 std::filesystem::path canonicalPath =
1515 std::filesystem::weakly_canonical(modulePath);
1516 std::string moduleId = canonicalPath.string();
1517 std::string entryModuleId =
1518 std::filesystem::weakly_canonical(documentsByUri_.at(entryUri).path)
1519 .string();
1520 bool isEntryModule = moduleId == entryModuleId;
1521 if (modules.find(moduleId) != modules.end()) {
1522 return true;
1523 }
1524 if (visiting.count(moduleId)) {
1525 return false;
1526 }
1527
1528 visiting.insert(moduleId);
1529
1530 auto source = sourceForPath(canonicalPath);
1531 if (!source) {
1532 visiting.erase(moduleId);
1533 return false;
1534 }
1535
1536 zap::DiagnosticEngine diagnostics(*source, canonicalPath.string());
1537 Lexer lex(diagnostics);
1538 auto tokens = lex.tokenize(*source);
1539 zap::Parser parser(tokens, diagnostics);
1540 auto ast = parser.parse();
1541
1542 auto uriIt = uriByCanonicalPath_.find(moduleId);
1543 if (uriIt != uriByCanonicalPath_.end()) {
1544 result.diagnosticsByUri[uriIt->second] = diagnostics.diagnostics();
1545 } else if (moduleId == std::filesystem::weakly_canonical(
1546 documentsByUri_.at(entryUri).path)
1547 .string()) {
1548 result.diagnosticsByUri[entryUri] = diagnostics.diagnostics();
1549 }
1550
1551 if (!ast ||
1552 (diagnostics.hadErrors() && !(allowEntryErrors && isEntryModule))) {
1553 visiting.erase(moduleId);
1554 return false;
1555 }
1556
1557 auto module = std::make_unique<sema::ModuleInfo>();
1558 module->moduleId = moduleId;
1559 module->moduleName = canonicalPath.stem().string();
1560 module->linkPath = computeLogicalModulePath(canonicalPath);
1561 module->sourceName = canonicalPath.string();
1562 module->root = std::move(ast);
1563
1564 for (const auto &child : module->root->children) {
1565 auto importNode = dynamic_cast<ImportNode *>(child.get());
1566 if (!importNode) {
1567 continue;
1568 }
1569
1570 sema::ResolvedImport resolved;
1571 resolved.rawPath = importNode->path;
1572 resolved.moduleAlias = importNode->moduleAlias;
1573 resolved.visibility = importNode->visibility_;
1574 resolved.span = importNode->span;
1575 for (const auto &binding : importNode->bindings) {
1576 resolved.bindings.push_back({binding.sourceName, binding.localName});
1577 }
1578
1579 std::vector<std::filesystem::path> importTargets;
1580 if (!resolveImportTargets(canonicalPath, *importNode, importTargets)) {
1581 continue;
1582 }
1583
1584 for (const auto &target : importTargets) {
1585 resolved.targetModuleIds.push_back(target.string());
1586 }
1587
1588 module->imports.push_back(std::move(resolved));
1589 }
1590
1591 for (const auto &import : module->imports) {
1592 for (const auto &targetId : import.targetModuleIds) {
1593 if (!loadModuleGraph(targetId, modules, visiting, result, entryUri,
1594 allowEntryErrors)) {
1595 visiting.erase(moduleId);
1596 return false;
1597 }
1598 }
1599 }
1600
1601 visiting.erase(moduleId);
1602 modules[moduleId] = std::move(module);
1603 return true;
1604 }

1688 result.diagnosticsByUri[uri] = {};

1693 if (entryModuleIt == project->moduleMap.end()) {
1694 return result;

1697 entryModuleIt->second->isEntry = true;

1700 docIt->second.path.string());

1707 auto flags = findAndReadFlags(docIt->second.path);

1709 sema::Binder binder(diagnostics, flags.allowUnsafe);
1710 auto boundAst = binder.bind(modules);
1711 (void)boundAst;

1713 result.diagnosticsByUri[uri] = diagnostics.diagnostics();
1714 return result;

1755std::vector<LspSymbol> collectImportedSymbols(const ProjectState &project,
1756 const sema::ModuleInfo &module) {
1757 std::vector<LspSymbol> symbols;
1758 for (const auto &import : module.imports) {
1759 for (const auto &targetId : import.targetModuleIds) {
1760 auto moduleIt = project.moduleMap.find(targetId);
1761 if (moduleIt == project.moduleMap.end()) {
1762 continue;
1763 }
1764 const auto &target = *moduleIt->second;
1765 auto alias = effectiveImportAlias(import, target);
1766 auto targetUriIt = project.uriByModuleId.find(targetId);
1767 std::string targetUri = targetUriIt == project.uriByModuleId.end()
1768 ? pathToUri(targetId)
1769 : targetUriIt->second;
1770
1771 symbols.push_back(*makeSymbol(project.uriByModuleId.at(module.moduleId),
1772 alias, import.span, 9, Visibility::Public));
1773
1774 auto exported = collectTopLevelSymbols(target, targetUri, true);
1775 if (import.bindings.empty()) {
1776 continue;
1777 }
1778 for (const auto &binding : import.bindings) {
1779 auto exportIt = std::find_if(exported.begin(), exported.end(),
1780 [&](const LspSymbol &symbol) {
1781 return symbol.name == binding.sourceName;
1782 });
1783 if (exportIt == exported.end()) {
1784 continue;
1785 }
1786 LspSymbol local = *exportIt;
1787 local.name = binding.localName;
1788 symbols.push_back(std::move(local));
1789 }
1790 }
1791 }
1792 return symbols;
1793}

1797 size_t offset) {

1812

1833std::optional<LspSymbol> resolveDefinition(const Workspace &workspace,
1834 const std::string &uri,
1835 const ProjectState &project,
1836 size_t offset) {
1837 const DocumentState *document = workspace.document(uri);
1838 if (!document) {
1839 return std::nullopt;
1840 }
1841 auto path = uriToPath(uri);
1842 if (!path) {
1843 return std::nullopt;
1844 }
1845 std::string moduleId = std::filesystem::weakly_canonical(*path).string();
1846 auto moduleIt = project.moduleMap.find(moduleId);
1847 if (moduleIt == project.moduleMap.end()) {
1848 return std::nullopt;
1849 }
1850 const sema::ModuleInfo &module = *moduleIt->second;
1851
1852 if (auto qualified = qualifiedIdentifierAtOffset(document->text, offset)) {
1853 const auto &[base, rest] = *qualified;
1854 if (!rest.empty()) {
1855 std::string memberName = rest;
1856 size_t dot = memberName.rfind('.');
1857 if (dot != std::string::npos) {
1858 memberName = memberName.substr(dot + 1);
1859 }
1860 for (const auto &import : module.imports) {
1861 for (const auto &targetId : import.targetModuleIds) {
1862 auto targetIt = project.moduleMap.find(targetId);
1863 if (targetIt == project.moduleMap.end()) {
1864 continue;
1865 }
1866 if (effectiveImportAlias(import, *targetIt->second) != base) {
1867 continue;
1868 }
1869 auto targetUriIt = project.uriByModuleId.find(targetId);
1870 std::string targetUri = targetUriIt == project.uriByModuleId.end()
1871 ? pathToUri(targetId)
1872 : targetUriIt->second;
1873 auto exported =
1874 collectTopLevelSymbols(*targetIt->second, targetUri, true);
1875 auto exportIt = std::find_if(exported.begin(), exported.end(),
1876 [&](const LspSymbol &symbol) {
1877 return symbol.name == memberName;
1878 });
1879 if (exportIt != exported.end()) {
1880 return *exportIt;
1881 }
1882 }
1883 }
1884
1885 if (base.find('.') == std::string::npos) {
1886 if (auto classType =
1887 resolveVariableClassType(project, module, offset, base)) {
1888 if (auto cls = resolveClassByTypeName(project, module, *classType)) {
1889 if (auto field = findClassField(cls, project, module, memberName)) {
1890 return *makeSymbol(uri, field->name, field->span, 8,
1891 field->visibility_);
1892 }
1893 if (auto method =
1894 findClassMethod(cls, project, module, memberName)) {
1895 return *makeSymbol(uri, method->name_, method->span, 2,
1896 method->visibility_);
1897 }
1898 }
1899 }
1900 }
1901 }
1902 }
1903
1904 auto name = identifierAt(document->text, offset);
1905 if (!name) {
1906 return std::nullopt;
1907 }
1908
1909 auto locals = collectLocalSymbols(*module.root, offset, uri);
1910 for (auto it = locals.rbegin(); it != locals.rend(); ++it) {
1911 if (it->name == *name) {
1912 return *it;
1913 }
1914 }
1915
1916 auto imported = collectImportedSymbols(project, module);
1917 for (const auto &symbol : imported) {
1918 if (symbol.name == *name) {
1919 return symbol;
1920 }
1921 }
1922
1923 auto topLevel = collectTopLevelSymbols(module, uri);
1924 for (const auto &symbol : topLevel) {
1925 if (symbol.name == *name) {
1926 return symbol;
1927 }
1928 }
1929
1930 return std::nullopt;
1931}
