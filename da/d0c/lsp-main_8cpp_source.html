91 const std::string &prefix = "")
92 -> std::optional<std::string> {
93 auto rootText = root.generic_string();

116std::optional<std::filesystem::path> uriToPath(std::string_view uri) {
117 constexpr std::string_view prefix = "file://";
118 if (uri.rfind(prefix, 0) != 0) {
119 return std::nullopt;
120 }
121
122 std::string decoded;
123 decoded.reserve(uri.size() - prefix.size());
124 for (size_t i = prefix.size(); i < uri.size(); ++i) {
125 char ch = uri[i];
126 if (ch == '%' && i + 2 < uri.size()) {
127 auto hex = [](char value) -> int {
128 if (value >= '0' && value <= '9') return value - '0';
129 if (value >= 'a' && value <= 'f') return 10 + value - 'a';
130 if (value >= 'A' && value <= 'F') return 10 + value - 'A';
131 return -1;
132 };
133 int hi = hex(uri[i + 1]);
134 int lo = hex(uri[i + 2]);
135 if (hi >= 0 && lo >= 0) {
136 decoded.push_back(static_cast<char>((hi << 4) | lo));
137 i += 2;
138 continue;
139 }
140 }
141 decoded.push_back(ch);
142 }
143
144#ifdef _WIN32
145 if (decoded.size() >= 3 && decoded[0] == '/' &&
146 std::isalpha(static_cast<unsigned char>(decoded[1])) &&
147 decoded[2] == ':') {
148 decoded.erase(decoded.begin());
149 }
150#endif
151
152 return std::filesystem::path(decoded).lexically_normal();
153}

170bool resolveImportTargets(const std::filesystem::path &modulePath,
171 const ImportNode &importNode,
172 std::vector<std::filesystem::path> &targets) {
173 std::filesystem::path resolvedPath;
174 if (importNode.path.rfind("std/", 0) == 0) {
175 resolvedPath =
176 stdlibRootPath() / importNode.path.substr(std::string("std/").size());
177 } else {
178 resolvedPath = modulePath.parent_path() / importNode.path;
179 }
180 resolvedPath = resolvedPath.lexically_normal();
181
182 auto tryFile = [&](const std::filesystem::path &path) -> bool {
183 if (std::filesystem::exists(path) && std::filesystem::is_regular_file(path)) {
184 targets.push_back(std::filesystem::weakly_canonical(path));
185 return true;
186 }
187 return false;
188 };
189
190 if (tryFile(resolvedPath)) {
191 return true;
192 }
193 if (resolvedPath.extension() != ".zp" && tryFile(resolvedPath.string() + ".zp")) {
194 return true;
195 }
196 if (std::filesystem::exists(resolvedPath) &&
197 std::filesystem::is_directory(resolvedPath)) {
198 for (const auto &entry : std::filesystem::directory_iterator(resolvedPath)) {
199 if (entry.is_regular_file() && entry.path().extension() == ".zp") {
200 targets.push_back(std::filesystem::weakly_canonical(entry.path()));
201 }
202 }
203 return !targets.empty();
204 }
205 return false;
206}

251std::string pathToUri(const std::filesystem::path &path) {
252 std::string uri = "file://";
253 std::string text = path.generic_string();
254 for (unsigned char ch : text) {
255 bool unreserved = (ch >= 'A' && ch <= 'Z') || (ch >= 'a' && ch <= 'z') ||
256 (ch >= '0' && ch <= '9') || ch == '/' || ch == '-' ||
257 ch == '_' || ch == '.' || ch == '~';
258 if (unreserved) {
259 uri.push_back(static_cast<char>(ch));
260 } else {
261 static constexpr char hex[] = "0123456789ABCDEF";
262 uri.push_back('%');
263 uri.push_back(hex[(ch >> 4) & 0x0F]);
264 uri.push_back(hex[ch & 0x0F]);
265 }
266 }
267 return uri;
268}

270std::optional<std::string> identifierAt(const std::string &source, size_t offset) {
271 if (source.empty()) {
272 return std::nullopt;
273 }
274 size_t pos = offset < source.size() ? offset : source.size() - 1;
275 if (!isIdentifierChar(source[pos])) {
276 if (pos == 0 || !isIdentifierChar(source[pos - 1])) {
277 return std::nullopt;
278 }
279 --pos;
280 }
281
282 size_t start = pos;
283 while (start > 0 && isIdentifierChar(source[start - 1])) {
284 --start;
285 }
286 size_t end = pos + 1;
287 while (end < source.size() && isIdentifierChar(source[end])) {
288 ++end;
289 }
290 return source.substr(start, end - start);

294memberAccessBeforeCursor(const std::string &source, size_t offset) {
295 if (offset == 0 || offset > source.size()) {
296 return std::nullopt;
297 }
298 size_t pos = offset;
299 while (pos > 0 && std::isspace(static_cast<unsigned char>(source[pos - 1]))) {
300 --pos;
301 }
302 if (pos == 0 || source[pos - 1] != '.') {
303 return std::nullopt;
304 }
305
306 size_t end = pos - 1;
307 size_t start = end;
308 while (start > 0 && isIdentifierChar(source[start - 1])) {
309 --start;
310 }
311 if (start == end) {
312 return std::nullopt;
313 }
314 std::string base = source.substr(start, end - start);
315
316 size_t left = start;
317 while (left > 1) {
318 size_t dot = left;
319 while (dot > 0 && std::isspace(static_cast<unsigned char>(source[dot - 1]))) {
320 --dot;
321 }
322 if (dot == 0 || source[dot - 1] != '.') {
323 break;
324 }
325 size_t nameEnd = dot - 1;
326 size_t nameStart = nameEnd;
327 while (nameStart > 0 && isIdentifierChar(source[nameStart - 1])) {
328 --nameStart;
329 }
330 if (nameStart == nameEnd) {
331 break;
332 }
333 base = source.substr(nameStart, end - nameStart);
334 left = nameStart;
335 }
336
337 size_t split = base.find('.');
338 if (split == std::string::npos) {
339 return std::make_pair(base, std::string());
340 }
341 return std::make_pair(base.substr(0, split), base.substr(split + 1));

477void collectLocalsFromNode(const Node *node, size_t offset, const std::string &uri,
478 std::vector<LspSymbol> &symbols) {
479 if (!node) {
480 return;
481 }
482
483 if (auto var = dynamic_cast<const VarDecl *>(node)) {
484 if (!containsOffset(var->span, offset) && var->span.offset > offset) {
485 return;
486 }
487 if (var->span.offset <= offset) {
488 symbols.push_back(*makeSymbol(uri, var->name_, var->span, 6));
489 }
490 return;
491 }
492 if (auto cnst = dynamic_cast<const ConstDecl *>(node)) {
493 if (!containsOffset(cnst->span, offset) && cnst->span.offset > offset) {
494 return;
495 }
496 if (cnst->span.offset <= offset) {
497 symbols.push_back(*makeSymbol(uri, cnst->name_, cnst->span, 21));
498 }
499 return;
500 }
501 if (auto body = dynamic_cast<const BodyNode *>(node)) {
502 collectLocalsInBody(body, offset, uri, symbols);
503 return;
504 }
505 if (auto ifNode = dynamic_cast<const IfNode *>(node)) {
506 if (!containsOffset(ifNode->span, offset)) {
507 return;
508 }
509 if (ifNode->thenBody_ &&
510 (!ifNode->thenBody_->statements.empty() || ifNode->thenBody_->result) &&
511 (ifNode->thenBody_->statements.empty() ||
512 offset >= ifNode->thenBody_->statements.front()->span.offset)) {
513 collectLocalsInBody(ifNode->thenBody_.get(), offset, uri, symbols);
514 } else if (ifNode->elseBody_) {
515 collectLocalsInBody(ifNode->elseBody_.get(), offset, uri, symbols);
516 }
517 return;
518 }
519 if (auto whileNode = dynamic_cast<const WhileNode *>(node)) {
520 if (containsOffset(whileNode->span, offset) && whileNode->body_) {
521 collectLocalsInBody(whileNode->body_.get(), offset, uri, symbols);
522 }
523 return;
524 }

602 bool loadModuleGraph(
603 const std::filesystem::path &modulePath,
604 std::map<std::string, std::unique_ptr<sema::ModuleInfo>> &modules,
605 std::set<std::string> &visiting, AnalysisResult &result,
606 const std::string &entryUri, bool allowEntryErrors = false) const {
607 std::filesystem::path canonicalPath = std::filesystem::weakly_canonical(modulePath);
608 std::string moduleId = canonicalPath.string();
609 std::string entryModuleId =
610 std::filesystem::weakly_canonical(documentsByUri_.at(entryUri).path).string();
611 bool isEntryModule = moduleId == entryModuleId;
612 if (modules.find(moduleId) != modules.end()) {
613 return true;
614 }
615 if (visiting.count(moduleId)) {
616 return false;
617 }
618
619 visiting.insert(moduleId);
620
621 auto source = sourceForPath(canonicalPath);
622 if (!source) {
623 visiting.erase(moduleId);
624 return false;
625 }
626
627 zap::DiagnosticEngine diagnostics(*source, canonicalPath.string());
628 Lexer lex(diagnostics);
629 auto tokens = lex.tokenize(*source);
630 zap::Parser parser(tokens, diagnostics);
631 auto ast = parser.parse();
632
633 auto uriIt = uriByCanonicalPath_.find(moduleId);
634 if (uriIt != uriByCanonicalPath_.end()) {
635 result.diagnosticsByUri[uriIt->second] = diagnostics.diagnostics();
636 } else if (moduleId == std::filesystem::weakly_canonical(
637 documentsByUri_.at(entryUri).path).string()) {
638 result.diagnosticsByUri[entryUri] = diagnostics.diagnostics();
639 }
640
641 if (!ast || (diagnostics.hadErrors() && !(allowEntryErrors && isEntryModule))) {
642 visiting.erase(moduleId);
643 return false;
644 }
645
646 auto module = std::make_unique<sema::ModuleInfo>();
647 module->moduleId = moduleId;
648 module->moduleName = canonicalPath.stem().string();
649 module->linkPath = computeLogicalModulePath(canonicalPath);
650 module->sourceName = canonicalPath.string();
651 module->root = std::move(ast);
652
653 for (const auto &child : module->root->children) {
654 auto importNode = dynamic_cast<ImportNode *>(child.get());
655 if (!importNode) {
656 continue;
657 }
658
659 sema::ResolvedImport resolved;
660 resolved.rawPath = importNode->path;
661 resolved.moduleAlias = importNode->moduleAlias;
662 resolved.visibility = importNode->visibility_;
663 resolved.span = importNode->span;
664 for (const auto &binding : importNode->bindings) {
665 resolved.bindings.push_back({binding.sourceName, binding.localName});
666 }
667
668 std::vector<std::filesystem::path> importTargets;
669 if (!resolveImportTargets(canonicalPath, *importNode, importTargets)) {
670 continue;
671 }
672
673 for (const auto &target : importTargets) {
674 resolved.targetModuleIds.push_back(target.string());
675 }
676
677 module->imports.push_back(std::move(resolved));
678 }
679
680 for (const auto &import : module->imports) {
681 for (const auto &targetId : import.targetModuleIds) {
682 if (!loadModuleGraph(targetId, modules, visiting, result, entryUri,
683 allowEntryErrors)) {
684 visiting.erase(moduleId);
685 return false;
686 }
687 }
688 }
689
690 visiting.erase(moduleId);
691 modules[moduleId] = std::move(module);
692 return true;
693 }

840 if (moduleIt == project.moduleMap.end()) {

847 targetUriIt == project.uriByModuleId.end() ? pathToUri(targetId) : targetUriIt->second;

850 import.span, 9, Visibility::Public));

859 return symbol.name == binding.sourceName;
860 });

862 continue;

864 LspSymbol local = *exportIt;
865 local.name = binding.localName;
866 symbols.push_back(std::move(local));
867 }
868 }

909std::optional<LspSymbol> resolveDefinition(const Workspace &workspace,
910 const std::string &uri,
911 const ProjectState &project,
912 size_t offset) {
913 const DocumentState *document = workspace.document(uri);
914 if (!document) {
915 return std::nullopt;
916 }
917 auto path = uriToPath(uri);
918 if (!path) {
919 return std::nullopt;
920 }
921 std::string moduleId = std::filesystem::weakly_canonical(*path).string();
922 auto moduleIt = project.moduleMap.find(moduleId);
923 if (moduleIt == project.moduleMap.end()) {
924 return std::nullopt;
925 }
926 const sema::ModuleInfo &module = *moduleIt->second;
927
928 if (auto qualified = qualifiedIdentifierAtOffset(document->text, offset)) {
929 const auto &[base, rest] = *qualified;
930 if (!rest.empty()) {
931 std::string memberName = rest;
932 size_t dot = memberName.rfind('.');
933 if (dot != std::string::npos) {
934 memberName = memberName.substr(dot + 1);
935 }
936 for (const auto &import : module.imports) {
937 for (const auto &targetId : import.targetModuleIds) {
938 auto targetIt = project.moduleMap.find(targetId);
939 if (targetIt == project.moduleMap.end()) {
940 continue;
941 }
942 if (effectiveImportAlias(import, *targetIt->second) != base) {
943 continue;
944 }
945 auto targetUriIt = project.uriByModuleId.find(targetId);
946 std::string targetUri = targetUriIt == project.uriByModuleId.end()
947 ? pathToUri(targetId)
948 : targetUriIt->second;
949 auto exported = collectTopLevelSymbols(*targetIt->second, targetUri, true);
950 auto exportIt =
951 std::find_if(exported.begin(), exported.end(), [&](const LspSymbol &symbol) {
952 return symbol.name == memberName;
953 });
954 if (exportIt != exported.end()) {
955 return *exportIt;
956 }
957 }
958 }
959 }
960 }
961
962 auto name = identifierAt(document->text, offset);
963 if (!name) {
964 return std::nullopt;
965 }
966
967 auto locals = collectLocalSymbols(*module.root, offset, uri);
968 for (auto it = locals.rbegin(); it != locals.rend(); ++it) {
969 if (it->name == *name) {
970 return *it;
971 }
972 }
973
974 auto imported = collectImportedSymbols(project, module);
975 for (const auto &symbol : imported) {
976 if (symbol.name == *name) {
977 return symbol;
978 }
979 }
980
981 auto topLevel = collectTopLevelSymbols(module, uri);
982 for (const auto &symbol : topLevel) {
983 if (symbol.name == *name) {
984 return symbol;
985 }
986 }
987
988 return std::nullopt;
989}
