79std::optional<LspSignature> signatureForNode(const Node *node) {
80 auto renderGenericParams =
81 [](const std::vector<std::unique_ptr<TypeNode>> &genericParams) {
82 if (genericParams.empty()) {
83 return std::string();
84 }
85 std::string out = "<";
86 for (size_t i = 0; i < genericParams.size(); ++i) {
87 if (i != 0) {
88 out += ", ";
89 }
90 out += genericParams[i] ? genericParams[i]->qualifiedName() : "?";
91 }
92 out += ">";
93 return out;

95
96 auto renderConstraints =
97 [](const std::vector<GenericConstraint> &constraints) {
98 if (constraints.empty()) {
99 return std::string();
100 }
101 std::string out = " where ";
102 for (size_t i = 0; i < constraints.size(); ++i) {
103 if (i != 0) {
104 out += ", ";
105 }
106 out += constraints[i].parameterName + ": ";
107 out += constraints[i].boundType
108 ? renderType(constraints[i].boundType.get())
109 : std::string("?");
110 }
111 return out;
112 };
113
114 if (auto fun = dynamic_cast<const FunDecl *>(node)) {
115 std::vector<std::string> params;
116 for (const auto &param : fun->params_) {
117 params.push_back(renderParameter(param.get()));
118 }
119 std::string label =
120 fun->name_ + renderGenericParams(fun->genericParams_) + "(";
121 for (size_t i = 0; i < params.size(); ++i) {
122 if (i != 0) {
123 label += ", ";
124 }
125 label += params[i];
126 }
127 label += ") " + renderType(fun->returnType_.get());
128 label += renderConstraints(fun->genericConstraints_);
129 return LspSignature{std::move(label), std::move(params)};
130 }
131 if (auto ext = dynamic_cast<const ExtDecl *>(node)) {
132 std::vector<std::string> params;
133 for (const auto &param : ext->params_) {
134 params.push_back(renderParameter(param.get()));
135 }
136 if (ext->isCVariadic_) {
137 params.push_back("...");

139 std::string label = ext->name_ + "(";
140 for (size_t i = 0; i < params.size(); ++i) {

142 label += ", ";

464std::optional<CallContext> callContextAtOffset(const std::string &source,
465 size_t offset) {
466 if (source.empty()) {
467 return std::nullopt;
468 }
469
470 size_t pos = std::min(offset, source.size());
471 int nestedParens = 0;
472 int nestedBrackets = 0;
473 int nestedBraces = 0;
474 int64_t activeParameter = 0;
475 bool foundOpenParen = false;
476 size_t openParen = 0;
477
478 while (pos > 0) {
479 --pos;
480 char ch = source[pos];
481 if (ch == ')') {
482 ++nestedParens;
483 } else if (ch == '(') {
484 if (nestedParens == 0 && nestedBrackets == 0 && nestedBraces == 0) {
485 openParen = pos;
486 foundOpenParen = true;
487 break;
488 }
489 if (nestedParens > 0) {
490 --nestedParens;
491 }
492 } else if (ch == ']') {
493 ++nestedBrackets;
494 } else if (ch == '[') {
495 if (nestedBrackets > 0) {
496 --nestedBrackets;
497 }
498 } else if (ch == '}') {
499 ++nestedBraces;
500 } else if (ch == '{') {
501 if (nestedBraces > 0) {
502 --nestedBraces;
503 }
504 } else if (ch == ',' && nestedParens == 0 && nestedBrackets == 0 &&
505 nestedBraces == 0) {
506 ++activeParameter;
507 }
508 }
509
510 if (!foundOpenParen) {
511 return std::nullopt;
512 }
513
514 size_t end = openParen;
515 while (end > 0 && std::isspace(static_cast<unsigned char>(source[end - 1]))) {
516 --end;
517 }
518 size_t start = end;
519 while (start > 0) {
520 char ch = source[start - 1];
521 if (isIdentifierChar(ch) || ch == '.' ||
522 std::isspace(static_cast<unsigned char>(ch))) {
523 --start;
524 continue;
525 }
526 break;
527 }
528
529 std::string callee = source.substr(start, end - start);
530 callee.erase(
531 std::remove_if(callee.begin(), callee.end(),
532 [](unsigned char ch) { return std::isspace(ch); }),
533 callee.end());
534 if (callee.empty()) {
535 return std::nullopt;
536 }
537
538 return CallContext{std::move(callee), activeParameter};
539}

541std::optional<CallContext> callContextNearOffset(const std::string &source,
542 size_t offset) {
543 if (auto call = callContextAtOffset(source, offset)) {
544 return call;
545 }
546
547 if (source.empty()) {
548 return std::nullopt;
549 }
550
551 size_t pos = std::min(offset, source.size() - 1);
552 if (isIdentifierChar(source[pos])) {
553 size_t end = pos + 1;
554 while (end < source.size() && isIdentifierChar(source[end])) {
555 ++end;
556 }
557 while (end < source.size() &&
558 std::isspace(static_cast<unsigned char>(source[end]))) {
559 ++end;
560 }
561 if (end < source.size() && source[end] == '(') {
562 return callContextAtOffset(source, end + 1);
563 }
564 }
565
566 if (pos < source.size() && source[pos] == '(') {
567 return callContextAtOffset(source, pos + 1);
568 }
569
570 return std::nullopt;
571}
