116std::optional<std::filesystem::path> uriToPath(std::string_view uri) {
117 constexpr std::string_view prefix = "file://";
118 if (uri.rfind(prefix, 0) != 0) {
119 return std::nullopt;
120 }
121
122 std::string decoded;
123 decoded.reserve(uri.size() - prefix.size());
124 for (size_t i = prefix.size(); i < uri.size(); ++i) {
125 char ch = uri[i];
126 if (ch == '%' && i + 2 < uri.size()) {
127 auto hex = [](char value) -> int {
128 if (value >= '0' && value <= '9') return value - '0';
129 if (value >= 'a' && value <= 'f') return 10 + value - 'a';
130 if (value >= 'A' && value <= 'F') return 10 + value - 'A';
131 return -1;
132 };
133 int hi = hex(uri[i + 1]);
134 int lo = hex(uri[i + 2]);
135 if (hi >= 0 && lo >= 0) {
136 decoded.push_back(static_cast<char>((hi << 4) | lo));
137 i += 2;
138 continue;
139 }
140 }
141 decoded.push_back(ch);
142 }
143
144#ifdef _WIN32
145 if (decoded.size() >= 3 && decoded[0] == '/' &&
146 std::isalpha(static_cast<unsigned char>(decoded[1])) &&
147 decoded[2] == ':') {
148 decoded.erase(decoded.begin());
149 }
150#endif
151
152 return std::filesystem::path(decoded).lexically_normal();
153}

565std::optional<CallContext> callContextAtOffset(const std::string &source, size_t offset) {
566 if (source.empty()) {
567 return std::nullopt;
568 }
569
570 size_t pos = std::min(offset, source.size());
571 int nestedParens = 0;
572 int nestedBrackets = 0;
573 int nestedBraces = 0;
574 int64_t activeParameter = 0;
575 bool foundOpenParen = false;
576 size_t openParen = 0;
577
578 while (pos > 0) {
579 --pos;
580 char ch = source[pos];
581 if (ch == ')') {
582 ++nestedParens;
583 } else if (ch == '(') {
584 if (nestedParens == 0 && nestedBrackets == 0 && nestedBraces == 0) {
585 openParen = pos;
586 foundOpenParen = true;
587 break;
588 }
589 if (nestedParens > 0) {
590 --nestedParens;
591 }
592 } else if (ch == ']') {
593 ++nestedBrackets;
594 } else if (ch == '[') {
595 if (nestedBrackets > 0) {
596 --nestedBrackets;
597 }
598 } else if (ch == '}') {
599 ++nestedBraces;
600 } else if (ch == '{') {
601 if (nestedBraces > 0) {
602 --nestedBraces;
603 }
604 } else if (ch == ',' && nestedParens == 0 && nestedBrackets == 0 && nestedBraces == 0) {
605 ++activeParameter;
606 }
607 }
608
609 if (!foundOpenParen) {
610 return std::nullopt;
611 }
612
613 size_t end = openParen;
614 while (end > 0 && std::isspace(static_cast<unsigned char>(source[end - 1]))) {
615 --end;
616 }
617 size_t start = end;
618 while (start > 0) {
619 char ch = source[start - 1];
620 if (isIdentifierChar(ch) || ch == '.' ||
621 std::isspace(static_cast<unsigned char>(ch))) {
622 --start;
623 continue;
624 }
625 break;
626 }
627
628 std::string callee = source.substr(start, end - start);
629 callee.erase(std::remove_if(callee.begin(), callee.end(),
630 [](unsigned char ch) { return std::isspace(ch); }),
631 callee.end());
632 if (callee.empty()) {
633 return std::nullopt;
634 }
635
636 return CallContext{std::move(callee), activeParameter};
637}

817std::optional<HoverInfo> resolveHover(const std::string &source,
818 const std::string &uri,
819 const ProjectState &project,
820 size_t offset) {
821 auto path = uriToPath(uri);
822 if (!path) {
823 return std::nullopt;
824 }
825 std::string moduleId = std::filesystem::weakly_canonical(*path).string();
826 auto moduleIt = project.moduleMap.find(moduleId);
827 if (moduleIt == project.moduleMap.end()) {
828 return std::nullopt;
829 }
830 const sema::ModuleInfo &module = *moduleIt->second;
831
832 if (auto qualified = qualifiedIdentifierAtOffset(source, offset)) {
833 const auto &[base, rest] = *qualified;
834 if (!rest.empty()) {
835 std::string memberName = rest;
836 size_t dot = memberName.rfind('.');
837 if (dot != std::string::npos) {
838 memberName = memberName.substr(dot + 1);
839 }
840 for (const auto &import : module.imports) {
841 for (const auto &targetId : import.targetModuleIds) {
842 auto targetIt = project.moduleMap.find(targetId);
843 if (targetIt == project.moduleMap.end()) {
844 continue;
845 }
846 if (effectiveImportAlias(import, *targetIt->second) != base) {
847 continue;
848 }
849 if (auto hover = findTopLevelHover(*targetIt->second, memberName, true)) {
850 return hover;
851 }
852 }
853 }
854 }
855 }
856
857 auto name = identifierAt(source, offset);
858 if (!name) {
859 return std::nullopt;
860 }
861
862 if (auto hover = findTopLevelHover(module, *name)) {
863 return hover;
864 }
865
866 for (const auto &import : module.imports) {
867 for (const auto &binding : import.bindings) {
868 if (binding.localName != *name) {
869 continue;
870 }
871 for (const auto &targetId : import.targetModuleIds) {
872 auto targetIt = project.moduleMap.find(targetId);
873 if (targetIt == project.moduleMap.end()) {
874 continue;
875 }
876 if (auto hover = findTopLevelHover(*targetIt->second, binding.sourceName, true)) {
877 return hover;
878 }
879 }
880 }
881 }
882
883 return std::nullopt;
884}

966void collectLocalsFromNode(const Node *node, size_t offset, const std::string &uri,
967 std::vector<LspSymbol> &symbols) {
968 if (!node) {
969 return;
970 }
971
972 if (auto var = dynamic_cast<const VarDecl *>(node)) {
973 if (!containsOffset(var->span, offset) && var->span.offset > offset) {
974 return;
975 }
976 if (var->span.offset <= offset) {
977 symbols.push_back(*makeSymbol(uri, var->name_, var->span, 6));
978 }
979 return;
980 }
981 if (auto cnst = dynamic_cast<const ConstDecl *>(node)) {
982 if (!containsOffset(cnst->span, offset) && cnst->span.offset > offset) {
983 return;
984 }
985 if (cnst->span.offset <= offset) {
986 symbols.push_back(*makeSymbol(uri, cnst->name_, cnst->span, 21));
987 }
988 return;
989 }
990 if (auto body = dynamic_cast<const BodyNode *>(node)) {
991 collectLocalsInBody(body, offset, uri, symbols);
992 return;
993 }
994 if (auto ifNode = dynamic_cast<const IfNode *>(node)) {
995 if (!containsOffset(ifNode->span, offset)) {
996 return;
997 }
998 if (ifNode->thenBody_ &&
999 (!ifNode->thenBody_->statements.empty() || ifNode->thenBody_->result) &&
1000 (ifNode->thenBody_->statements.empty() ||
1001 offset >= ifNode->thenBody_->statements.front()->span.offset)) {
1002 collectLocalsInBody(ifNode->thenBody_.get(), offset, uri, symbols);
1003 } else if (ifNode->elseBody_) {
1004 collectLocalsInBody(ifNode->elseBody_.get(), offset, uri, symbols);
1005 }
1006 return;
1007 }
1008 if (auto whileNode = dynamic_cast<const WhileNode *>(node)) {
1009 if (containsOffset(whileNode->span, offset) && whileNode->body_) {
1010 collectLocalsInBody(whileNode->body_.get(), offset, uri, symbols);
1011 }
1012 return;
1013 }
1014}

1025 symbols.push_back(*makeSymbol(uri, cnst->name_, cnst->span, 21));

1091 bool loadModuleGraph(
1092 const std::filesystem::path &modulePath,
1093 std::map<std::string, std::unique_ptr<sema::ModuleInfo>> &modules,
1094 std::set<std::string> &visiting, AnalysisResult &result,
1095 const std::string &entryUri, bool allowEntryErrors = false) const {
1096 std::filesystem::path canonicalPath = std::filesystem::weakly_canonical(modulePath);
1097 std::string moduleId = canonicalPath.string();
1098 std::string entryModuleId =
1099 std::filesystem::weakly_canonical(documentsByUri_.at(entryUri).path).string();
1100 bool isEntryModule = moduleId == entryModuleId;
1101 if (modules.find(moduleId) != modules.end()) {
1102 return true;
1103 }
1104 if (visiting.count(moduleId)) {
1105 return false;
1106 }
1107
1108 visiting.insert(moduleId);
1109
1110 auto source = sourceForPath(canonicalPath);
1111 if (!source) {
1112 visiting.erase(moduleId);
1113 return false;
1114 }
1115
1116 zap::DiagnosticEngine diagnostics(*source, canonicalPath.string());
1117 Lexer lex(diagnostics);
1118 auto tokens = lex.tokenize(*source);
1119 zap::Parser parser(tokens, diagnostics);
1120 auto ast = parser.parse();
1121
1122 auto uriIt = uriByCanonicalPath_.find(moduleId);
1123 if (uriIt != uriByCanonicalPath_.end()) {
1124 result.diagnosticsByUri[uriIt->second] = diagnostics.diagnostics();
1125 } else if (moduleId == std::filesystem::weakly_canonical(
1126 documentsByUri_.at(entryUri).path).string()) {
1127 result.diagnosticsByUri[entryUri] = diagnostics.diagnostics();
1128 }
1129
1130 if (!ast || (diagnostics.hadErrors() && !(allowEntryErrors && isEntryModule))) {
1131 visiting.erase(moduleId);
1132 return false;
1133 }
1134
1135 auto module = std::make_unique<sema::ModuleInfo>();
1136 module->moduleId = moduleId;
1137 module->moduleName = canonicalPath.stem().string();
1138 module->linkPath = computeLogicalModulePath(canonicalPath);
1139 module->sourceName = canonicalPath.string();
1140 module->root = std::move(ast);
1141
1142 for (const auto &child : module->root->children) {
1143 auto importNode = dynamic_cast<ImportNode *>(child.get());
1144 if (!importNode) {
1145 continue;
1146 }
1147
1148 sema::ResolvedImport resolved;
1149 resolved.rawPath = importNode->path;
1150 resolved.moduleAlias = importNode->moduleAlias;
1151 resolved.visibility = importNode->visibility_;
1152 resolved.span = importNode->span;
1153 for (const auto &binding : importNode->bindings) {
1154 resolved.bindings.push_back({binding.sourceName, binding.localName});
1155 }
1156
1157 std::vector<std::filesystem::path> importTargets;
1158 if (!resolveImportTargets(canonicalPath, *importNode, importTargets)) {
1159 continue;
1160 }
1161
1162 for (const auto &target : importTargets) {
1163 resolved.targetModuleIds.push_back(target.string());
1164 }
1165
1166 module->imports.push_back(std::move(resolved));
1167 }
1168
1169 for (const auto &import : module->imports) {
1170 for (const auto &targetId : import.targetModuleIds) {
1171 if (!loadModuleGraph(targetId, modules, visiting, result, entryUri,
1172 allowEntryErrors)) {
1173 visiting.erase(moduleId);
1174 return false;
1175 }
1176 }
1177 }
1178
1179 visiting.erase(moduleId);
1180 modules[moduleId] = std::move(module);
1181 return true;
1182 }
