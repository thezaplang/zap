10uint64_t parseIntegerLiteral(const std::string &literal) {
11 if (literal.size() > 2 && literal[0] == '0') {
12 if (literal[1] == 'x' || literal[1] == 'X') {
13 return std::stoull(literal, nullptr, 16);
14 }
15 if (literal[1] == 'b' || literal[1] == 'B') {
16 return std::stoull(literal.substr(2), nullptr, 2);
17 }
18 if (literal[1] == 'o' || literal[1] == 'O') {
19 return std::stoull(literal.substr(2), nullptr, 8);
20 }
21 }
22 return std::stoull(literal, nullptr, 10);
23}

54llvm::Constant *LLVMCodeGen::lowerZIRConstant(const zir::Constant &constant) {
55 if (constant.getType()->getKind() == zir::TypeKind::Record) {
56 const auto &rt = static_cast<const zir::RecordType &>(*constant.getType());
57 if (isStringRecordName(rt.getName())) {
58 std::string gname;
59 auto *ptrConst = getOrCreateGlobalString(constant.getLiteral(), gname);
60 auto *lenConst = llvm::ConstantInt::get(
61 llvm::Type::getInt64Ty(ctx_),
62 static_cast<uint64_t>(constant.getLiteral().size()));
63 auto *structTy =
64 static_cast<llvm::StructType *>(toLLVMType(*constant.getType()));
65 return llvm::ConstantStruct::get(structTy, {ptrConst, lenConst});
66 }
67 }
68
69 auto *ty = toLLVMType(*constant.getType());
70 const auto &literal = constant.getLiteral();
71 if (ty->isIntegerTy(1)) {
72 return llvm::ConstantInt::get(ty, literal == "true" ? 1 : 0);
73 }
74 if (ty->isIntegerTy(8) &&
75 constant.getType()->getKind() == zir::TypeKind::Char) {
76 if (literal == "null") {
77 return llvm::ConstantInt::get(ty, 0, false);
78 }
79 int64_t code = 0;
80 if (!literal.empty()) {
81 if (literal.size() >= 2 && literal[0] == '\\') {
82 switch (literal[1]) {
83 case 'n':
84 code = '\n';
85 break;
86 case 't':
87 code = '\t';
88 break;
89 case 'r':
90 code = '\r';
91 break;
92 case '\\':
93 code = '\\';
94 break;
95 case '\'':
96 code = '\'';
97 break;
98 case '0':
99 code = '\0';
100 break;
101 default:
102 code = static_cast<unsigned char>(literal[1]);
103 break;
104 }
105 } else {
106 code = static_cast<unsigned char>(literal[0]);
107 }
108 }
109 return llvm::ConstantInt::get(ty, code, false);
110 }
111 if (ty->isIntegerTy()) {
112 if (literal == "null") {
113 return llvm::ConstantInt::get(ty, 0, false);
114 }
115 uint64_t unsignedValue = parseIntegerLiteral(literal);
116 if (constant.getType()->isUnsigned()) {
117 return llvm::ConstantInt::get(ty, unsignedValue, false);
118 }
119
120 return llvm::ConstantInt::get(ty, unsignedValue, true);
121 }
122 if (ty->isFloatTy()) {
123 return llvm::ConstantFP::get(ty, std::stof(literal));
124 }
125 if (ty->isDoubleTy()) {
126 return llvm::ConstantFP::get(ty, std::stod(literal));
127 }
128 if (ty->isPointerTy()) {
129 if (literal == "null") {
130 return llvm::ConstantPointerNull::get(llvm::cast<llvm::PointerType>(ty));
131 }
132 auto *address =
133 llvm::ConstantInt::get(module_->getDataLayout().getIntPtrType(ctx_),
134 parseIntegerLiteral(literal), false);
135 return llvm::ConstantExpr::getIntToPtr(address, ty);
136 }
137 return llvm::Constant::getNullValue(ty);

222 }
223 }

226 if (!elems[i]) {

236LLVMCodeGen::lowerZIRArrayConstant(const zir::ArrayConstant &constant) {
237 auto *ty = toLLVMType(*constant.getType());
238 auto *arrayTy = llvm::dyn_cast<llvm::ArrayType>(ty);
239 if (!arrayTy) {
240 return llvm::Constant::getNullValue(ty);
241 }
242
243 std::vector<llvm::Constant *> elems;
244 elems.reserve(arrayTy->getNumElements());
245 for (const auto &element : constant.getElements()) {
246 llvm::Constant *elementConst = nullptr;
247 if (element->getKind() == zir::ValueKind::Constant) {
248 elementConst =
249 lowerZIRConstant(static_cast<const zir::Constant &>(*element));
250 } else if (element->getKind() == zir::ValueKind::AggregateConstant) {
251 elementConst = lowerZIRAggregateConstant(
252 static_cast<const zir::AggregateConstant &>(*element));
253 } else if (element->getKind() == zir::ValueKind::ArrayConstant) {
254 elementConst = lowerZIRArrayConstant(
255 static_cast<const zir::ArrayConstant &>(*element));
256 }
257
258 if (elementConst) {
259 auto *expectedTy = arrayTy->getElementType();
260 if (elementConst->getType() != expectedTy) {
261 if (elementConst->getType()->isIntegerTy() &&
262 expectedTy->isIntegerTy()) {
263 auto srcBits = elementConst->getType()->getIntegerBitWidth();
264 auto dstBits = expectedTy->getIntegerBitWidth();
265 if (dstBits < srcBits) {
266 elementConst = llvm::cast<llvm::Constant>(
267 llvm::ConstantExpr::getTrunc(elementConst, expectedTy));
268 } else if (dstBits > srcBits) {
269 elementConst =
270 llvm::cast<llvm::Constant>(llvm::ConstantExpr::getCast(
271 llvm::Instruction::SExt, elementConst, expectedTy));
272 }
273 } else {
274 elementConst = llvm::cast<llvm::Constant>(
275 llvm::ConstantExpr::getBitCast(elementConst, expectedTy));
276 }
277 }
278 }
279
280 elems.push_back(
281 elementConst ? elementConst
282 : llvm::Constant::getNullValue(arrayTy->getElementType()));
283 }
284

286 elems.push_back(llvm::Constant::getNullValue(arrayTy->getElementType()));
287 }
288
289 return llvm::ConstantArray::get(arrayTy, elems);
290}
