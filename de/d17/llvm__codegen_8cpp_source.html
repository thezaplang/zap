22 if (literal.size() > 2 && literal[0] == '0') {

194 if (optimization_level < 0) {

197 optimization_level = 3;

201 if (optimization_level == 1) {

203 } else if (optimization_level == 2) {

207 }

219 return false;

432llvm::Constant *LLVMCodeGen::lowerZIRConstant(const zir::Constant &constant) {
433 if (constant.getType()->getKind() == zir::TypeKind::Record) {
434 const auto &rt =
435 static_cast<const zir::RecordType &>(*constant.getType());
436 if (rt.getName() == "String") {
437 std::string gname;
438 auto *ptrConst = getOrCreateGlobalString(constant.getLiteral(), gname);
439 auto *lenConst = llvm::ConstantInt::get(
440 llvm::Type::getInt64Ty(ctx_),
441 static_cast<uint64_t>(constant.getLiteral().size()));
442 auto *structTy =
443 static_cast<llvm::StructType *>(toLLVMType(*constant.getType()));
444 return llvm::ConstantStruct::get(structTy, {ptrConst, lenConst});
445 }
446 }
447
448 auto *ty = toLLVMType(*constant.getType());
449 const auto &literal = constant.getLiteral();
450 if (ty->isIntegerTy(1)) {
451 return llvm::ConstantInt::get(ty, literal == "true" ? 1 : 0);
452 }
453 if (ty->isIntegerTy(8) && constant.getType()->getKind() == zir::TypeKind::Char) {
454 if (literal == "null") {
455 return llvm::ConstantInt::get(ty, 0, false);
456 }
457 int64_t code = 0;
458 if (!literal.empty()) {
459 if (literal.size() >= 2 && literal[0] == '\\') {
460 switch (literal[1]) {
461 case 'n':
462 code = '\n';
463 break;
464 case 't':
465 code = '\t';
466 break;
467 case 'r':
468 code = '\r';
469 break;
470 case '\\':
471 code = '\\';
472 break;
473 case '\'':
474 code = '\'';
475 break;
476 case '0':
477 code = '\0';
478 break;
479 default:
480 code = static_cast<unsigned char>(literal[1]);
481 break;
482 }
483 } else {
484 code = static_cast<unsigned char>(literal[0]);
485 }
486 }
487 return llvm::ConstantInt::get(ty, code, false);
488 }
489 if (ty->isIntegerTy()) {
490 if (literal == "null") {
491 return llvm::ConstantInt::get(ty, 0, false);
492 }
493 uint64_t unsignedValue = parseIntegerLiteral(literal);
494 if (constant.getType()->isUnsigned()) {
495 return llvm::ConstantInt::get(ty, unsignedValue, false);
496 }
497
498 return llvm::ConstantInt::get(ty, unsignedValue, true);
499 }
500 if (ty->isFloatTy()) {
501 return llvm::ConstantFP::get(ty, std::stof(literal));
502 }
503 if (ty->isDoubleTy()) {
504 return llvm::ConstantFP::get(ty, std::stod(literal));
505 }
506 if (ty->isPointerTy()) {
507 return llvm::ConstantPointerNull::get(
508 llvm::cast<llvm::PointerType>(ty));
509 }
510 return llvm::Constant::getNullValue(ty);
511}

513llvm::Constant *
514LLVMCodeGen::lowerZIRAggregateConstant(const zir::AggregateConstant &constant) {
515 auto *ty = toLLVMType(*constant.getType());
516
517 auto *structTy = llvm::dyn_cast<llvm::StructType>(ty);
518 if (!structTy) {
519 return llvm::Constant::getNullValue(ty);
520 }
521
522 const auto *recordTy =
523 dynamic_cast<const zir::RecordType *>(constant.getType().get());
524 if (!recordTy) {
525 return llvm::Constant::getNullValue(ty);
526 }
527
528 const auto &recordFields = recordTy->getFields();
529 std::vector<llvm::Constant *> elems(recordFields.size(), nullptr);
530
531 for (const auto &fieldInit : constant.getFields()) {
532 int fieldIndex = -1;
533 for (size_t i = 0; i < recordFields.size(); ++i) {
534 if (recordFields[i].name == fieldInit.name) {
535 fieldIndex = static_cast<int>(i);
536 break;
537 }
538 }
539
540 if (fieldIndex < 0 || fieldInit.value == nullptr) {
541 continue;
542 }
543
544 llvm::Constant *fieldConst = nullptr;
545 if (fieldInit.value->getKind() == zir::ValueKind::Constant) {
546 fieldConst = lowerZIRConstant(
547 static_cast<const zir::Constant &>(*fieldInit.value));
548 } else if (fieldInit.value->getKind() == zir::ValueKind::AggregateConstant) {
549 fieldConst = lowerZIRAggregateConstant(
550 static_cast<const zir::AggregateConstant &>(*fieldInit.value));
551 }
552
553 if (fieldConst) {
554 auto *expectedFieldTy = toLLVMType(*recordFields[static_cast<size_t>(fieldIndex)].type);
555 if (fieldConst->getType() != expectedFieldTy) {
556 fieldConst = llvm::dyn_cast<llvm::Constant>(
557 llvm::ConstantExpr::getBitCast(fieldConst, expectedFieldTy));
558 }
559 elems[static_cast<size_t>(fieldIndex)] = fieldConst;
560 }
561 }
562
563 for (size_t i = 0; i < elems.size(); ++i) {
564 if (!elems[i]) {
565 elems[i] = llvm::Constant::getNullValue(toLLVMType(*recordFields[i].type));
566 }
567 }
568
569 return llvm::ConstantStruct::get(structTy, elems);
570}

641 return src;
642 }
643 if (srcTy->isPointerTy() && destTy->isPointerTy()) {
644 return builder_.CreateBitCast(src, destTy);
645 }
646 if (srcTy->isPointerTy() && destTy->isIntegerTy()) {
647 return builder_.CreatePtrToInt(src, destTy);
648 }
649 if (srcTy->isIntegerTy() && destTy->isPointerTy()) {
650 return builder_.CreateIntToPtr(src, destTy);
651 }
652 if (srcTy->isIntegerTy() && destTy->isIntegerTy()) {
653 unsigned srcBits = srcTy->getIntegerBitWidth();
654 unsigned destBits = destTy->getIntegerBitWidth();
655 if (destBits > srcBits) {
656 bool shouldZeroExtend =
657 sourceType->isUnsigned() ||
658 sourceType->getKind() == zir::TypeKind::Bool ||
659 sourceType->getKind() == zir::TypeKind::Char;
660 return shouldZeroExtend ? builder_.CreateZExt(src, destTy)
661 : builder_.CreateSExt(src, destTy);
662 }
663 if (destBits < srcBits) {
664 return builder_.CreateTrunc(src, destTy);
665 }
666 return src;
667 }
668 if (srcTy->isIntegerTy() && destTy->isFloatingPointTy()) {
669 return sourceType->isUnsigned() ? builder_.CreateUIToFP(src, destTy)
670 : builder_.CreateSIToFP(src, destTy);
671 }
672 if (srcTy->isFloatingPointTy() && destTy->isIntegerTy()) {
673 return targetType->isUnsigned() ? builder_.CreateFPToUI(src, destTy)
674 : builder_.CreateFPToSI(src, destTy);
675 }
676 if (srcTy->isFloatingPointTy() && destTy->isFloatingPointTy()) {
677 if (srcTy->getPrimitiveSizeInBits() < destTy->getPrimitiveSizeInBits()) {
678 return builder_.CreateFPExt(src, destTy);
679 }
680 if (srcTy->getPrimitiveSizeInBits() > destTy->getPrimitiveSizeInBits()) {
681 return builder_.CreateFPTrunc(src, destTy);
682 }
683 return src;
684 }
685 return builder_.CreateBitCast(src, destTy);
686}

688llvm::Value *LLVMCodeGen::emitStringConcat(
689 llvm::Value *lhs, llvm::Value *rhs, const std::shared_ptr<zir::Type> &lhsType,
690 const std::shared_ptr<zir::Type> &rhsType,
691 const std::shared_ptr<zir::Type> &resultType) {
692 auto *i8Ty = llvm::Type::getInt8Ty(ctx_);
693 auto *i64Ty = llvm::Type::getInt64Ty(ctx_);
694
695 llvm::Value *lhsPtr = nullptr;
696 llvm::Value *lhsLen = nullptr;
697 llvm::Value *rhsPtr = nullptr;
698 llvm::Value *rhsLen = nullptr;
699
700 if (isStringType(lhsType)) {
701 lhsPtr = builder_.CreateExtractValue(lhs, {0});
702 lhsLen = builder_.CreateExtractValue(lhs, {1});
703 } else if (lhsType && lhsType->getKind() == zir::TypeKind::Char) {
704 auto *buf = createEntryAlloca(currentFn_, "zir_char_buf_l", i8Ty);
705 builder_.CreateStore(lhs, buf);
706 lhsPtr = buf;
707 lhsLen = llvm::ConstantInt::get(i64Ty, 1);
708 }
709
710 if (isStringType(rhsType)) {
711 rhsPtr = builder_.CreateExtractValue(rhs, {0});
712 rhsLen = builder_.CreateExtractValue(rhs, {1});
713 } else if (rhsType && rhsType->getKind() == zir::TypeKind::Char) {
714 auto *buf = createEntryAlloca(currentFn_, "zir_char_buf_r", i8Ty);
715 builder_.CreateStore(rhs, buf);
716 rhsPtr = buf;
717 rhsLen = llvm::ConstantInt::get(i64Ty, 1);
718 }
719
720 if (functionMap_.count("string_concat_ptrlen") == 0) {
721 std::vector<llvm::Type *> params = {
722 llvm::PointerType::getUnqual(i8Ty), i64Ty,
723 llvm::PointerType::getUnqual(i8Ty), i64Ty};
724 auto *ft = llvm::FunctionType::get(llvm::PointerType::getUnqual(i8Ty),
725 params, false);
726 auto *fn = llvm::Function::Create(ft, llvm::Function::ExternalLinkage,
727 "string_concat_ptrlen", *module_);
728 functionMap_["string_concat_ptrlen"] = fn;
729 }
730
731 auto *concatFn = functionMap_.at("string_concat_ptrlen");
732 auto *call = builder_.CreateCall(concatFn, {lhsPtr, lhsLen, rhsPtr, rhsLen});
733 auto *sumLen = builder_.CreateAdd(lhsLen, rhsLen);
734
735 auto *structTy = static_cast<llvm::StructType *>(toLLVMType(*resultType));
736 llvm::Value *res = llvm::UndefValue::get(structTy);
737 res = builder_.CreateInsertValue(res, call, {0});
738 res = builder_.CreateInsertValue(res, sumLen, {1});
739 return res;
740}
741
742void LLVMCodeGen::emitZIRInstruction(const zir::Instruction &inst) {
743 using namespace zir;
744
745 switch (inst.getOpCode()) {
746 case OpCode::Alloca: {
747 const auto &allocaInst = static_cast<const AllocaInst &>(inst);
748 bool isParamSpill =
749 currentZIRFunction_ &&
750 zirParamSpillIndex_ < currentZIRFunction_->getArguments().size();
751 bool isBorrowedSelf = false;
752 if (isParamSpill) {
753 isBorrowedSelf =
754 !currentZIRFunction_->ownerTypeName.empty() &&
755 currentZIRFunction_->getArguments()[zirParamSpillIndex_]->getRawName() ==
756 "self";
757 }
758 auto *alloca = createEntryAlloca(currentFn_,
759 static_cast<const Register &>(
760 *allocaInst.getResult()).getRawName(),
761 toLLVMType(*allocaInst.getAllocatedType()));
762 zirValueMap_[allocaInst.getResult().get()] = alloca;
763 if (isClassType(allocaInst.getAllocatedType())) {
764 builder_.CreateStore(
765 llvm::Constant::getNullValue(toLLVMType(*allocaInst.getAllocatedType())),
766 alloca);
767 if (isParamSpill) {
768 zirClassParamAllocas_.insert(allocaInst.getResult().get());
769 zirPendingClassParamInitAllocas_.insert(allocaInst.getResult().get());
770 }
771 if (!isBorrowedSelf) {
772 zirFunctionClassLocals_.push_back({allocaInst.getAllocatedType(), alloca});
773 }
774 }
775 if (isParamSpill) {
776 ++zirParamSpillIndex_;
777 }
778 return;
779 }
780 case OpCode::Load: {
781 const auto &loadInst = static_cast<const LoadInst &>(inst);
782 auto *src = lowerZIRValue(loadInst.getSource());
783 auto *value = builder_.CreateLoad(toLLVMType(*loadInst.getResult()->getType()),
784 src,
785 static_cast<const Register &>(
786 *loadInst.getResult()).getRawName());
787 zirValueMap_[loadInst.getResult().get()] = value;
788 return;
789 }
790 case OpCode::Store: {
791 const auto &storeInst = static_cast<const StoreInst &>(inst);
792 auto *dst = lowerZIRValue(storeInst.getDestination());
793 auto *src = lowerZIRRValue(storeInst.getSource());
794 auto dstType = storeInst.getDestination()->getType();
795 auto ptrType = std::dynamic_pointer_cast<zir::PointerType>(dstType);
796 auto valueType = ptrType ? ptrType->getBaseType() : nullptr;
797 if (valueType && isClassType(valueType)) {
798 if (zirPendingClassParamInitAllocas_.count(storeInst.getDestination().get()) >
799 0) {
800 builder_.CreateStore(src, dst);
801 zirPendingClassParamInitAllocas_.erase(storeInst.getDestination().get());
802 } else {
803 bool valueIsOwned =
804 zirOwnedClassValues_.count(storeInst.getSource().get()) > 0;
805 emitStoreWithArc(dst, src, valueType, valueIsOwned);
806 if (valueIsOwned) {
807 zirOwnedClassValues_.erase(storeInst.getSource().get());
808 }
809 }
810 } else {
811 builder_.CreateStore(src, dst);
812 }
813 return;
814 }
815 case OpCode::Add:
816 case OpCode::Sub:
817 case OpCode::Mul:
818 case OpCode::SDiv:
819 case OpCode::UDiv:
820 case OpCode::SRem:
821 case OpCode::URem:
822 case OpCode::Shl:
823 case OpCode::LShr:
824 case OpCode::AShr:
825 case OpCode::BitAnd:
826 case OpCode::BitOr:
827 case OpCode::BitXor: {
828 const auto &binaryInst = static_cast<const BinaryInst &>(inst);
829 auto *lhs = lowerZIRRValue(binaryInst.getLhs());
830 auto *rhs = lowerZIRRValue(binaryInst.getRhs());
831 llvm::Value *result = nullptr;
832 bool lhsIsPointer = binaryInst.getLhs()->getType()->getKind() ==
833 zir::TypeKind::Pointer;
834 bool rhsIsPointer = binaryInst.getRhs()->getType()->getKind() ==
835 zir::TypeKind::Pointer;
836 switch (inst.getOpCode()) {
837 case OpCode::Add:
838 if (isStringType(binaryInst.getLhs()->getType()) ||
839 isStringType(binaryInst.getRhs()->getType()) ||
840 binaryInst.getLhs()->getType()->getKind() == zir::TypeKind::Char ||
841 binaryInst.getRhs()->getType()->getKind() == zir::TypeKind::Char) {
842 result = emitStringConcat(lhs, rhs, binaryInst.getLhs()->getType(),
843 binaryInst.getRhs()->getType(),
844 binaryInst.getResult()->getType());
845 } else if (lhsIsPointer || rhsIsPointer) {
846 llvm::Value *pointerValue = lhsIsPointer ? lhs : rhs;
847 llvm::Value *offsetValue = lhsIsPointer ? rhs : lhs;
848 auto pointerType = std::static_pointer_cast<zir::PointerType>(
849 lhsIsPointer ? binaryInst.getLhs()->getType()
850 : binaryInst.getRhs()->getType());
851 auto *elemTy = toLLVMType(*pointerType->getBaseType());
852 auto *indexTy = llvm::Type::getInt64Ty(ctx_);
853 auto *index =
854 builder_.CreateIntCast(offsetValue, indexTy, /*isSigned=*/true);
855 result = builder_.CreateInBoundsGEP(elemTy, pointerValue, index);
856 } else if (lhs->getType()->isFloatingPointTy()) {
857 result = builder_.CreateFAdd(lhs, rhs);
858 } else {
859 result = builder_.CreateAdd(lhs, rhs);
860 }
861 break;
862 case OpCode::Sub:
863 if (lhsIsPointer && rhsIsPointer) {
864 auto pointerType =
865 std::static_pointer_cast<zir::PointerType>(binaryInst.getLhs()->getType());
866 auto *elemTy = toLLVMType(*pointerType->getBaseType());
867 auto *i64Ty = llvm::Type::getInt64Ty(ctx_);
868 auto *lhsInt = builder_.CreatePtrToInt(lhs, i64Ty);
869 auto *rhsInt = builder_.CreatePtrToInt(rhs, i64Ty);
870 auto *bytes = builder_.CreateSub(lhsInt, rhsInt);
871 llvm::Value *elemSize = llvm::ConstantExpr::getSizeOf(elemTy);
872 if (elemSize->getType() != i64Ty) {
873 elemSize = builder_.CreateIntCast(elemSize, i64Ty, false);
874 }
875 result = builder_.CreateSDiv(bytes, elemSize);
876 } else if (lhsIsPointer) {
877 auto pointerType =
878 std::static_pointer_cast<zir::PointerType>(binaryInst.getLhs()->getType());
879 auto *elemTy = toLLVMType(*pointerType->getBaseType());
880 auto *indexTy = llvm::Type::getInt64Ty(ctx_);
881 auto *index = builder_.CreateIntCast(rhs, indexTy, /*isSigned=*/true);
882 index = builder_.CreateNeg(index);
883 result = builder_.CreateInBoundsGEP(elemTy, lhs, index);
884 } else if (lhs->getType()->isFloatingPointTy()) {
885 result = builder_.CreateFSub(lhs, rhs);
886 } else {
887 result = builder_.CreateSub(lhs, rhs);
888 }
889 break;
890 case OpCode::Mul:
891 result = lhs->getType()->isFloatingPointTy() ? builder_.CreateFMul(lhs, rhs)
892 : builder_.CreateMul(lhs, rhs);
893 break;
894 case OpCode::SDiv:
895 result = lhs->getType()->isFloatingPointTy() ? builder_.CreateFDiv(lhs, rhs)
896 : builder_.CreateSDiv(lhs, rhs);
897 break;
898 case OpCode::UDiv:
899 result = lhs->getType()->isFloatingPointTy() ? builder_.CreateFDiv(lhs, rhs)
900 : builder_.CreateUDiv(lhs, rhs);
901 break;
902 case OpCode::SRem:
903 result = lhs->getType()->isFloatingPointTy() ? builder_.CreateFRem(lhs, rhs)
904 : builder_.CreateSRem(lhs, rhs);
905 break;
906 case OpCode::URem:
907 result = lhs->getType()->isFloatingPointTy() ? builder_.CreateFRem(lhs, rhs)
908 : builder_.CreateURem(lhs, rhs);
909 break;
910 case OpCode::Shl:
911 result = builder_.CreateShl(lhs, rhs);
912 break;
913 case OpCode::LShr:
914 result = builder_.CreateLShr(lhs, rhs);
915 break;
916 case OpCode::AShr:
917 result = builder_.CreateAShr(lhs, rhs);
918 break;
919 case OpCode::BitAnd:
920 result = builder_.CreateAnd(lhs, rhs);
921 break;
922 case OpCode::BitOr:
923 result = builder_.CreateOr(lhs, rhs);
924 break;
925 case OpCode::BitXor:
926 result = builder_.CreateXor(lhs, rhs);
927 break;
928 default:
929 break;
930 }
931 zirValueMap_[binaryInst.getResult().get()] = result;
932 return;
933 }
934 case OpCode::Cmp: {
935 const auto &cmpInst = static_cast<const CmpInst &>(inst);
936 auto *lhs = lowerZIRRValue(cmpInst.getLhs());
937 auto *rhs = lowerZIRRValue(cmpInst.getRhs());
938 llvm::Value *result = nullptr;
939 const auto &pred = cmpInst.getPredicate();
940 auto *lhsTy = lhs->getType();
941 auto *rhsTy = rhs->getType();
942 if (lhsTy != rhsTy) {
943 if (lhsTy->isIntegerTy() && rhsTy->isIntegerTy()) {
944 unsigned lhsBits = lhsTy->getIntegerBitWidth();
945 unsigned rhsBits = rhsTy->getIntegerBitWidth();
946 if (lhsBits < rhsBits) {
947 bool lhsUnsigned =
948 cmpInst.getLhs() && cmpInst.getLhs()->getType()
949 ? cmpInst.getLhs()->getType()->isUnsigned()
950 : false;
951 lhs = lhsUnsigned ? builder_.CreateZExt(lhs, rhsTy)
952 : builder_.CreateSExt(lhs, rhsTy);
953 lhsTy = lhs->getType();
954 } else if (rhsBits < lhsBits) {
955 bool rhsUnsigned =
956 cmpInst.getRhs() && cmpInst.getRhs()->getType()
957 ? cmpInst.getRhs()->getType()->isUnsigned()
958 : false;
959 rhs = rhsUnsigned ? builder_.CreateZExt(rhs, lhsTy)
960 : builder_.CreateSExt(rhs, lhsTy);
961 rhsTy = rhs->getType();
962 }
963 } else if (lhsTy->isFloatingPointTy() && rhsTy->isFloatingPointTy()) {
964 unsigned lhsBits = lhsTy->getPrimitiveSizeInBits();
965 unsigned rhsBits = rhsTy->getPrimitiveSizeInBits();
966 if (lhsBits < rhsBits) {
967 lhs = builder_.CreateFPExt(lhs, rhsTy);
968 lhsTy = lhs->getType();
969 } else if (rhsBits < lhsBits) {
970 rhs = builder_.CreateFPExt(rhs, lhsTy);
971 rhsTy = rhs->getType();
972 }
973 } else if (lhsTy->isIntegerTy() && rhsTy->isFloatingPointTy()) {
974 bool lhsUnsigned =
975 cmpInst.getLhs() && cmpInst.getLhs()->getType()
976 ? cmpInst.getLhs()->getType()->isUnsigned()
977 : false;
978 lhs = lhsUnsigned ? builder_.CreateUIToFP(lhs, rhsTy)
979 : builder_.CreateSIToFP(lhs, rhsTy);
980 lhsTy = lhs->getType();
981 } else if (lhsTy->isFloatingPointTy() && rhsTy->isIntegerTy()) {
982 bool rhsUnsigned =
983 cmpInst.getRhs() && cmpInst.getRhs()->getType()
984 ? cmpInst.getRhs()->getType()->isUnsigned()
985 : false;
986 rhs = rhsUnsigned ? builder_.CreateUIToFP(rhs, lhsTy)
987 : builder_.CreateSIToFP(rhs, lhsTy);
988 rhsTy = rhs->getType();
989 }
990 }
991 if (lhsTy != rhsTy) {
992 throw std::runtime_error("ZIR cmp operand type mismatch after coercion: " +
993 cmpInst.getLhs()->getTypeName() + " vs " +
994 cmpInst.getRhs()->getTypeName());
995 }
996 if (lhsTy->isFloatingPointTy()) {
997 if (pred == "eq")
998 result = builder_.CreateFCmpOEQ(lhs, rhs);
999 else if (pred == "ne")
1000 result = builder_.CreateFCmpONE(lhs, rhs);
1001 else if (pred == "slt" || pred == "ult")
1002 result = builder_.CreateFCmpOLT(lhs, rhs);
1003 else if (pred == "sle" || pred == "ule")
1004 result = builder_.CreateFCmpOLE(lhs, rhs);
1005 else if (pred == "sgt" || pred == "ugt")
1006 result = builder_.CreateFCmpOGT(lhs, rhs);
1007 else if (pred == "sge" || pred == "uge")
1008 result = builder_.CreateFCmpOGE(lhs, rhs);
1009 } else if (lhsTy->isIntOrIntVectorTy() || lhsTy->isPtrOrPtrVectorTy()) {
1010 if (pred == "eq")
1011 result = builder_.CreateICmpEQ(lhs, rhs);
1012 else if (pred == "ne")
1013 result = builder_.CreateICmpNE(lhs, rhs);
1014 else if (pred == "slt")
1015 result = builder_.CreateICmpSLT(lhs, rhs);
1016 else if (pred == "sle")
1017 result = builder_.CreateICmpSLE(lhs, rhs);
1018 else if (pred == "sgt")
1019 result = builder_.CreateICmpSGT(lhs, rhs);
1020 else if (pred == "sge")
1021 result = builder_.CreateICmpSGE(lhs, rhs);
1022 else if (pred == "ult")
1023 result = builder_.CreateICmpULT(lhs, rhs);
1024 else if (pred == "ule")
1025 result = builder_.CreateICmpULE(lhs, rhs);
1026 else if (pred == "ugt")
1027 result = builder_.CreateICmpUGT(lhs, rhs);
1028 else if (pred == "uge")
1029 result = builder_.CreateICmpUGE(lhs, rhs);
1030 }
1031 if (!result) {
1032 throw std::runtime_error("unsupported ZIR cmp predicate/type: " + pred +
1033 " on " + cmpInst.getLhs()->getTypeName());
1034 }
1035 zirValueMap_[cmpInst.getResult().get()] = result;
1036 return;
1037 }
1038 case OpCode::Br: {
1039 const auto &branchInst = static_cast<const BranchInst &>(inst);
1040 builder_.CreateBr(zirBlockMap_.at(branchInst.getTarget()));
1041 return;
1042 }
1043 case OpCode::CondBr: {
1044 const auto &branchInst = static_cast<const CondBranchInst &>(inst);
1045 builder_.CreateCondBr(lowerZIRRValue(branchInst.getCondition()),
1046 zirBlockMap_.at(branchInst.getTrueLabel()),
1047 zirBlockMap_.at(branchInst.getFalseLabel()));
1048 return;
1049 }
1050 case OpCode::Ret: {
1051 const auto &returnInst = static_cast<const ReturnInst &>(inst);
1052 if (returnInst.getValue()) {
1053 auto *retValue = lowerZIRRValue(returnInst.getValue());
1054 auto retType = returnInst.getValue()->getType();
1055 if (isClassType(retType) &&
1056 zirOwnedClassValues_.count(returnInst.getValue().get()) == 0) {
1057 emitRetainIfNeeded(retValue, retType);
1058 }
1059 for (auto it = zirFunctionClassLocals_.rbegin();
1060 it != zirFunctionClassLocals_.rend(); ++it) {
1061 auto *value = builder_.CreateLoad(toLLVMType(*it->first), it->second,
1062 "zir.arc.ret.release");
1063 if (isWeakClassType(it->first)) {
1064 emitReleaseWeakIfNeeded(value, it->first);
1065 } else {
1066 emitReleaseIfNeeded(value, it->first);
1067 }
1068 }
1069 builder_.CreateRet(retValue);
1070 } else {
1071 for (auto it = zirFunctionClassLocals_.rbegin();
1072 it != zirFunctionClassLocals_.rend(); ++it) {
1073 auto *value = builder_.CreateLoad(toLLVMType(*it->first), it->second,
1074 "zir.arc.ret.release");
1075 if (isWeakClassType(it->first)) {
1076 emitReleaseWeakIfNeeded(value, it->first);
1077 } else {
1078 emitReleaseIfNeeded(value, it->first);
1079 }
1080 }
1081 builder_.CreateRetVoid();
1082 }
1083 return;
1084 }
1085 case OpCode::Call: {
1086 const auto &callInst = static_cast<const CallInst &>(inst);
1087 std::vector<llvm::Value *> args;
1088
1089 if (callInst.isIndirect()) {
1090 auto *calleePtr = lowerZIRRValue(callInst.getCalleeValue());
1091 const auto &fpType = static_cast<const zir::FunctionPointerType &>(
1092 *callInst.getCalleeValue()->getType());
1093 std::vector<llvm::Type *> paramTypes;
1094 for (const auto &p : fpType.getParams())
1095 paramTypes.push_back(toLLVMType(*p));
1096 auto *fnTy = llvm::FunctionType::get(toLLVMType(*fpType.getReturnType()),
1097 paramTypes, false);
1098 for (const auto &arg : callInst.getArguments())
1099 args.push_back(lowerZIRRValue(arg));
1100 auto *call = builder_.CreateCall(fnTy, calleePtr, args);
1101 if (callInst.getResult()) {
1102 zirValueMap_[callInst.getResult().get()] = call;
1103 if (callInst.returnsRef()) {
1104 refReturnValues_.insert(callInst.getResult().get());
1105 }
1106 }
1107 return;
1108 }
1109
1110 auto calleeIt = functionMap_.find(callInst.getFunctionName());
1111 if (calleeIt == functionMap_.end()) {
1112 auto zirDeclIt = zirFunctionMap_.find(callInst.getFunctionName());
1113 if (zirDeclIt != zirFunctionMap_.end()) {
1114 declareZIRFunction(*zirDeclIt->second, true);
1115 calleeIt = functionMap_.find(callInst.getFunctionName());
1116 }
1117 }
1118 if (calleeIt == functionMap_.end()) {
1119 auto *existing = module_->getFunction(callInst.getFunctionName());
1120 if (!existing) {
1121 std::vector<llvm::Type *> inferredParamTypes;
1122 inferredParamTypes.reserve(callInst.getArguments().size());
1123 for (size_t i = 0; i < callInst.getArguments().size(); ++i) {
1124 bool isRef =
1125 i < callInst.getArgumentIsRef().size() &&
1126 callInst.getArgumentIsRef()[i];
1127 auto *argTy = isRef ? lowerZIRValue(callInst.getArguments()[i])->getType()
1128 : lowerZIRRValue(callInst.getArguments()[i])->getType();
1129 inferredParamTypes.push_back(argTy);
1130 }
1131
1132 llvm::Type *retTy = llvm::Type::getVoidTy(ctx_);
1133 if (callInst.getResult()) {
1134 retTy = toLLVMType(*callInst.getResult()->getType());
1135 }
1136
1137 auto *fnTy =
1138 llvm::FunctionType::get(retTy, inferredParamTypes, false);
1139 existing = llvm::Function::Create(
1140 fnTy, llvm::Function::ExternalLinkage, callInst.getFunctionName(),
1141 *module_);
1142
1143 }
1144
1145 functionMap_[callInst.getFunctionName()] = existing;
1146 calleeIt = functionMap_.find(callInst.getFunctionName());
1147 }
1148 if (calleeIt == functionMap_.end()) {
1149 throw std::runtime_error("missing callee in LLVM function map: " +
1150 callInst.getFunctionName());
1151 }
1152 auto *callee = calleeIt->second;
1153 auto *calleeTy = callee->getFunctionType();
1154 auto zirIt = zirFunctionMap_.find(callInst.getFunctionName());
1155 size_t fixedParamCount = callInst.getArguments().size();
1156 bool hasVariadicParameter = false;
1157 std::shared_ptr<zir::Type> variadicElementType = nullptr;
1158 bool isCVariadic = calleeTy->isVarArg();
1159 if (zirIt != zirFunctionMap_.end()) {
1160 fixedParamCount = 0;
1161 for (const auto &param : zirIt->second->getArguments()) {
1162 if (param->isVariadicPack()) {
1163 hasVariadicParameter = true;
1164 variadicElementType = param->getVariadicElementType();
1165 break;
1166 }
1167 ++fixedParamCount;
1168 }
1169 isCVariadic = zirIt->second->isCVariadic;
1170 }
1171 for (size_t i = 0; i < callInst.getArguments().size(); ++i) {
1172 bool isRef =
1173 i < callInst.getArgumentIsRef().size() && callInst.getArgumentIsRef()[i];
1174 auto *arg = isRef ? lowerZIRValue(callInst.getArguments()[i])
1175 : lowerZIRRValue(callInst.getArguments()[i]);
1176 std::shared_ptr<zir::Type> calleeParamType = nullptr;
1177 if (zirIt != zirFunctionMap_.end() && i < fixedParamCount &&
1178 i < zirIt->second->getArguments().size()) {
1179 calleeParamType = zirIt->second->getArguments()[i]->getType();
1180 }
1181 llvm::Type *paramTy = nullptr;
1182 if (i < fixedParamCount && static_cast<unsigned>(i) < calleeTy->getNumParams()) {
1183 paramTy = calleeTy->getParamType(static_cast<unsigned>(i));
1184 }
1185 if (paramTy && arg->getType() != paramTy) {
1186 auto *argPtrTy = llvm::dyn_cast<llvm::PointerType>(arg->getType());
1187 if (argPtrTy && !paramTy->isPointerTy()) {
1188 arg = builder_.CreateLoad(paramTy, arg, "zir.call.arg");
1189 }
1190 } else if (isCVariadic && i >= fixedParamCount) {
1191 // C varargs default argument promotions:
1192 // - integer types narrower than int -> int
1193 // - float -> double
1194 // (already-typed varargs keep their value category)
1195 auto *argTy = arg->getType();
1196 if (argTy->isIntegerTy()) {
1197 unsigned bits = argTy->getIntegerBitWidth();
1198 if (bits < 32) {
1199 auto argType = callInst.getArguments()[i]
1200 ? callInst.getArguments()[i]->getType()
1201 : nullptr;
1202 bool isUnsignedArg =
1203 argType &&
1204 (argType->isUnsigned() ||
1205 argType->getKind() == zir::TypeKind::Bool ||
1206 argType->getKind() == zir::TypeKind::Char);
1207 auto *i32Ty = llvm::Type::getInt32Ty(ctx_);
1208 arg = isUnsignedArg ? builder_.CreateZExt(arg, i32Ty, "cvararg.zext")
1209 : builder_.CreateSExt(arg, i32Ty, "cvararg.sext");
1210 }
1211 } else if (argTy->isFloatTy()) {
1212 auto *f64Ty = llvm::Type::getDoubleTy(ctx_);
1213 arg = builder_.CreateFPExt(arg, f64Ty, "cvararg.fpext");
1214 }
1215 }
1216 bool isBorrowedSelfArg =
1217 zirIt != zirFunctionMap_.end() && i == 0 &&
1218 !zirIt->second->ownerTypeName.empty() &&
1219 i < zirIt->second->getArguments().size() &&
1220 zirIt->second->getArguments()[i]->getRawName() == "self";
1221 if (!isRef && i < fixedParamCount && !isBorrowedSelfArg &&
1222 !(calleeParamType && isWeakClassType(calleeParamType)) &&
1223 isClassType(callInst.getArguments()[i]->getType()) &&
1224 zirOwnedClassValues_.count(callInst.getArguments()[i].get()) == 0) {
1225 emitRetainIfNeeded(arg, callInst.getArguments()[i]->getType());
1226 }
1227 args.push_back(arg);
1228 }
1229 if (hasVariadicParameter) {
1230 auto *elemTy = toLLVMType(*variadicElementType);
1231 auto *elemPtrTy = llvm::PointerType::getUnqual(elemTy);
1232 size_t explicitVariadicCount = callInst.getArguments().size() - fixedParamCount;
1233
1234 llvm::Value *forwardedCount =
1235 llvm::ConstantInt::get(llvm::Type::getInt32Ty(ctx_), 0);
1236 llvm::Value *forwardedData = llvm::ConstantPointerNull::get(elemPtrTy);
1237 if (callInst.getVariadicPack()) {
1238 auto *packValue = lowerZIRRValue(callInst.getVariadicPack());
1239 forwardedData =
1240 builder_.CreateExtractValue(packValue, {0}, "varargs.forward.data");
1241 forwardedCount =
1242 builder_.CreateExtractValue(packValue, {1}, "varargs.forward.len");
1243 if (forwardedCount->getType() != llvm::Type::getInt32Ty(ctx_)) {
1244 forwardedCount = builder_.CreateIntCast(
1245 forwardedCount, llvm::Type::getInt32Ty(ctx_), /*isSigned=*/true);
1246 }
1247 }
1248
1249 llvm::Value *explicitCount =
1250 llvm::ConstantInt::get(llvm::Type::getInt32Ty(ctx_),
1251 static_cast<uint64_t>(explicitVariadicCount));
1252
1253 if (explicitVariadicCount == 0 && !callInst.getVariadicPack()) {
1254 args.push_back(explicitCount);
1255 args.push_back(llvm::ConstantPointerNull::get(elemPtrTy));
1256 } else if (explicitVariadicCount == 0) {
1257 args.push_back(forwardedCount);
1258 args.push_back(forwardedData);
1259 } else {
1260 llvm::Value *totalCount = explicitCount;
1261 if (callInst.getVariadicPack()) {
1262 totalCount =
1263 builder_.CreateAdd(explicitCount, forwardedCount, "varargs.total");
1264 }
1265
1266 auto *buffer = builder_.CreateAlloca(elemTy, totalCount, "varargs.buf");
1267 for (size_t i = 0; i < explicitVariadicCount; ++i) {
1268 auto *dst = builder_.CreateInBoundsGEP(
1269 elemTy, buffer,
1270 llvm::ConstantInt::get(llvm::Type::getInt32Ty(ctx_),
1271 static_cast<uint64_t>(i)));
1272 builder_.CreateStore(args[fixedParamCount + i], dst);
1273 }
1274
1275 if (callInst.getVariadicPack()) {
1276 auto *copyCondBB =
1277 llvm::BasicBlock::Create(ctx_, "varargs.copy.cond", currentFn_);
1278 auto *copyBodyBB =
1279 llvm::BasicBlock::Create(ctx_, "varargs.copy.body", currentFn_);
1280 auto *copyDoneBB =
1281 llvm::BasicBlock::Create(ctx_, "varargs.copy.done", currentFn_);
1282 auto *copyPreheaderBB = builder_.GetInsertBlock();
1283
1284 builder_.CreateBr(copyCondBB);
1285 builder_.SetInsertPoint(copyCondBB);
1286 auto *indexPhi =
1287 builder_.CreatePHI(llvm::Type::getInt32Ty(ctx_), 2, "varargs.i");
1288 indexPhi->addIncoming(
1289 llvm::ConstantInt::get(llvm::Type::getInt32Ty(ctx_), 0),
1290 copyPreheaderBB);
1291 auto *cond =
1292 builder_.CreateICmpULT(indexPhi, forwardedCount, "varargs.cond");
1293 builder_.CreateCondBr(cond, copyBodyBB, copyDoneBB);
1294
1295 builder_.SetInsertPoint(copyBodyBB);
1296 auto *src = builder_.CreateInBoundsGEP(elemTy, forwardedData, indexPhi,
1297 "varargs.src");
1298 auto *srcVal = builder_.CreateLoad(elemTy, src, "varargs.load");
1299 auto *dstIndex = builder_.CreateAdd(indexPhi, explicitCount);
1300 auto *dst = builder_.CreateInBoundsGEP(elemTy, buffer, dstIndex,
1301 "varargs.dst");
1302 builder_.CreateStore(srcVal, dst);
1303 auto *next = builder_.CreateAdd(
1304 indexPhi, llvm::ConstantInt::get(llvm::Type::getInt32Ty(ctx_), 1));
1305 builder_.CreateBr(copyCondBB);
1306 indexPhi->addIncoming(next, copyBodyBB);
1307
1308 builder_.SetInsertPoint(copyDoneBB);
1309 }
1310
1311 args.erase(args.begin() + static_cast<std::ptrdiff_t>(fixedParamCount),
1312 args.end());
1313 args.push_back(totalCount);
1314 args.push_back(buffer);
1315 }
1316 } else if (isCVariadic) {
1317 // Extra arguments are passed through unchanged after the fixed params.
1318 }
1319 llvm::Value *call = nullptr;
1320 if (zirIt != zirFunctionMap_.end() && zirIt->second->vtableSlot >= 0 &&
1321 !args.empty()) {
1322 auto receiverType = callInst.getArguments().front()->getType();
1323 std::shared_ptr<zir::ClassType> classType = nullptr;
1324 if (receiverType->getKind() == zir::TypeKind::Class) {
1325 classType = std::static_pointer_cast<zir::ClassType>(receiverType);
1326 } else if (receiverType->getKind() == zir::TypeKind::Pointer) {
1327 auto baseType =
1328 std::static_pointer_cast<zir::PointerType>(receiverType)
1329 ->getBaseType();
1330 if (baseType->getKind() == zir::TypeKind::Class) {
1331 classType = std::static_pointer_cast<zir::ClassType>(baseType);
1332 }
1333 }
1334
1335 if (classType) {
1336 auto *objectTy = structCache_.at(classType->getCodegenName() + ".obj");
1337 auto *selfPtr = builder_.CreateBitCast(
1338 args[0], llvm::PointerType::getUnqual(objectTy), "zir.method.self");
1339 auto *vtableAddr = builder_.CreateStructGEP(
1340 objectTy, selfPtr, 7, "zir.method.vtable.addr");
1341 auto *i8PtrTy =
1342 llvm::PointerType::getUnqual(llvm::Type::getInt8Ty(ctx_));
1343 auto *vtablePtrTy = llvm::PointerType::getUnqual(i8PtrTy);
1344 auto *vtablePtr = builder_.CreateLoad(vtablePtrTy, vtableAddr,
1345 "zir.method.vtable");
1346 auto *slotAddr = builder_.CreateInBoundsGEP(
1347 i8PtrTy, vtablePtr,
1348 llvm::ConstantInt::get(llvm::Type::getInt32Ty(ctx_),
1349 static_cast<uint64_t>(zirIt->second->vtableSlot)));
1350 auto *fnRaw =
1351 builder_.CreateLoad(i8PtrTy, slotAddr, "zir.method.fn.raw");
1352 auto *fnPtr = builder_.CreateBitCast(
1353 fnRaw, llvm::PointerType::getUnqual(calleeTy), "zir.method.fn");
1354 call = builder_.CreateCall(calleeTy, fnPtr, args);
1355 }
1356 }
1357 if (!call) {
1358 call = builder_.CreateCall(callee, args);
1359 }
1360 if (callInst.getResult()) {
1361 zirValueMap_[callInst.getResult().get()] = call;
1362 if (callInst.returnsRef()) {
1363 refReturnValues_.insert(callInst.getResult().get());
1364 }
1365 if (isClassType(callInst.getResult()->getType())) {
1366 zirOwnedClassValues_.insert(callInst.getResult().get());
1367 }
1368 }
1369 return;
1370 }
1371 case OpCode::GetElementPtr: {
1372 const auto &gepInst = static_cast<const GetElementPtrInst &>(inst);
1373 auto *ptr = lowerZIRValue(gepInst.getPointer());
1374 auto operandType = gepInst.getPointer()->getType();
1375 auto pointerType = std::dynamic_pointer_cast<zir::PointerType>(operandType);
1376 auto baseType = pointerType ? pointerType->getBaseType() : operandType;
1377 llvm::Value *gep = nullptr;
1378
1379 if (baseType->getKind() == zir::TypeKind::Record) {
1380 auto recordType = std::static_pointer_cast<zir::RecordType>(baseType);
1381 auto *structTy = llvm::cast<llvm::StructType>(toLLVMType(*recordType));
1382 llvm::Value *structPtr = ptr;
1383 if (!pointerType) {
1384 auto *tmp = createEntryAlloca(
1385 currentFn_,
1386 static_cast<const Register &>(*gepInst.getResult()).getRawName() +
1387 ".addr",
1388 structTy);
1389 builder_.CreateStore(ptr, tmp);
1390 structPtr = tmp;
1391 }
1392 gep = builder_.CreateStructGEP(
1393 structTy, structPtr, static_cast<unsigned>(gepInst.getIndex()),
1394 static_cast<const Register &>(*gepInst.getResult()).getRawName());
1395 } else if (baseType->getKind() == zir::TypeKind::Class) {
1396 auto classType = std::static_pointer_cast<zir::ClassType>(baseType);
1397 auto *objectTy = structCache_.at(classType->getCodegenName() + ".obj");
1398 llvm::Value *objectPtr = ptr;
1399 gep = builder_.CreateStructGEP(
1400 objectTy, objectPtr, static_cast<unsigned>(gepInst.getIndex() + 8),
1401 static_cast<const Register &>(*gepInst.getResult()).getRawName());
1402 } else {
1403 llvm::Value *basePtr = ptr;
1404 auto *index = llvm::ConstantInt::get(llvm::Type::getInt32Ty(ctx_),
1405 gepInst.getIndex());
1406 if (baseType->getKind() == zir::TypeKind::Array) {
1407 auto *arrayTy = toLLVMType(*baseType);
1408 if (!pointerType) {
1409 auto *tmp = createEntryAlloca(
1410 currentFn_,
1411 static_cast<const Register &>(*gepInst.getResult()).getRawName() +
1412 ".array.addr",
1413 arrayTy);
1414 builder_.CreateStore(ptr, tmp);
1415 basePtr = tmp;
1416 }
1417 auto *i32Ty = llvm::Type::getInt32Ty(ctx_);
1418 llvm::Value *indices[] = {llvm::ConstantInt::get(i32Ty, 0), index};
1419 gep = builder_.CreateInBoundsGEP(
1420 arrayTy, basePtr, indices,
1421 static_cast<const Register &>(*gepInst.getResult()).getRawName());
1422 } else {
1423 if (!pointerType) {
1424 throw std::runtime_error("ZIR getelementptr expects pointer operand");
1425 }
1426 auto *elemTy = toLLVMType(*baseType);
1427 gep = builder_.CreateInBoundsGEP(
1428 elemTy, ptr, index,
1429 static_cast<const Register &>(*gepInst.getResult()).getRawName());
1430 }
1431 }
1432 zirValueMap_[gepInst.getResult().get()] = gep;
1433 return;
1434 }
1435 case OpCode::Phi: {
1436 const auto &phiInst = static_cast<const PhiInst &>(inst);
1437 auto *phi = builder_.CreatePHI(toLLVMType(*phiInst.getResult()->getType()),
1438 phiInst.getIncoming().size(),
1439 static_cast<const Register &>(
1440 *phiInst.getResult()).getRawName());
1441 bool phiOwnsClassValue = isClassType(phiInst.getResult()->getType());
1442 for (const auto &incoming : phiInst.getIncoming()) {
1443 auto blockIt = zirBlockExitMap_.find(incoming.first);
1444 auto *incomingBlock =
1445 blockIt != zirBlockExitMap_.end() ? blockIt->second
1446 : zirBlockMap_.at(incoming.first);
1447 phi->addIncoming(lowerZIRValue(incoming.second), incomingBlock);
1448 if (phiOwnsClassValue &&
1449 zirOwnedClassValues_.count(incoming.second.get()) == 0) {
1450 phiOwnsClassValue = false;
1451 }
1452 }
1453 zirValueMap_[phiInst.getResult().get()] = phi;
1454 if (phiOwnsClassValue) {
1455 zirOwnedClassValues_.insert(phiInst.getResult().get());
1456 }
1457 return;
1458 }
1459 case OpCode::Cast: {
1460 const auto &castInst = static_cast<const CastInst &>(inst);
1461 auto *result = lowerZIRCast(lowerZIRRValue(castInst.getSource()),
1462 castInst.getSource()->getType(),
1463 castInst.getTargetType());
1464 zirValueMap_[castInst.getResult().get()] = result;
1465 if (isClassType(castInst.getTargetType()) &&
1466 zirOwnedClassValues_.count(castInst.getSource().get()) > 0) {
1467 zirOwnedClassValues_.insert(castInst.getResult().get());
1468 zirOwnedClassValues_.erase(castInst.getSource().get());
1469 }
1470 return;
1471 }
1472 case OpCode::WeakLock: {
1473 const auto &weakLockInst = static_cast<const WeakLockInst &>(inst);
1474 auto *result = emitWeakLock(lowerZIRRValue(weakLockInst.getWeakValue()),
1475 weakLockInst.getWeakValue()->getType());
1476 zirValueMap_[weakLockInst.getResult().get()] = result;
1477 if (isClassType(weakLockInst.getResult()->getType())) {
1478 zirOwnedClassValues_.insert(weakLockInst.getResult().get());
1479 }
1480 return;
1481 }
1482 case OpCode::WeakAlive: {
1483 const auto &weakAliveInst = static_cast<const WeakAliveInst &>(inst);
1484 auto *result = emitWeakAlive(lowerZIRRValue(weakAliveInst.getWeakValue()),
1485 weakAliveInst.getWeakValue()->getType());
1486 zirValueMap_[weakAliveInst.getResult().get()] = result;
1487 return;
1488 }
1489 case OpCode::Alloc:
1490 {
1491 const auto &allocInst = static_cast<const AllocInst &>(inst);
1492 auto allocType = allocInst.getAllocatedType();
1493 if (allocType->getKind() != zir::TypeKind::Class) {
1494 throw std::runtime_error("ZIR alloc currently supports only class types");
1495 }
1496
1497 auto classType = std::static_pointer_cast<zir::ClassType>(allocType);
1498 auto *ptrTy = llvm::cast<llvm::PointerType>(toLLVMType(*classType));
1499 auto *objectTy = structCache_.at(classType->getCodegenName() + ".obj");
1500 auto *sizeOfObj = llvm::ConstantExpr::getSizeOf(objectTy);
1501 auto *sizeTy = llvm::Type::getInt64Ty(ctx_);
1502 llvm::Value *sizeValue = sizeOfObj;
1503 if (sizeValue->getType() != sizeTy) {
1504 sizeValue = builder_.CreateIntCast(sizeValue, sizeTy, /*isSigned=*/false);
1505 }
1506
1507 if (functionMap_.count("malloc") == 0) {
1508 auto *mallocTy = llvm::FunctionType::get(
1509 llvm::PointerType::getUnqual(llvm::Type::getInt8Ty(ctx_)), {sizeTy},
1510 false);
1511 auto *mallocFn = llvm::Function::Create(
1512 mallocTy, llvm::Function::ExternalLinkage, "malloc", *module_);
1513 functionMap_["malloc"] = mallocFn;
1514 }
1515
1516 auto *rawPtr = builder_.CreateCall(functionMap_.at("malloc"), {sizeValue},
1517 "class.alloc");
1518 auto *typedPtr = builder_.CreateBitCast(rawPtr, ptrTy, "class.obj");
1519
1520 if (functionMap_.count("zap_arc_register") == 0) {
1521 auto *registerTy = llvm::FunctionType::get(
1522 llvm::Type::getVoidTy(ctx_),
1523 {llvm::PointerType::getUnqual(llvm::Type::getInt8Ty(ctx_))}, false);
1524 auto *registerFn =
1525 llvm::Function::Create(registerTy, llvm::Function::ExternalLinkage,
1526 "zap_arc_register", *module_);
1527 functionMap_["zap_arc_register"] = registerFn;
1528 }
1529 builder_.CreateCall(functionMap_.at("zap_arc_register"), {rawPtr});
1530
1531 auto *refCountAddr =
1532 builder_.CreateStructGEP(objectTy, typedPtr, 0, "refcount.addr");
1533 builder_.CreateStore(
1534 llvm::ConstantInt::get(llvm::Type::getInt64Ty(ctx_), 1), refCountAddr);
1535 auto *weakCountAddr =
1536 builder_.CreateStructGEP(objectTy, typedPtr, 1, "weakcount.addr");
1537 builder_.CreateStore(
1538 llvm::ConstantInt::get(llvm::Type::getInt64Ty(ctx_), 0), weakCountAddr);
1539 auto *aliveAddr =
1540 builder_.CreateStructGEP(objectTy, typedPtr, 2, "alive.addr");
1541 builder_.CreateStore(
1542 llvm::ConstantInt::get(llvm::Type::getInt8Ty(ctx_), 1), aliveAddr);
1543 auto *gcMarkAddr =
1544 builder_.CreateStructGEP(objectTy, typedPtr, 3, "gcmark.addr");
1545 builder_.CreateStore(
1546 llvm::ConstantInt::get(llvm::Type::getInt8Ty(ctx_), 0), gcMarkAddr);
1547 auto *releaseFnAddr =
1548 builder_.CreateStructGEP(objectTy, typedPtr, 4, "release.fn.addr");
1549 auto *releaseFnPtr = builder_.CreateBitCast(
1550 classReleaseFns_.at(classType->getName()),
1551 llvm::PointerType::getUnqual(llvm::Type::getInt8Ty(ctx_)));
1552 builder_.CreateStore(releaseFnPtr, releaseFnAddr);
1553 auto *destroyFnAddr =
1554 builder_.CreateStructGEP(objectTy, typedPtr, 5, "destroy.fn.addr");
1555 auto *destroyFnPtr = builder_.CreateBitCast(
1556 classDestroyFns_.at(classType->getName()),
1557 llvm::PointerType::getUnqual(llvm::Type::getInt8Ty(ctx_)));
1558 builder_.CreateStore(destroyFnPtr, destroyFnAddr);
1559 auto *metadataAddr =
1560 builder_.CreateStructGEP(objectTy, typedPtr, 6, "metadata.addr");
1561 auto *metadataPtr = builder_.CreateBitCast(
1562 classMetadataGlobals_.at(classType->getName()),
1563 llvm::PointerType::getUnqual(llvm::Type::getInt8Ty(ctx_)));
1564 builder_.CreateStore(metadataPtr, metadataAddr);
1565 auto *vtableAddr =
1566 builder_.CreateStructGEP(objectTy, typedPtr, 7, "vtable.addr");
1567 auto *i8PtrTy = llvm::PointerType::getUnqual(llvm::Type::getInt8Ty(ctx_));
1568 auto *vtablePtrTy = llvm::PointerType::getUnqual(i8PtrTy);
1569 auto *vtableGlobal = classVTables_.at(classType->getName());
1570 auto *zero = llvm::ConstantInt::get(llvm::Type::getInt32Ty(ctx_), 0);
1571 llvm::Constant *vtableIndices[] = {zero, zero};
1572 auto *vtablePtr = llvm::ConstantExpr::getInBoundsGetElementPtr(
1573 vtableGlobal->getValueType(), vtableGlobal, vtableIndices);
1574 builder_.CreateStore(llvm::ConstantExpr::getBitCast(vtablePtr, vtablePtrTy),
1575 vtableAddr);
1576
1577 for (size_t i = 0; i < classType->getFields().size(); ++i) {
1578 auto *fieldAddr = builder_.CreateStructGEP(
1579 objectTy, typedPtr, static_cast<unsigned>(i + 8));
1580 builder_.CreateStore(
1581 llvm::Constant::getNullValue(
1582 toLLVMType(*classType->getFields()[i].type)),
1583 fieldAddr);
1584 }
1585
1586 zirValueMap_[allocInst.getResult().get()] = typedPtr;
1587 zirOwnedClassValues_.insert(allocInst.getResult().get());
1588 return;
1589 }
1590 case OpCode::Retain:
1591 case OpCode::Release:
1592 throw std::runtime_error("ZIR opcode not lowered yet in LLVM backend");
1593 }
1594}

1596void LLVMCodeGen::emitZIRFunction(const zir::Function &fn) {
1597 currentZIRFunction_ = &fn;
1598 currentFn_ = functionMap_.at(fn.name);
1599 zirBlockMap_.clear();
1600 zirValueMap_.clear();
1601 zirOwnedClassValues_.clear();
1602 zirClassParamAllocas_.clear();
1603 zirPendingClassParamInitAllocas_.clear();
1604 zirFunctionClassLocals_.clear();
1605 zirParamSpillIndex_ = 0;
1606 zirBlockExitMap_.clear();
1607
1608 auto llvmArgIt = currentFn_->arg_begin();
1609 if (fn.name == "main") {
1610 auto *i32Ty = llvm::Type::getInt32Ty(ctx_);
1611 auto *i8PtrTy = llvm::PointerType::getUnqual(llvm::Type::getInt8Ty(ctx_));
1612 auto *argvTy = llvm::PointerType::getUnqual(i8PtrTy);
1613
1614 llvmArgIt->setName("argc");
1615 llvm::Value *argcValue = &*llvmArgIt++;
1616 llvmArgIt->setName("argv");
1617 llvm::Value *argvValue = &*llvmArgIt++;
1618
1619 if (functionMap_.count("__zap_process_set_args") == 0) {
1620 auto *ft = llvm::FunctionType::get(llvm::Type::getVoidTy(ctx_),
1621 {i32Ty, argvTy}, false);
1622 auto *setArgsFn =
1623 llvm::Function::Create(ft, llvm::Function::ExternalLinkage,
1624 "__zap_process_set_args", *module_);
1625 functionMap_["__zap_process_set_args"] = setArgsFn;
1626 }
1627 builder_.SetInsertPoint(
1628 llvm::BasicBlock::Create(ctx_, "entry", currentFn_));
1629 builder_.CreateCall(
1630 functionMap_.at("__zap_process_set_args"),
1631 {builder_.CreateIntCast(argcValue, i32Ty, true), argvValue});
1632 builder_.CreateBr(
1633 llvm::BasicBlock::Create(ctx_, fn.getBlocks().front()->label, currentFn_));
1634 zirBlockMap_[fn.getBlocks().front()->label] = &currentFn_->back();
1635 }
1636
1637 std::vector<llvm::Value *> physicalArgs;
1638 for (; llvmArgIt != currentFn_->arg_end(); ++llvmArgIt) {
1639 physicalArgs.push_back(&*llvmArgIt);
1640 }
1641
1642 for (size_t i = 0; i < fn.getBlocks().size(); ++i) {
1643 const auto &block = fn.getBlocks()[i];
1644 if (fn.name == "main" && i == 0 && zirBlockMap_.count(block->label) != 0) {
1645 continue;
1646 }
1647 zirBlockMap_[block->label] =
1648 llvm::BasicBlock::Create(ctx_, block->label, currentFn_);
1649 }
1650
1651 auto *argInsertBlock = zirBlockMap_.at(fn.getBlocks().front()->label);
1652 builder_.SetInsertPoint(argInsertBlock, argInsertBlock->begin());
1653 size_t physicalArgIndex = 0;
1654 for (const auto &arg : fn.getArguments()) {
1655 if (arg->isVariadicPack()) {
1656 auto *sliceTy = static_cast<llvm::StructType *>(toLLVMType(*arg->getType()));
1657 llvm::Value *sliceValue = llvm::PoisonValue::get(sliceTy);
1658 llvm::Value *countValue = physicalArgs.at(physicalArgIndex++);
1659 llvm::Value *dataValue = physicalArgs.at(physicalArgIndex++);
1660 sliceValue = builder_.CreateInsertValue(sliceValue, dataValue, {0},
1661 arg->getRawName() + ".data");
1662 if (countValue->getType() != sliceTy->getElementType(1)) {
1663 countValue = builder_.CreateIntCast(
1664 countValue, sliceTy->getElementType(1),
1665 /*isSigned=*/true, arg->getRawName() + ".len.cast");
1666 }
1667 sliceValue = builder_.CreateInsertValue(sliceValue, countValue, {1},
1668 arg->getRawName() + ".len");
1669 zirValueMap_[arg.get()] = sliceValue;
1670 continue;
1671 }
1672 zirValueMap_[arg.get()] = physicalArgs.at(physicalArgIndex++);
1673 }
1674
1675 for (const auto &block : fn.getBlocks()) {
1676 builder_.SetInsertPoint(zirBlockMap_.at(block->label));
1677 for (const auto &inst : block->getInstructions()) {
1678 emitZIRInstruction(*inst);
1679 if (builder_.GetInsertBlock()->getTerminator()) {
1680 break;
1681 }
1682 }
1683
1684 auto *endBlock = builder_.GetInsertBlock();
1685 if (endBlock && endBlock->getTerminator()) {
1686 zirBlockExitMap_[block->label] = endBlock;
1687 } else {
1688 zirBlockExitMap_[block->label] = zirBlockMap_.at(block->label);
1689 }
1690 }
1691
1692 currentFn_ = nullptr;
1693 currentZIRFunction_ = nullptr;
1694 zirBlockMap_.clear();
1695 zirValueMap_.clear();
1696 zirOwnedClassValues_.clear();
1697 zirBlockExitMap_.clear();
1698 zirClassParamAllocas_.clear();
1699 zirPendingClassParamInitAllocas_.clear();
1700 zirFunctionClassLocals_.clear();
1701}

1703void LLVMCodeGen::visit(sema::BoundRootNode &node) {
1704 for (const auto &extFn : node.externalFunctions) {
1705 auto *ft = buildFunctionType(*extFn->symbol);
1706 auto *f = llvm::Function::Create(ft, llvm::Function::ExternalLinkage,
1707 extFn->symbol->linkName, *module_);
1708 auto argIt = f->arg_begin();
1709 for (const auto &param : extFn->symbol->parameters) {
1710 if (param->is_variadic_pack) {
1711 argIt->setName(param->name + ".count");
1712 ++argIt;
1713 argIt->setName(param->name + ".data");
1714 ++argIt;
1715 continue;
1716 }
1717 argIt->setName(param->name);
1718 ++argIt;
1719 }
1720

2315 lastValue_ = isFP ? builder_.CreateFCmpOEQ(lhs, rhs)
2316 : builder_.CreateICmpEQ(lhs, rhs);
2317 else if (node.op == "!=")
2318 lastValue_ = isFP ? builder_.CreateFCmpONE(lhs, rhs)
2319 : builder_.CreateICmpNE(lhs, rhs);
2320 else if (node.op == "<")
2321 lastValue_ = isFP ? builder_.CreateFCmpOLT(lhs, rhs)
2322 : (isUnsigned ? builder_.CreateICmpULT(lhs, rhs)
2323 : builder_.CreateICmpSLT(lhs, rhs));
2324 else if (node.op == "<=")
2325 lastValue_ = isFP ? builder_.CreateFCmpOLE(lhs, rhs)
2326 : (isUnsigned ? builder_.CreateICmpULE(lhs, rhs)
2327 : builder_.CreateICmpSLE(lhs, rhs));
2328 else if (node.op == ">")
2329 lastValue_ = isFP ? builder_.CreateFCmpOGT(lhs, rhs)
2330 : (isUnsigned ? builder_.CreateICmpUGT(lhs, rhs)
2331 : builder_.CreateICmpSGT(lhs, rhs));
2332 else if (node.op == ">=")
2333 lastValue_ = isFP ? builder_.CreateFCmpOGE(lhs, rhs)
2334 : (isUnsigned ? builder_.CreateICmpUGE(lhs, rhs)
2335 : builder_.CreateICmpSGE(lhs, rhs));
2336 else if (node.op == "+") {
2337 if (isStringType(node.left->type) || isStringType(node.right->type) ||
2338 node.left->type->getKind() == zir::TypeKind::Char ||
2339 node.right->type->getKind() == zir::TypeKind::Char) {
2340 lastValue_ = emitStringConcat(lhs, rhs, node.left->type, node.right->type,
2341 node.type);
2342 } else {
2343 lastValue_ =
2344 isFP ? builder_.CreateFAdd(lhs, rhs) : builder_.CreateAdd(lhs, rhs);
2345 }
2346 }

2380 bool old = evaluateAsAddr_;

2383 evaluateAsAddr_ = old;

2388 node.expr->accept(*this);

2413void LLVMCodeGen::visit(sema::BoundFunctionCall &node) {
2414 auto *callee = functionMap_.at(node.symbol->linkName);
2415 std::vector<llvm::Value *> args;
2416 size_t fixedParamCount = node.symbol->fixedParameterCount();
2417 for (size_t i = 0; i < node.arguments.size(); ++i) {
2418 bool isRef = false;
2419 if (i < fixedParamCount && i < node.argumentIsRef.size())
2420 isRef = node.argumentIsRef[i];
2421
2422 bool old = evaluateAsAddr_;
2423 evaluateAsAddr_ = isRef; // false for value args, true for ref args
2424
2425 node.arguments[i]->accept(*this);
2426 if (!isRef && i < fixedParamCount) {
2427 const auto &param = node.symbol->parameters[i];
2428 bool isBorrowedSelf =
2429 node.symbol->isMethod && !node.symbol->isStatic && i == 0;
2430 if (!isBorrowedSelf && isClassType(param->type) &&
2431 !isWeakClassType(param->type) &&
2432 !expressionProducesOwnedClass(node.arguments[i].get())) {
2433 emitRetainIfNeeded(lastValue_, param->type);
2434 }
2435 }
2436 args.push_back(lastValue_);
2437
2438 evaluateAsAddr_ = old;
2439 }
2440
2441 if (node.symbol->hasVariadicParameter()) {
2442 auto variadicParam = node.symbol->variadicParameter();
2443 auto *elemTy = toLLVMType(*variadicParam->variadic_element_type);
2444 auto *elemPtrTy = llvm::PointerType::getUnqual(elemTy);
2445 size_t explicitVariadicCount = node.arguments.size() - fixedParamCount;
2446
2447 llvm::Value *forwardedCount =
2448 llvm::ConstantInt::get(llvm::Type::getInt32Ty(ctx_), 0);
2449 llvm::Value *forwardedData = llvm::ConstantPointerNull::get(elemPtrTy);
2450 if (node.variadicPack) {
2451 node.variadicPack->accept(*this);
2452 forwardedData =
2453 builder_.CreateExtractValue(lastValue_, {0}, "varargs.forward.data");
2454 forwardedCount =
2455 builder_.CreateExtractValue(lastValue_, {1}, "varargs.forward.len");
2456 if (forwardedCount->getType() != llvm::Type::getInt32Ty(ctx_)) {
2457 forwardedCount = builder_.CreateIntCast(
2458 forwardedCount, llvm::Type::getInt32Ty(ctx_), /*isSigned=*/true);
2459 }
2460 }
2461
2462 llvm::Value *explicitCount =
2463 llvm::ConstantInt::get(llvm::Type::getInt32Ty(ctx_),
2464 static_cast<uint64_t>(explicitVariadicCount));
2465
2466 if (explicitVariadicCount == 0 && !node.variadicPack) {
2467 args.push_back(explicitCount);
2468 args.push_back(llvm::ConstantPointerNull::get(elemPtrTy));
2469 } else if (explicitVariadicCount == 0) {
2470 args.push_back(forwardedCount);
2471 args.push_back(forwardedData);
2472 } else {
2473 llvm::Value *totalCount = explicitCount;
2474 if (node.variadicPack) {
2475 totalCount =
2476 builder_.CreateAdd(explicitCount, forwardedCount, "varargs.total");
2477 }
2478
2479 auto *buffer = builder_.CreateAlloca(elemTy, totalCount, "varargs.buf");
2480 for (size_t i = 0; i < explicitVariadicCount; ++i) {
2481 auto *dst = builder_.CreateInBoundsGEP(
2482 elemTy, buffer,
2483 llvm::ConstantInt::get(llvm::Type::getInt32Ty(ctx_),
2484 static_cast<uint64_t>(i)));
2485 builder_.CreateStore(args[fixedParamCount + i], dst);
2486 }
2487
2488 if (node.variadicPack) {
2489 auto *copyCondBB =
2490 llvm::BasicBlock::Create(ctx_, "varargs.copy.cond", currentFn_);
2491 auto *copyBodyBB =
2492 llvm::BasicBlock::Create(ctx_, "varargs.copy.body", currentFn_);
2493 auto *copyDoneBB =
2494 llvm::BasicBlock::Create(ctx_, "varargs.copy.done", currentFn_);
2495 auto *copyPreheaderBB = builder_.GetInsertBlock();
2496
2497 builder_.CreateBr(copyCondBB);
2498 builder_.SetInsertPoint(copyCondBB);
2499 auto *indexPhi =
2500 builder_.CreatePHI(llvm::Type::getInt32Ty(ctx_), 2, "varargs.i");
2501 indexPhi->addIncoming(
2502 llvm::ConstantInt::get(llvm::Type::getInt32Ty(ctx_), 0),
2503 copyPreheaderBB);
2504 auto *hasMore = builder_.CreateICmpSLT(indexPhi, forwardedCount);
2505 builder_.CreateCondBr(hasMore, copyBodyBB, copyDoneBB);
2506
2507 builder_.SetInsertPoint(copyBodyBB);
2508 auto *src = builder_.CreateInBoundsGEP(elemTy, forwardedData, indexPhi);
2509 auto *dstIndex = builder_.CreateAdd(explicitCount, indexPhi);
2510 auto *dst = builder_.CreateInBoundsGEP(elemTy, buffer, dstIndex);
2511 auto *loaded = builder_.CreateLoad(elemTy, src);
2512 builder_.CreateStore(loaded, dst);
2513 auto *nextIndex = builder_.CreateAdd(
2514 indexPhi, llvm::ConstantInt::get(llvm::Type::getInt32Ty(ctx_), 1));
2515 builder_.CreateBr(copyCondBB);
2516 indexPhi->addIncoming(nextIndex, copyBodyBB);
2517
2518 builder_.SetInsertPoint(copyDoneBB);
2519 }
2520
2521 args.resize(fixedParamCount);
2522 args.push_back(totalCount);
2523 args.push_back(buffer);
2524 }
2525 }
2526 if (node.symbol->vtableSlot >= 0 && !args.empty()) {
2527 auto classType =
2528 std::static_pointer_cast<zir::ClassType>(node.arguments[0]->type);
2529 auto *objectTy = structCache_.at(classType->getCodegenName() + ".obj");
2530 auto *selfPtr = builder_.CreateBitCast(
2531 args[0], llvm::PointerType::getUnqual(objectTy), "method.self");
2532 auto *vtableAddr =
2533 builder_.CreateStructGEP(objectTy, selfPtr, 7, "method.vtable.addr");
2534 auto *i8PtrTy = llvm::PointerType::getUnqual(llvm::Type::getInt8Ty(ctx_));
2535 auto *vtablePtrTy = llvm::PointerType::getUnqual(i8PtrTy);
2536 auto *vtablePtr =
2537 builder_.CreateLoad(vtablePtrTy, vtableAddr, "method.vtable");
2538 auto *slotAddr = builder_.CreateInBoundsGEP(
2539 i8PtrTy, vtablePtr,
2540 llvm::ConstantInt::get(llvm::Type::getInt32Ty(ctx_),
2541 static_cast<uint64_t>(node.symbol->vtableSlot)));
2542 auto *fnRaw = builder_.CreateLoad(i8PtrTy, slotAddr, "method.fn.raw");
2543 auto *fnPtr = builder_.CreateBitCast(
2544 fnRaw, llvm::PointerType::getUnqual(callee->getFunctionType()),
2545 "method.fn");
2546 lastValue_ = builder_.CreateCall(callee->getFunctionType(), fnPtr, args);
2547 if (node.symbol->returnsRef && !evaluateAsAddr_) {
2548 lastValue_ = builder_.CreateLoad(toLLVMType(*node.symbol->returnType),
2549 lastValue_, "ref.load");
2550 }
2551 return;
2552 }
2553
2554 lastValue_ = builder_.CreateCall(callee, args);
2555
2556 if (node.symbol->returnsRef && !evaluateAsAddr_) {
2557 lastValue_ = builder_.CreateLoad(toLLVMType(*node.symbol->returnType),
2558 lastValue_, "ref.load");
2559 }
2560
2561 for (size_t i = 0; i < node.arguments.size() && i < fixedParamCount; ++i) {
2562 if (i < node.argumentIsRef.size() && node.argumentIsRef[i]) {
2563 continue;
2564 }
2565 const auto &param = node.symbol->parameters[i];
2566 if (isWeakClassType(param->type) &&
2567 expressionProducesOwnedClass(node.arguments[i].get())) {
2568 auto strongType = std::make_shared<zir::ClassType>(
2569 *std::static_pointer_cast<zir::ClassType>(param->type));
2570 strongType->setWeak(false);
2571 emitReleaseIfNeeded(args[i], strongType);
2572 }
2573 }
2574}
