87 parameters.push_back(JsonObject(std::move(parameter)));
88 }
89
90 JsonObject::Object sig;
91 sig.emplace("label", JsonObject(signature.label));
92 sig.emplace("parameters", JsonObject(std::move(parameters)));
93 signatureItems.push_back(JsonObject(std::move(sig)));
94 }
95

97 signatures.empty()
98 ? 0
99 : std::max<int64_t>(0, std::min<int64_t>(activeSignature,
100 static_cast<int64_t>(
101 signatures.size() - 1)));
102 int64_t clampedParameter = 0;
103 if (!signatures.empty()) {
104 clampedParameter = std::max<int64_t>(
105 0, std::min<int64_t>(
106 activeParameter,
107 static_cast<int64_t>(
108 signatures[clampedSignature].parameters.empty()
109 ? 0
110 : signatures[clampedSignature].parameters.size() - 1)));
111 }
112
113 JsonObject::Object result;
114 result.emplace("signatures", JsonObject(std::move(signatureItems)));
115 result.emplace("activeSignature", JsonObject(clampedSignature));
116 result.emplace("activeParameter", JsonObject(clampedParameter));
