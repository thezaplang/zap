46bool parseHex4(const char *&it, const char *end, unsigned &codepoint) {
47 codepoint = 0;
48 for (int i = 0; i < 4; ++i) {
49 if (it == end) {
50 return false;
51 }
52 codepoint <<= 4;
53 char ch = *it++;
54 if (ch >= '0' && ch <= '9') {
55 codepoint |= static_cast<unsigned>(ch - '0');
56 } else if (ch >= 'a' && ch <= 'f') {
57 codepoint |= static_cast<unsigned>(10 + ch - 'a');
58 } else if (ch >= 'A' && ch <= 'F') {
59 codepoint |= static_cast<unsigned>(10 + ch - 'A');
60 } else {
61 return false;
62 }
63 }

296void stringifyJson(std::string &out, const JsonObject &json) {
297 switch (json.getType()) {
298 case JsonObject::ObjectType::OBJECT_T: {
