41inline size_t utf8CodePointLength(unsigned char lead) {
42 if ((lead & 0x80) == 0x00)
43 return 1;
44 if ((lead & 0xE0) == 0xC0)
45 return 2;
46 if ((lead & 0xF0) == 0xE0)
47 return 3;
48 if ((lead & 0xF8) == 0xF0)
49 return 4;
50 return 1; // invalid lead byte fallback
51}

53inline uint32_t decodeUtf8At(const std::string &s, size_t i, size_t *lenOut) {
54 if (i >= s.size()) {
55 if (lenOut)
56 *lenOut = 0;
57 return 0;
58 }
59
60 const unsigned char lead = static_cast<unsigned char>(s[i]);
61 const size_t len = std::min(utf8CodePointLength(lead), s.size() - i);
62
63 if (len == 1) {
64 if (lenOut)
65 *lenOut = 1;
66 return lead;
67 }
68
69 // Validate continuation bytes; if invalid, treat as single-byte char.
70 for (size_t j = 1; j < len; ++j) {
71 if (!isUtf8ContinuationByte(static_cast<unsigned char>(s[i + j]))) {
72 if (lenOut)
73 *lenOut = 1;
74 return lead;
75 }
76 }
77
78 uint32_t cp = 0;
79 if (len == 2) {
80 cp = (lead & 0x1F);
81 } else if (len == 3) {
82 cp = (lead & 0x0F);
83 } else {
84 cp = (lead & 0x07);
85 }
86
87 for (size_t j = 1; j < len; ++j) {
88 cp = (cp << 6) | (static_cast<unsigned char>(s[i + j]) & 0x3F);
89 }
90
91 if (lenOut)
92 *lenOut = len;
93 return cp;
94}

96// A practical display-width heuristic:
97// - tabs: to next tab stop
98// - control chars: width 1
99// - combining marks: width 0
100// - common wide emoji / East-Asian wide ranges: width 2
101// - otherwise width 1
102inline bool isCombining(uint32_t cp) {
103 return (cp >= 0x0300 && cp <= 0x036F) || (cp >= 0x1AB0 && cp <= 0x1AFF) ||
104 (cp >= 0x1DC0 && cp <= 0x1DFF) || (cp >= 0x20D0 && cp <= 0x20FF) ||
105 (cp >= 0xFE20 && cp <= 0xFE2F);
106}

109 return (cp >= 0x1100 && cp <= 0x115F) || (cp >= 0x2329 && cp <= 0x232A) ||

111 (cp >= 0xF900 && cp <= 0xFAFF) || (cp >= 0xFE10 && cp <= 0xFE19) ||
112 (cp >= 0xFE30 && cp <= 0xFE6F) || (cp >= 0xFF00 && cp <= 0xFF60) ||
113 (cp >= 0xFFE0 && cp <= 0xFFE6) || (cp >= 0x1F300 && cp <= 0x1FAFF);

116inline size_t displayWidthOfCodePoint(uint32_t cp, size_t currentColumn) {
117 if (cp == '\t') {
118 const size_t t = tabWidth();
119 const size_t rem = currentColumn % t;
120 return rem == 0 ? t : (t - rem);
121 }
122 if (cp < 0x20 || cp == 0x7F)
123 return 1;
124 if (isCombining(cp))
125 return 0;
126 if (isWide(cp))
127 return 2;
128 return 1;
129}

171inline size_t visualColumnsForBytes(const std::string &line, size_t bytesUpto) {
172 size_t i = 0;
173 size_t col = 0;
174 const size_t end = std::min(bytesUpto, line.size());
175
176 while (i < end) {
177 size_t len = 0;
178 uint32_t cp = decodeUtf8At(line, i, &len);
179 if (len == 0)
180 break;
181
182 // Prevent stepping beyond requested byte count.
183 if (i + len > end)
184 break;
185
186 col += displayWidthOfCodePoint(cp, col);
187 i += len;
188 }
189
190 return col;
191}

193inline std::string expandTabsForDisplay(const std::string &line) {
194 std::string out;
195 out.reserve(line.size());
196
197 size_t i = 0;
198 size_t col = 0;
199 while (i < line.size()) {
200 size_t len = 0;
201 uint32_t cp = decodeUtf8At(line, i, &len);
202 if (len == 0)
203 break;
204
205 if (cp == '\t') {
206 const size_t spaces = displayWidthOfCodePoint(cp, col);
207 out.append(spaces, ' ');
208 col += spaces;
209 i += len;
210 continue;
211 }
212
213 out.append(line, i, len);
214 col += displayWidthOfCodePoint(cp, col);
215 i += len;
216 }
217
218 return out;
219}

221inline std::pair<size_t, size_t>
222computeUnderlineColumns(const SourceSpan &span, const std::string &lineContent,
223 size_t lineStartOffset) {
224 size_t startByteInLine = 0;
225 if (span.offset >= lineStartOffset) {
226 startByteInLine = span.offset - lineStartOffset;
227 } else if (span.column > 0) {
228 // Approx fallback when offset is unavailable: best-effort by characters.
229 startByteInLine = std::min(span.column - 1, lineContent.size());
230 }
231 startByteInLine = clampToCodePointBoundary(lineContent, startByteInLine);
232 startByteInLine = std::min(startByteInLine, lineContent.size());
233
234 size_t endByteInLine = startByteInLine;
235 if (span.length > 0) {
236 size_t desired = startByteInLine + span.length;
237 endByteInLine =
238 clampToCodePointBoundary(lineContent, std::min(desired, lineContent.size()));
239 } else {
240 // zero-length span: point to the next visible cell
241 endByteInLine = startByteInLine;
242 }
243
244 size_t startCol = visualColumnsForBytes(lineContent, startByteInLine);
245 size_t endCol = visualColumnsForBytes(lineContent, endByteInLine);
246
247 size_t width = (endCol > startCol) ? (endCol - startCol) : 1;
248 return {startCol, width};
249}

421 const std::string &message) {

423 for (const auto &existing : diagnostics_) {
424 if (existing.level == level && existing.message == message &&
425 existing.span.line == span.line &&
426 existing.span.column == span.column &&

661 DiagnosticRange makeRange(SourceSpan span) const {
662 DiagnosticPosition start{span.line, span.column, span.offset};
663
664 size_t endOffset = std::min(span.offset, source.size());
665 size_t remaining = span.length;
666
667 size_t line = start.line;
668 size_t column = start.column;
669
670 while (endOffset < source.size() && remaining > 0) {
671 if (source[endOffset] == '\n') {
672 ++line;
673 column = 1;
674 } else {
675 ++column;
676 }
677 ++endOffset;
678 --remaining;
679 }
680
681 if (span.length == 0) {
682 ++column;
683 }
684
685 return DiagnosticRange{start, DiagnosticPosition{line, column, endOffset}};
