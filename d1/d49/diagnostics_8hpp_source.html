45inline size_t utf8CodePointLength(unsigned char lead) {
46 if ((lead & 0x80) == 0x00)
47 return 1;
48 if ((lead & 0xE0) == 0xC0)
49 return 2;
50 if ((lead & 0xF0) == 0xE0)
51 return 3;
52 if ((lead & 0xF8) == 0xF0)
53 return 4;
54 return 1; // invalid lead byte fallback
55}

57inline uint32_t decodeUtf8At(const std::string &s, size_t i, size_t *lenOut) {
58 if (i >= s.size()) {
59 if (lenOut)
60 *lenOut = 0;
61 return 0;
62 }
63
64 const unsigned char lead = static_cast<unsigned char>(s[i]);
65 const size_t len = std::min(utf8CodePointLength(lead), s.size() - i);
66
67 if (len == 1) {
68 if (lenOut)
69 *lenOut = 1;
70 return lead;
71 }
72
73 // Validate continuation bytes; if invalid, treat as single-byte char.
74 for (size_t j = 1; j < len; ++j) {
75 if (!isUtf8ContinuationByte(static_cast<unsigned char>(s[i + j]))) {
76 if (lenOut)
77 *lenOut = 1;
78 return lead;
79 }
80 }
81
82 uint32_t cp = 0;
83 if (len == 2) {
84 cp = (lead & 0x1F);
85 } else if (len == 3) {
86 cp = (lead & 0x0F);
87 } else {
88 cp = (lead & 0x07);
89 }
90
91 for (size_t j = 1; j < len; ++j) {
92 cp = (cp << 6) | (static_cast<unsigned char>(s[i + j]) & 0x3F);
93 }
94
95 if (lenOut)
96 *lenOut = len;
97 return cp;
98}

100// A practical display-width heuristic:
101// - tabs: to next tab stop
102// - control chars: width 1
103// - combining marks: width 0
104// - common wide emoji / East-Asian wide ranges: width 2
105// - otherwise width 1
106inline bool isCombining(uint32_t cp) {
107 return (cp >= 0x0300 && cp <= 0x036F) || (cp >= 0x1AB0 && cp <= 0x1AFF) ||
108 (cp >= 0x1DC0 && cp <= 0x1DFF) || (cp >= 0x20D0 && cp <= 0x20FF) ||
109 (cp >= 0xFE20 && cp <= 0xFE2F);
110}

112inline bool isWide(uint32_t cp) {
113 return (cp >= 0x1100 && cp <= 0x115F) || (cp >= 0x2329 && cp <= 0x232A) ||
114 (cp >= 0x2E80 && cp <= 0xA4CF) || (cp >= 0xAC00 && cp <= 0xD7A3) ||
115 (cp >= 0xF900 && cp <= 0xFAFF) || (cp >= 0xFE10 && cp <= 0xFE19) ||
116 (cp >= 0xFE30 && cp <= 0xFE6F) || (cp >= 0xFF00 && cp <= 0xFF60) ||
117 (cp >= 0xFFE0 && cp <= 0xFFE6) || (cp >= 0x1F300 && cp <= 0x1FAFF);

120inline size_t displayWidthOfCodePoint(uint32_t cp, size_t currentColumn) {
121 if (cp == '\t') {
122 const size_t t = tabWidth();
123 const size_t rem = currentColumn % t;
124 return rem == 0 ? t : (t - rem);
125 }
126 if (cp < 0x20 || cp == 0x7F)
127 return 1;
128 if (isCombining(cp))
129 return 0;
130 if (isWide(cp))
131 return 2;
132 return 1;
133}

175inline size_t visualColumnsForBytes(const std::string &line, size_t bytesUpto) {
176 size_t i = 0;
177 size_t col = 0;
178 const size_t end = std::min(bytesUpto, line.size());
179
180 while (i < end) {
181 size_t len = 0;
182 uint32_t cp = decodeUtf8At(line, i, &len);
183 if (len == 0)
184 break;
185
186 // Prevent stepping beyond requested byte count.
187 if (i + len > end)
188 break;
189
190 col += displayWidthOfCodePoint(cp, col);
191 i += len;
192 }
193
194 return col;
195}

197inline std::string expandTabsForDisplay(const std::string &line) {
198 std::string out;
199 out.reserve(line.size());
200
201 size_t i = 0;
202 size_t col = 0;
203 while (i < line.size()) {
204 size_t len = 0;
205 uint32_t cp = decodeUtf8At(line, i, &len);
206 if (len == 0)
207 break;
208
209 if (cp == '\t') {
210 const size_t spaces = displayWidthOfCodePoint(cp, col);
211 out.append(spaces, ' ');
212 col += spaces;
213 i += len;
214 continue;
215 }
216
217 out.append(line, i, len);
218 col += displayWidthOfCodePoint(cp, col);
219 i += len;
220 }
221
222 return out;
223}

225inline std::pair<size_t, size_t>
226computeUnderlineColumns(const SourceSpan &span, const std::string &lineContent,
227 size_t lineStartOffset) {
228 size_t startByteInLine = 0;
229 if (span.offset >= lineStartOffset) {
230 startByteInLine = span.offset - lineStartOffset;
231 } else if (span.column > 0) {
232 // Approx fallback when offset is unavailable: best-effort by characters.
233 startByteInLine = std::min(span.column - 1, lineContent.size());
234 }
235 startByteInLine = clampToCodePointBoundary(lineContent, startByteInLine);
236 startByteInLine = std::min(startByteInLine, lineContent.size());
237
238 size_t endByteInLine = startByteInLine;
239 if (span.length > 0) {
240 size_t desired = startByteInLine + span.length;
241 endByteInLine = clampToCodePointBoundary(
242 lineContent, std::min(desired, lineContent.size()));
243 } else {
244 // zero-length span: point to the next visible cell
245 endByteInLine = startByteInLine;
246 }
247
248 size_t startCol = visualColumnsForBytes(lineContent, startByteInLine);
249 size_t endCol = visualColumnsForBytes(lineContent, endByteInLine);
250
251 size_t width = (endCol > startCol) ? (endCol - startCol) : 1;
252 return {startCol, width};
253}

432 void report(SourceSpan span, DiagnosticLevel level, const std::string &code,
433 const std::string &message) {
434 // Deduplicate identical diagnostics (same level, message and span).
435 for (const auto &existing : diagnostics_) {
436 if (existing.level == level && existing.message == message &&
437 existing.span.line == span.line &&
438 existing.span.column == span.column &&
439 existing.span.offset == span.offset &&
440 existing.span.length == span.length) {
441 return;
442 }
443 }
444
445 // Stop flooding on errors, but emit one explanatory note once.
446 if (level == DiagnosticLevel::Error && errorCount >= MAX_ERRORS) {
447 if (!errorCapReached_) {
448 errorCapReached_ = true;
449 const SourceSpan capSpan = span;
450 const auto &capSource = sourceFor(capSpan);
451 diagnostics_.push_back(Diagnostic{
452 DiagnosticLevel::Note, "N0001",
453 "Too many errors emitted; stopping after " +
454 std::to_string(MAX_ERRORS) +
455 " errors. Fix the first errors and re-run for more details.",
456 fileNameFor(capSpan), capSource, capSpan,
457 makeRange(capSpan, capSource)});
458 }
459 return;
460 }
461
462 if (diagnostics_.size() >= MAX_DIAGNOSTICS) {
463 return;
464 }
465
466 const auto &diagnosticSource = sourceFor(span);
467 diagnostics_.push_back(Diagnostic{level, code, message, fileNameFor(span),

700 DiagnosticRange makeRange(SourceSpan span,
701 const std::string &rangeSource) const {
702 DiagnosticPosition start{span.line, span.column, span.offset};
703
704 size_t endOffset = std::min(span.offset, rangeSource.size());
705 size_t remaining = span.length;
706
707 size_t line = start.line;
708 size_t column = start.column;
709
710 while (endOffset < rangeSource.size() && remaining > 0) {
711 if (rangeSource[endOffset] == '\n') {
712 ++line;
713 column = 1;
714 } else {
715 ++column;
716 }
717 ++endOffset;
718 --remaining;
719 }
720
721 if (span.length == 0) {
722 ++column;
723 }
724
725 return DiagnosticRange{start, DiagnosticPosition{line, column, endOffset}};
