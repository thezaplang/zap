158void collectFromBody(const BodyNode *body, size_t offset,
159 const std::string &uri, std::vector<LspSymbol> &symbols) {
160 if (!bodyContainsOffset(body, offset)) {
161 return;
162 }
163
164 for (const auto &statement : body->statements) {
165 if (!statement || statement->span.offset > offset) {
166 break;
167 }
168 collectFromStatement(statement.get(), offset, uri, symbols);
169 }
170 collectFromExpression(body->result.get(), offset, uri, symbols);

189std::vector<LspSymbol> collectLocalSymbols(const RootNode &root, size_t offset,
190 const std::string &uri) {
191 std::vector<LspSymbol> symbols;
192 for (const auto &child : root.children) {
193 if (!child || !containsOffset(child->span, offset)) {
194 continue;
195 }
196 if (auto fun = dynamic_cast<const FunDecl *>(child.get())) {
197 collectFunctionLocals(fun, offset, uri, symbols);
198 break;
199 }
200 if (auto cls = dynamic_cast<const ClassDecl *>(child.get())) {
201 for (const auto &field : cls->fields_) {
202 if (field && field->span.offset <= offset) {
203 addSymbol(symbols, uri, field->name, field->span, 8,
204 field->visibility_);
205 }
206 }
207 for (const auto &method : cls->methods_) {
208 collectFunctionLocals(method.get(), offset, uri, symbols);
209 }
210 break;
211 }
212 }
