743 if (expressionStack_.empty())
744 return;
745 auto left = std::move(expressionStack_.top());
746 expressionStack_.pop();
747
748 if (left->type->getKind() != zir::TypeKind::Array)
749 {
750 error(node.span, "Type '" + left->type->toString() + "' does not support indexing.");
751 return;
752 }
753
754 node.index_->accept(*this);
755 if (expressionStack_.empty())
756 return;
757 auto index = std::move(expressionStack_.top());
758 expressionStack_.pop();
759
760 if (!index->type->isInteger())
761 {
762 error(node.span, "Array index must be an integer, but got '" + index->type->toString() + "'");
763 }
764
765 auto arrayType = std::static_pointer_cast<zir::ArrayType>(left->type);
766 expressionStack_.push(std::make_unique<BoundIndexAccess>(std::move(left), std::move(index), arrayType->getBaseType()));
767 }

769 void Binder::visit(MemberAccessNode &node)
770 {
771 node.left_->accept(*this);
772 if (expressionStack_.empty())
773 return;
774
775 auto left = std::move(expressionStack_.top());
776 expressionStack_.pop();
777
778 if (left->type->getKind() == zir::TypeKind::Enum)
779 {

781 int value = enumType->getVariantIndex(node.member_);
782 if (value != -1)
783 {
784 expressionStack_.push(std::make_unique<BoundLiteral>(
785 std::to_string(value),
786 enumType));

788 }
789 }
790 else if (left->type->getKind() == zir::TypeKind::Record)
791 {
792 auto recordType = std::static_pointer_cast<zir::RecordType>(left->type);
793 for (const auto &field : recordType->getFields())
794 {
795 if (field.name == node.member_)
796 {
797 expressionStack_.push(std::make_unique<BoundMemberAccess>(
798 std::move(left), node.member_, field.type));
799 return;
800 }
801 }
802 }
803
804 error(node.span, "Member '" + node.member_ + "' not found in type '" +
805 left->type->toString() + "'");
806 }

808 void Binder::visit(FunCall &node)
809 {
810 auto symbol = currentScope_->lookup(node.funcName_);
811 if (!symbol)
812 {
813 error(node.span, "Undefined function: " + node.funcName_);
814 return;
815 }
816
817 auto funcSymbol = std::dynamic_pointer_cast<FunctionSymbol>(symbol);
818 if (!funcSymbol)
819 {
820 error(node.span, "'" + node.funcName_ + "' is not a function.");
821 return;
822 }
823
824 if (node.params_.size() != funcSymbol->parameters.size())
825 {
826 error(node.span, "Function '" + node.funcName_ + "' expects " +
827 std::to_string(funcSymbol->parameters.size()) +
828 " arguments, but received " +
829 std::to_string(node.params_.size()));
830 }
831
832 std::vector<std::unique_ptr<BoundExpression>> boundArgs;
833 std::vector<bool> argIsRefList;
834 std::vector<std::shared_ptr<VariableSymbol>> refSymbols;
835
836 for (size_t i = 0; i < node.params_.size(); ++i)
837 {
838 node.params_[i]->value->accept(*this);
839 if (expressionStack_.empty())
840 return;
841 auto arg = std::move(expressionStack_.top());
842 expressionStack_.pop();
843
844 bool paramExpectsRef = false;
845 if (i < funcSymbol->parameters.size())
846 {
847 paramExpectsRef = funcSymbol->parameters[i]->is_ref;
848 if (node.params_[i]->isRef != paramExpectsRef)
849 {
850 if (paramExpectsRef)
851 error(node.params_[i]->value->span, "Argument " + std::to_string(i + 1) + " of function '" + node.funcName_ + "' must be passed with 'ref'.");
852 else
853 error(node.params_[i]->value->span, "Argument " + std::to_string(i + 1) + " of function '" + node.funcName_ + "' should not be passed with 'ref'.");
854 }
855
856 if (paramExpectsRef)
857 {
858 // Check if l-value
859 bool isLValue = false;
860 std::shared_ptr<VariableSymbol> varSym = nullptr;
861
862 if (auto varExpr = dynamic_cast<BoundVariableExpression *>(arg.get()))
863 {
864 isLValue = true;
865 varSym = varExpr->symbol;
866 }
867 else if (dynamic_cast<BoundIndexAccess *>(arg.get()))
868 isLValue = true;
869 else if (dynamic_cast<BoundMemberAccess *>(arg.get()))
870 isLValue = true;
871
872 if (!isLValue)
873 {
874 error(node.params_[i]->value->span, "Argument " + std::to_string(i + 1) + " passed by 'ref' must be an l-value.");
875 }
876 else if (varSym)
877 {
878 if (varSym->is_const)
879 {
880 error(node.params_[i]->value->span, "Cannot pass constant '" + varSym->name + "' by 'ref'.");
881 }
882
883 // Aliasing check
884 for (const auto& s : refSymbols)
885 {
886 if (s == varSym)
887 {
888 error(node.params_[i]->value->span, "Variable '" + varSym->name + "' is already passed as a 'ref' argument in this call.");
889 }
890 }
891 refSymbols.push_back(varSym);
892 }
893 }
894
895 auto expectedType = funcSymbol->parameters[i]->type;
896 if (!canConvert(arg->type, expectedType))
897 {
898 error(node.params_[i]->value->span, "Argument " + std::to_string(i + 1) +
899 " of function '" + node.funcName_ +
900 "' expected type '" + expectedType->toString() +
901 "', but received type '" + arg->type->toString() +
902 "'");
903 }
904 else if (!paramExpectsRef) // Don't wrap in cast if it's a ref
905 {
906 arg = wrapInCast(std::move(arg), expectedType);
907 }
908 else if (arg->type->toString() != expectedType->toString())
909 {
910 error(node.params_[i]->value->span, "Argument " + std::to_string(i + 1) + " passed by 'ref' must match the parameter type exactly. Expected '" + expectedType->toString() + "', but got '" + arg->type->toString() + "'.");
911 }
912 }
913 boundArgs.push_back(std::move(arg));
914 argIsRefList.push_back(node.params_[i]->isRef);
915 }
916
917 expressionStack_.push(
918 std::make_unique<BoundFunctionCall>(funcSymbol, std::move(boundArgs), std::move(argIsRefList)));
919 }

921 void Binder::visit(IfNode &node)
922 {

924 {
925 error(node.span, "If condition is missing.");
926 return;
927 }
928
929 node.condition_->accept(*this);
930 if (expressionStack_.empty())
931 return;
932 auto cond = std::move(expressionStack_.top());
933 expressionStack_.pop();
934
935 if (cond->type->getKind() != zir::TypeKind::Bool)

942 node.thenBody_->accept(*this);

948 if (node.elseBody_)
949 {
950 pushScope();

952 node.elseBody_->accept(*this);
953 elseBound = std::move(currentBlock_);
954 currentBlock_ = std::move(savedMainBlockElse);
955 popScope();
956 }
957
958 if (thenBound)
959 thenBound->result.reset();
960 if (elseBound)
961 elseBound->result.reset();
962
963 statementStack_.push(std::make_unique<BoundIfStatement>(
964 std::move(cond), std::move(thenBound), std::move(elseBound)));

1041 std::optional<int64_t> Binder::evaluateConstantInt(const BoundExpression *expr)
1042 {
1043 if (!expr)
1044 return std::nullopt;
1045
1046 if (auto literal = dynamic_cast<const BoundLiteral *>(expr))
1047 {
1048 try
1049 {
1050 return std::stoll(literal->value);
1051 }
1052 catch (...)
1053 {
1054 return std::nullopt;
1055 }
1056 }
1057
1058 if (auto varExpr = dynamic_cast<const BoundVariableExpression *>(expr))
1059 {
1060 if (varExpr->symbol->is_const && varExpr->symbol->constant_value)
1061 {
1062 return evaluateConstantInt(varExpr->symbol->constant_value.get());
1063 }
1064 }
1065
1066 if (auto cast = dynamic_cast<const BoundCast *>(expr))
1067 {
1068 return evaluateConstantInt(cast->expression.get());
1069 }
1070
1071 if (auto binary = dynamic_cast<const BoundBinaryExpression *>(expr))
1072 {
1073 auto left = evaluateConstantInt(binary->left.get());
1074 auto right = evaluateConstantInt(binary->right.get());
1075 if (left && right)
1076 {
1077 if (binary->op == "+")
1078 return *left + *right;
1079 if (binary->op == "-")
1080 return *left - *right;
1081 if (binary->op == "*")
1082 return *left * *right;
1083 if (binary->op == "/")
1084 return *right != 0 ? std::make_optional(*left / *right) : std::nullopt;
1085 }
1086 }
1087
1088 if (auto unary = dynamic_cast<const BoundUnaryExpression *>(expr))
1089 {
1090 auto val = evaluateConstantInt(unary->expr.get());
1091 if (val)
1092 {
1093 if (unary->op == "-")
1094 return -*val;
1095 if (unary->op == "+")
1096 return *val;
1097 }
1098 }
1099
1100 return std::nullopt;
1101 }

1104 {
1105 if (typeNode.isArray)
1106 {
1107 if (!typeNode.baseType)
1108 return nullptr;

1110 size_t size = 0;
1111
1112 if (typeNode.arraySize)
1113 {
1114 typeNode.arraySize->accept(*this);
1115 if (!expressionStack_.empty())
1116 {
1117 auto boundSize = std::move(expressionStack_.top());
1118 expressionStack_.pop();
1119 auto evaluated = evaluateConstantInt(boundSize.get());
1120 if (evaluated)
1121 {
1122 size = static_cast<size_t>(*evaluated);
1123 }
1124 else
1125 {
1126 error(typeNode.span, "Array size must be a constant integer expression.");
1127 }
1128 }
1129 }

1170 type = std::make_shared<zir::PrimitiveType>(zir::TypeKind::UInt64);
1171 else if (typeNode.typeName == "Float")

1300 void Binder::visit(StructLiteralNode &node)
1301 {
1302 auto symbol = currentScope_->lookup(node.type_name_);
1303 if (!symbol || symbol->getKind() != SymbolKind::Type)
1304 {
1305 error(node.span, "Unknown type: " + node.type_name_);
1306 return;
1307 }
1308
1309 auto typeSymbol = std::static_pointer_cast<TypeSymbol>(symbol);
1310 if (typeSymbol->type->getKind() != zir::TypeKind::Record)
1311 {
1312 error(node.span, "'" + node.type_name_ + "' is not a struct.");
1313 return;
1314 }
1315
1316 auto recordType = std::static_pointer_cast<zir::RecordType>(typeSymbol->type);
1317 std::vector<std::pair<std::string, std::unique_ptr<BoundExpression>>> boundFields;
1318
1319 for (auto &fieldInit : node.fields_)
1320 {
1321 fieldInit.value->accept(*this);
1322 if (expressionStack_.empty())
1323 continue;
1324 auto boundVal = std::move(expressionStack_.top());
1325 expressionStack_.pop();
1326
1327 // Check if field exists
1328 bool found = false;
1329 for (const auto &f : recordType->getFields())
1330 {
1331 if (f.name == fieldInit.name)
1332 {
1333 if (!canConvert(boundVal->type, f.type))
1334 {
1335 error(node.span, "Cannot assign type '" + boundVal->type->toString() +
1336 "' to field '" + f.name + "' of type '" +
1337 f.type->toString() + "'");
1338 }
1339 found = true;
1340 break;
1341 }
1342 }
1343
1344 if (!found)
1345 {
1346 error(node.span, "Field '" + fieldInit.name + "' not found in struct '" +
1347 node.type_name_ + "'");
1348 }
1349
1350 boundFields.push_back({fieldInit.name, std::move(boundVal)});
1351 }
1352
1353 // Check if all fields are initialized
1354 const auto &allFields = recordType->getFields();
1355 if (boundFields.size() < allFields.size())
1356 {
1357 for (const auto &f : allFields)
1358 {
1359 bool initialized = false;
1360 for (const auto &bf : boundFields)
1361 {
1362 if (bf.first == f.name)
1363 {
1364 initialized = true;
1365 break;
1366 }
1367 }
1368 if (!initialized)
1369 {
1370 error(node.span, "Field '" + f.name + "' of struct '" + node.type_name_ + "' is not initialized.");
1371 }
1372 }
1373 }
1374 else if (boundFields.size() > allFields.size())
1375 {
1376 // This should have been caught during the field name lookup loop,
1377 // but just in case we have duplicates or something.
1378 }
1379
1380 expressionStack_.push(std::make_unique<BoundStructLiteral>(std::move(boundFields), recordType));
1381 }

1398 bool Binder::canConvert(std::shared_ptr<zir::Type> from,
1399 std::shared_ptr<zir::Type> to)
1400 {
1401 if (from->getKind() == to->getKind())
1402 {
1403 if (from->getKind() == zir::TypeKind::Record)
1404 {
1405 return from->toString() == to->toString();
1406 }
1407 if (from->getKind() == zir::TypeKind::Array)
1408 {
1409 auto a1 = std::static_pointer_cast<zir::ArrayType>(from);
1410 auto a2 = std::static_pointer_cast<zir::ArrayType>(to);
1411 return a1->getSize() == a2->getSize() &&
1412 canConvert(a1->getBaseType(), a2->getBaseType());
1413 }
1414 return true;
1415 }
1416
1417 if (from->getKind() == zir::TypeKind::Enum &&
1418 (to->isInteger() || to->getKind() == zir::TypeKind::Enum))
1419 {
1420 return true;
1421 }
1422
1423 if (from->isInteger() && to->isInteger())
1424 {
1425 return true;
1426 }
1427
1428 if (from->isInteger() &&
1429 to->isFloatingPoint())
1430 {
1431 return true;
1432 }
1433
1434 if (from->isFloatingPoint() && to->isFloatingPoint())
1435 {
1436 if (from->getKind() == zir::TypeKind::Float64)
1437 return to->getKind() == zir::TypeKind::Float64;
1438 return true;
1439 }
1440
1441 return false;
1442 }

1444 std::shared_ptr<zir::Type>
1445 Binder::getPromotedType(std::shared_ptr<zir::Type> t1,
1446 std::shared_ptr<zir::Type> t2)
1447 {
1448 if (t1->isFloatingPoint() || t2->isFloatingPoint())
1449 {
1450 if (t1->getKind() == zir::TypeKind::Float64 ||
1451 t2->getKind() == zir::TypeKind::Float64)
1452 {
1453 return std::make_shared<zir::PrimitiveType>(zir::TypeKind::Float64);
1454 }
1455 return std::make_shared<zir::PrimitiveType>(zir::TypeKind::Float);
1456 }
1457
1458 if (t1->isInteger() && t2->isInteger()) {
1459 auto getWidth = [](zir::TypeKind k) {
1460 switch (k) {
