574 mangled += c;
575 return;
576 }
577 char buf[8];
578 std::snprintf(buf, sizeof(buf), "_%02X", static_cast<unsigned int>(uc));
579 mangled += buf;
580 };
581 for (char c : modulePath) {
582 appendEscaped(c);

585 mangled += '$';
586 }

588 appendEscaped(c);
589 }

686 const FunctionSymbol &rhs) {
687 const size_t lhsOffset = lhs.isMethod ? 1 : 0;
688 const size_t rhsOffset = rhs.isMethod ? 1 : 0;
689 if (lhs.parameters.size() < lhsOffset || rhs.parameters.size() < rhsOffset) {
690 return false;

693 lhs.isCVariadic != rhs.isCVariadic) {

697 const auto &left = lhs.parameters[i + lhsOffset];
698 const auto &right = rhs.parameters[i + rhsOffset];

700 left->is_variadic_pack != right->is_variadic_pack || !left->type ||
701 !right->type || left->type->toString() != right->type->toString()) {
702 return false;

720 updated->overloads = overloads->overloads;
721 } else if (auto existing = std::dynamic_pointer_cast<FunctionSymbol>(entry)) {
722 updated->addOverload(existing);
723 }
724 updated->overloads.erase(
725 std::remove_if(updated->overloads.begin(), updated->overloads.end(),
726 [&](const std::shared_ptr<FunctionSymbol> &candidate) {
727 return candidate &&
728 sameMethodDispatchSignature(*candidate, *method);
729 }),
730 updated->overloads.end());
731 updated->addOverload(method);
732 entry = updated;
733 return updated;
734}

745 sameMethodDispatchSignature(*candidate, method)) {

844std::shared_ptr<Symbol>
845Binder::resolveModuleMember(const std::string &moduleName,
846 const std::string &memberName, SourceSpan span) {
847 auto moduleSym = std::dynamic_pointer_cast<ModuleSymbol>(
848 currentScope_->lookup(moduleName));
849 if (!moduleSym) {
850 error(span, "Undefined module: " + moduleName);
851 return nullptr;
852 }
853
854 auto exportedIt = moduleSym->exports.find(memberName);
855 if (exportedIt != moduleSym->exports.end()) {
856 return exportedIt->second;
857 }
858
859 auto memberIt = moduleSym->members.find(memberName);
860 if (memberIt == moduleSym->members.end()) {
861 error(span,
862 "Module '" + moduleName + "' has no member '" + memberName + "'.");
863 return nullptr;
864 }
865 error(span, "Member '" + memberName + "' of module '" + moduleName +
866 "' is private.");

870std::shared_ptr<Symbol>
871Binder::resolveQualifiedSymbol(const std::vector<std::string> &parts,
872 SourceSpan span, SymbolKind expectedKind,
873 bool allowAnyKind) {
874 if (parts.empty()) {
875 return nullptr;
876 }
877
878 auto symbol = lookupVisibleSymbol(parts.front());
879 if (!symbol) {
880 error(span, "Undefined identifier: " + parts.front());
881 return nullptr;
882 }
883
884 for (size_t i = 1; i < parts.size(); ++i) {
885 auto moduleSym = std::dynamic_pointer_cast<ModuleSymbol>(symbol);
886 if (!moduleSym) {
887 error(span, "'" + parts[i - 1] + "' is not a module.");
888 return nullptr;
889 }
890
891 auto memberIt = moduleSym->exports.find(parts[i]);
892 if (memberIt == moduleSym->exports.end()) {
893 auto privateIt = moduleSym->members.find(parts[i]);
894 if (privateIt != moduleSym->members.end()) {
895 error(span, "Member '" + parts[i] + "' of module '" + moduleSym->name +
896 "' is private.");
897 } else {
898 error(span, "Module '" + moduleSym->name + "' has no member '" +
899 parts[i] + "'.");
900 }
901 return nullptr;
902 }
903
904 symbol = memberIt->second;
905 }
906
907 if (!allowAnyKind && symbol->getKind() != expectedKind &&
908 !(expectedKind == SymbolKind::Function &&
909 symbol->getKind() == SymbolKind::OverloadSet)) {
910 return nullptr;
911 }
912 return symbol;
913}

935std::optional<int64_t>
936Binder::evaluateConstantInt(const BoundExpression *expr) {
937 if (auto lit = dynamic_cast<const BoundLiteral *>(expr)) {
938 try {
939 const std::string &v = lit->value;
940 if (v.size() > 2 && v[0] == '0') {
941 if (v[1] == 'x' || v[1] == 'X') {
942 return static_cast<int64_t>(std::stoull(v, nullptr, 16));
943 }
944 if (v[1] == 'b' || v[1] == 'B') {
945 return static_cast<int64_t>(std::stoull(v.substr(2), nullptr, 2));
946 }
947 if (v[1] == 'o' || v[1] == 'O') {
948 return static_cast<int64_t>(std::stoull(v.substr(2), nullptr, 8));
949 }
950 }
951 return std::stoll(v);
952 } catch (...) {
953 return std::nullopt;
954 }
955 }
956
957 if (auto unary = dynamic_cast<const BoundUnaryExpression *>(expr)) {
958 auto inner = evaluateConstantInt(unary->expr.get());
959 if (!inner)
960 return std::nullopt;
961 if (unary->op == "-")
962 return -*inner;
963 return inner;
964 }
965
966 if (auto binary = dynamic_cast<const BoundBinaryExpression *>(expr)) {
967 auto left = evaluateConstantInt(binary->left.get());
968 auto right = evaluateConstantInt(binary->right.get());
969 if (!left || !right)
970 return std::nullopt;
971
972 if (binary->op == "+")
973 return *left + *right;
974 if (binary->op == "-")
975 return *left - *right;
976 if (binary->op == "*")
977 return *left * *right;
978 if (binary->op == "/")
979 return *right == 0 ? std::nullopt
980 : std::optional<int64_t>(*left / *right);
981 if (binary->op == "%")
982 return *right == 0 ? std::nullopt
983 : std::optional<int64_t>(*left % *right);
984 if (binary->op == "&")
985 return *left & *right;
986 if (binary->op == "|")
987 return *left | *right;
988 if (binary->op == "^")
989 return *left ^ *right;
990 if (binary->op == "<<") {
991 if (*right < 0)
992 return std::nullopt;
993 unsigned width =
994 binary->left->type
995 ? static_cast<unsigned>(typeBitWidth(binary->left->type))
996 : 0u;
997 if (width == 0 || static_cast<uint64_t>(*right) >= width)
998 return std::nullopt;
999 return static_cast<int64_t>(static_cast<uint64_t>(*left)
1000 << static_cast<uint64_t>(*right));
1001 }
1002 if (binary->op == ">>") {
1003 if (*right < 0)
1004 return std::nullopt;
1005 unsigned width =
1006 binary->left->type
1007 ? static_cast<unsigned>(typeBitWidth(binary->left->type))
1008 : 0u;
1009 if (width == 0 || static_cast<uint64_t>(*right) >= width)
1010 return std::nullopt;
1011 return *left >> static_cast<uint64_t>(*right);
1012 }
1013 }
1014
1015 return std::nullopt;
1016}

1018std::unique_ptr<BoundExpression>
1019Binder::foldConstantBinary(const BoundBinaryExpression *binary) {
1020 if (!binary)
1021 return nullptr;
1022
1023 auto *left = dynamic_cast<const BoundLiteral *>(binary->left.get());
1024 auto *right = dynamic_cast<const BoundLiteral *>(binary->right.get());
1025 if (!left || !right)
1026 return nullptr;
1027
1028 // String literal concatenation.
1029 if (binary->op == "+" && isStringType(binary->type)) {
1030 if (isStringType(left->type) && isStringType(right->type)) {
1031 return std::make_unique<BoundLiteral>(
1032 left->value + right->value,
1033 std::make_shared<zir::RecordType>("StringView", "StringView"));
1034 }
1035 return nullptr;
1036 }
1037
1038 // Integer arithmetic/bitwise.
1039 if (binary->type && binary->type->isInteger()) {
1040 if (auto value = evaluateConstantInt(binary))
1041 return std::make_unique<BoundLiteral>(std::to_string(*value),
1042 binary->type);
1043 }
1044
1045 return nullptr;
1046}
