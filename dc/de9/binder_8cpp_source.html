408int Binder::conversionCost(std::shared_ptr<zir::Type> from,
409 std::shared_ptr<zir::Type> to) const {
410 if (!from || !to) {
411 return 1000;
412 }
413 if (from->toString() == to->toString()) {
414 return 0;
415 }
416 if (isNullType(from) && isPointerType(to)) {
417 return 3;
418 }
419 if (!canConvert(from, to)) {
420 return 1000;
421 }
422
423 if (from->isFloatingPoint() && to->isFloatingPoint()) {
424 return typeBitWidth(to) >= typeBitWidth(from) ? 1 : 5;
425 }
426
427 if (isSignedIntegerType(from) && isSignedIntegerType(to)) {
428 return typeBitWidth(to) >= typeBitWidth(from) ? 1 : 4;
429 }
430
431 if (isUnsignedIntegerType(from) && isUnsignedIntegerType(to)) {
432 return typeBitWidth(to) >= typeBitWidth(from) ? 1 : 4;
433 }
434
435 if (from->isInteger() && to->isFloatingPoint()) {
436 return typeBitWidth(to) >= typeBitWidth(from) ? 2 : 5;
437 }
438
439 if (from->isFloatingPoint() && to->isInteger()) {
440 return 6;
441 }
442
443 if (from->isInteger() && to->isInteger()) {
444 return 5;
445 }
446
447 return 7;
448}

450std::string Binder::describeConversion(std::shared_ptr<zir::Type> from,
451 std::shared_ptr<zir::Type> to) const {
452 if (!from || !to) {
453 return "invalid conversion";
454 }
455 int cost = conversionCost(from, to);
456 if (cost == 0) {
457 return "exact match";
458 }
459 if (isNullType(from) &&
460 (isPointerType(to) || to->getKind() == zir::TypeKind::Class)) {
461 return "null-to-reference conversion";
462 }
463 if (from->isFloatingPoint() && to->isFloatingPoint()) {
464 return typeBitWidth(to) >= typeBitWidth(from) ? "floating widening"
465 : "floating narrowing";
466 }
467 if (isSignedIntegerType(from) && isSignedIntegerType(to)) {
468 return typeBitWidth(to) >= typeBitWidth(from) ? "signed widening"
469 : "signed narrowing";
470 }
471 if (isUnsignedIntegerType(from) && isUnsignedIntegerType(to)) {
472 return typeBitWidth(to) >= typeBitWidth(from) ? "unsigned widening"
473 : "unsigned narrowing";
474 }
475 if (from->isInteger() && to->isFloatingPoint()) {
476 return "integer-to-float conversion";
477 }
478 if (from->isFloatingPoint() && to->isInteger()) {
479 return "float-to-integer conversion";
480 }
481 if (from->isInteger() && to->isInteger()) {
482 return "signedness-changing integer conversion";
483 }
484 return cost >= 1000 ? "not convertible" : "implicit conversion";

632void Binder::applyImports(ModuleState &module, bool allowIncomplete) {
633 std::map<std::string, std::string> namespaceOwners;
634
635 for (const auto &import : module.info->imports) {
636 if (!import.moduleAlias.empty() && import.targetModuleIds.size() != 1) {
637 if (!allowIncomplete) {
638 error(import.span, "Module alias imports are only allowed when the "
639 "path resolves to a single module.");
640 }
641 continue;
642 }
643
644 for (const auto &targetId : import.targetModuleIds) {
645 auto targetIt = modules_.find(targetId);
646 if (targetIt == modules_.end()) {
647 continue;
648 }
649
650 auto &target = targetIt->second;
651 auto alias = import.moduleAlias.empty() ? target.info->moduleName
652 : import.moduleAlias;
653
654 auto existingOwner = namespaceOwners.find(alias);
655 if (existingOwner != namespaceOwners.end() &&
656 existingOwner->second != targetId) {
657 if (!allowIncomplete) {
658 error(import.span, "Import namespace '" + alias +
659 "' is ambiguous because multiple files share "
660 "that module name.");
661 }
662 continue;
663 }
664
665 if (!module.scope->lookupLocal(alias)) {
666 module.scope->declare(alias, target.symbol);
667 namespaceOwners[alias] = targetId;
668 } else if (module.scope->lookupLocal(alias) != target.symbol) {
669 if (!allowIncomplete) {
670 error(import.span, "Cannot import module '" + alias +
671 "' because that name is already declared in "
672 "the current file.");
673 }
674 continue;
675 }
676
677 if (import.bindings.empty()) {
678 if (import.visibility == Visibility::Public) {
679 module.symbol->exports[alias] = target.symbol;
680 }
681 continue;
682 }
683
684 if (import.targetModuleIds.size() != 1) {
685 if (!allowIncomplete) {
686 error(import.span, "Selective imports are only allowed when the path "
687 "resolves to a single module.");
688 }
689 continue;
690 }
691
692 for (const auto &binding : import.bindings) {
693 auto exportedIt = target.symbol->exports.find(binding.sourceName);
694 if (exportedIt == target.symbol->exports.end()) {
695 if (allowIncomplete) {
696 continue;
697 }
698 auto memberIt = target.symbol->members.find(binding.sourceName);
699 if (memberIt != target.symbol->members.end() &&
700 memberIt->second->visibility != Visibility::Public) {
701 error(import.span, "Member '" + binding.sourceName +
702 "' of module '" + alias + "' is private.");
703 } else {
704 error(import.span, "Module '" + alias + "' has no public member '" +
705 binding.sourceName + "'.");
706 }
707 continue;
708 }
709
710 auto existing = module.scope->lookupLocal(binding.localName);
711 if (existing && existing != exportedIt->second) {
712 if (!allowIncomplete) {
713 error(import.span, "Imported name '" + binding.localName +
714 "' conflicts with an existing declaration "
715 "in the current file.");
716 }
717 continue;
718 }
719 if (!existing) {
720 module.scope->declare(binding.localName, exportedIt->second);
721 }
722 if (import.visibility == Visibility::Public) {
723 module.symbol->exports[binding.localName] = exportedIt->second;
724 }
725 }
726 }
727 }

735 if (auto funDecl = dynamic_cast<FunDecl *>(child.get())) {
736 if (funDecl->isUnsafe_) {
737 requireUnsafeEnabled(funDecl->span, "'unsafe fun'");
738 ++unsafeTypeContextDepth_;
739 }
740
741 std::vector<std::shared_ptr<VariableSymbol>> params;
742 for (size_t i = 0; i < funDecl->params_.size(); ++i) {
743 const auto &p = funDecl->params_[i];
744 if (p->isVariadic && i + 1 != funDecl->params_.size()) {
745 error(p->span, "Variadic parameter must be the last parameter.");
746 }
747 if (p->isVariadic && p->isRef) {
748 error(p->span, "Variadic parameter cannot be passed by 'ref'.");
749 }
750 auto mappedType = mapType(*p->type);
751 if (!mappedType) {
752 error(p->span, "Unknown type: " + p->type->qualifiedName());
753 mappedType =
754 std::make_shared<zir::PrimitiveType>(zir::TypeKind::Void);
755 }
756 auto symbol = std::make_shared<VariableSymbol>(
757 p->name, mappedType, false, p->isRef, p->name,
758 module.info->moduleName, Visibility::Private);
759 if (p->isVariadic) {
760 symbol->is_variadic_pack = true;
761 symbol->variadic_element_type = mappedType;
762 symbol->type = makeVariadicViewType(mappedType);
763 }
764 params.push_back(std::move(symbol));
765 }
766
767 std::shared_ptr<zir::Type> retType = nullptr;
768 if (funDecl->returnType_) {
769 retType = mapType(*funDecl->returnType_);
770 } else if (funDecl->name_ == "main" && module.info->isEntry) {
771 retType = std::make_shared<zir::PrimitiveType>(zir::TypeKind::Int);
772 } else {
773 retType = std::make_shared<zir::PrimitiveType>(zir::TypeKind::Void);
774 }
775
776 if (!retType) {
777 error(funDecl->span,
778 "Unknown return type in function '" + funDecl->name_ + "'.");
779 retType = std::make_shared<zir::PrimitiveType>(zir::TypeKind::Void);
780 }
781
782 if (funDecl->isUnsafe_) {
783 --unsafeTypeContextDepth_;
784 }
785
786 auto linkName = (funDecl->name_ == "main" && module.info->isEntry)
787 ? std::string("main")
788 : std::string();
789 auto symbol = std::make_shared<FunctionSymbol>(
790 funDecl->name_, std::move(params), std::move(retType), "",
791 module.info->moduleName, funDecl->visibility_, funDecl->isUnsafe_);
792 if (linkName != "main") {
793 symbol->linkName = mangleFunctionName(module.info->linkPath.empty()
794 ? module.info->moduleId
795 : module.info->linkPath,
796 *symbol);
797 } else {
798 symbol->linkName = linkName;
799 }
800
801 auto existing = module.scope->lookupLocal(funDecl->name_);
802 if (findFunctionBySignature(existing, *symbol)) {
803 error(funDecl->span,
804 "Function '" + funDecl->name_ + "' already declared.");
805 continue;
806 }
807 auto overloads = module.scope->declareFunction(funDecl->name_, symbol);
808 if (!overloads) {
809 error(funDecl->span, "Function '" + funDecl->name_ +
810 "' conflicts with an existing declaration.");
811 continue;
812 }
813 declaredFunctionSymbols_[funDecl] = symbol;
814 module.symbol->members[funDecl->name_] = overloads;
815 if (funDecl->visibility_ == Visibility::Public) {
816 auto exportIt = module.symbol->exports.find(funDecl->name_);
817 std::shared_ptr<OverloadSetSymbol> exportSet;
818 if (exportIt == module.symbol->exports.end()) {
819 exportSet = std::make_shared<OverloadSetSymbol>(
820 funDecl->name_, module.info->moduleName, Visibility::Public);
821 module.symbol->exports[funDecl->name_] = exportSet;
822 } else {
823 exportSet =
824 std::dynamic_pointer_cast<OverloadSetSymbol>(exportIt->second);
825 }
826 if (!exportSet) {
827 error(funDecl->span, "Function '" + funDecl->name_ +
828 "' conflicts with an exported declaration.");
829 } else {
830 exportSet->addOverload(symbol);
831 }
832 }
833 } else if (auto classDecl = dynamic_cast<ClassDecl *>(child.get())) {
834 auto classSymbol = std::dynamic_pointer_cast<TypeSymbol>(
835 module.scope->lookup(classDecl->name_));
836 if (!classSymbol || !classSymbol->isClass) {
837 continue;
838 }
839 auto classType =
840 std::static_pointer_cast<zir::ClassType>(classSymbol->type);
841 auto &classInfo = classInfos_[classType->getName()];
842
843 if (classDecl->baseType_) {
844 auto baseType = mapType(*classDecl->baseType_);
845 if (baseType && baseType->getKind() == zir::TypeKind::Class) {
846 auto baseClass = std::static_pointer_cast<zir::ClassType>(baseType);
847 auto baseIt = classInfos_.find(baseClass->getName());
848 if (baseIt != classInfos_.end()) {
849 classInfo.methods.insert(baseIt->second.methods.begin(),
850 baseIt->second.methods.end());
851 classInfo.nextVirtualSlot = baseIt->second.nextVirtualSlot;
852 }
853 }
854 }
855
856 for (const auto &methodDecl : classDecl->methods_) {
857 std::vector<std::shared_ptr<VariableSymbol>> params;
858 if (!methodDecl->isStatic_) {
859 params.push_back(std::make_shared<VariableSymbol>(
860 "self", classType, false, false, "self", module.info->moduleName,
861 Visibility::Private));
862 }
863
864 for (size_t i = 0; i < methodDecl->params_.size(); ++i) {
865 const auto &p = methodDecl->params_[i];
866 auto mappedType = mapType(*p->type);
867 if (!mappedType) {
868 error(p->span, "Unknown type: " + p->type->qualifiedName());
869 mappedType =
870 std::make_shared<zir::PrimitiveType>(zir::TypeKind::Void);
871 }
872 params.push_back(std::make_shared<VariableSymbol>(
873 p->name, mappedType, false, p->isRef, p->name,
874 module.info->moduleName, Visibility::Private));
875 }
876
877 std::shared_ptr<zir::Type> retType;
878 bool isCtor = methodDecl->name_ == "init";
879 bool isDtor = methodDecl->name_ == "deinit";
880 if (isDtor &&
881 (!methodDecl->params_.empty() || methodDecl->returnType_)) {
882 error(methodDecl->span,
883 "Destructor 'deinit' cannot have parameters or a return type.");
884 }
885 if (isCtor) {
886 retType = std::make_shared<zir::PrimitiveType>(zir::TypeKind::Void);
887 } else if (isDtor) {
888 retType = std::make_shared<zir::PrimitiveType>(zir::TypeKind::Void);
889 } else if (methodDecl->returnType_) {
890 retType = mapType(*methodDecl->returnType_);
891 } else {
892 retType = std::make_shared<zir::PrimitiveType>(zir::TypeKind::Void);
893 }
894 if (!retType) {
895 retType = std::make_shared<zir::PrimitiveType>(zir::TypeKind::Void);
896 }
897
898 auto symbol = std::make_shared<FunctionSymbol>(
899 methodDecl->name_, std::move(params), std::move(retType), "",
900 module.info->moduleName, methodDecl->visibility_,
901 methodDecl->isUnsafe_);
902 symbol->isMethod = !methodDecl->isStatic_;
903 symbol->isStatic = methodDecl->isStatic_;
904 symbol->isConstructor = isCtor;
905 symbol->isDestructor = isDtor;
906 symbol->ownerTypeName = classType->getName();
907 if (symbol->isMethod && !symbol->isStatic && !symbol->isConstructor &&
908 !symbol->isDestructor) {
909 auto existingIt = classInfo.methods.find(methodDecl->name_);
910 if (existingIt != classInfo.methods.end()) {
911 auto existingMethod =
912 std::dynamic_pointer_cast<FunctionSymbol>(existingIt->second);
913 if (existingMethod && existingMethod->vtableSlot >= 0) {
914 symbol->vtableSlot = existingMethod->vtableSlot;
915 }
916 }
917 if (symbol->vtableSlot < 0) {
918 symbol->vtableSlot = classInfo.nextVirtualSlot++;
919 }
920 }
921 symbol->linkName =
922 mangleName(module.info->linkPath.empty() ? module.info->moduleId
923 : module.info->linkPath,
924 classDecl->name_ + "$" + methodDecl->name_ + "$" +
925 sanitizeTypeName(functionSignatureKey(*symbol)));
926 declaredFunctionSymbols_[methodDecl.get()] = symbol;
927 classInfo.methods[methodDecl->name_] = symbol;
928 if (isCtor) {
929 classInfo.constructor = symbol;
930 } else if (isDtor) {
931 classInfo.destructor = symbol;
932 }
933 }
934 } else if (auto extDecl = dynamic_cast<ExtDecl *>(child.get())) {
935 ++externTypeContextDepth_;
936 std::vector<std::shared_ptr<VariableSymbol>> params;
937 for (const auto &p : extDecl->params_) {
938 if (p->isVariadic) {
939 error(p->span, "Variadic parameters are only supported in Zap "
940 "function declarations.");
941 }
942 auto mappedType = mapType(*p->type);
943 if (!mappedType) {
944 error(p->span, "Unknown type: " + p->type->qualifiedName());
945 mappedType =
946 std::make_shared<zir::PrimitiveType>(zir::TypeKind::Void);
947 }
948 params.push_back(std::make_shared<VariableSymbol>(
949 p->name, mappedType, false, p->isRef, p->name,
950 module.info->moduleName, Visibility::Private));
951 }
952
953 auto retType =
954 extDecl->returnType_
955 ? mapType(*extDecl->returnType_)
956 : std::make_shared<zir::PrimitiveType>(zir::TypeKind::Void);
957 if (!retType) {
958 error(extDecl->span, "Unknown return type in external function '" +
959 extDecl->name_ + "'.");
960 retType = std::make_shared<zir::PrimitiveType>(zir::TypeKind::Void);
961 }
962 --externTypeContextDepth_;
963
964 auto linkName = extDecl->name_;
965 bool isStdFsModule =
966 module.info->moduleName == "fs" &&
967 module.info->sourceName.find("/std/fs.zp") != std::string::npos;
968 bool isStdIoModule =
969 module.info->moduleName == "io" &&
970 module.info->sourceName.find("/std/io.zp") != std::string::npos;
971 bool isStdPathModule =
972 module.info->moduleName == "path" &&
973 module.info->sourceName.find("/std/path.zp") != std::string::npos;
974 if (isStdFsModule && extDecl->name_ == "mkdir") {
975 linkName = "zap_fs_mkdir";
976 } else if (isStdIoModule && extDecl->name_ == "printf") {
977 linkName = "zap_printf";
978 } else if (isStdIoModule && extDecl->name_ == "printfln") {
979 linkName = "zap_printfln";
980 } else if (isStdPathModule && extDecl->name_ == "basename") {
981 linkName = "zap_path_basename";
982 }
983
984 auto symbol = std::make_shared<FunctionSymbol>(
985 extDecl->name_, std::move(params), std::move(retType), linkName,
986 module.info->moduleName, extDecl->visibility_, false,
987 extDecl->isCVariadic_);
988 auto existing = module.scope->lookupLocal(extDecl->name_);
989 if (findFunctionBySignature(existing, *symbol)) {
990 error(extDecl->span,
991 "External function '" + extDecl->name_ + "' already declared.");
992 continue;
993 }
994 auto overloads = module.scope->declareFunction(extDecl->name_, symbol);
995 if (!overloads) {
996 error(extDecl->span, "External function '" + extDecl->name_ +
997 "' conflicts with an existing declaration.");
998 continue;
999 }
1000 declaredFunctionSymbols_[extDecl] = symbol;
1001 module.symbol->members[extDecl->name_] = overloads;
1002 if (extDecl->visibility_ == Visibility::Public) {
1003 auto exportIt = module.symbol->exports.find(extDecl->name_);
1004 std::shared_ptr<OverloadSetSymbol> exportSet;
1005 if (exportIt == module.symbol->exports.end()) {
1006 exportSet = std::make_shared<OverloadSetSymbol>(
1007 extDecl->name_, module.info->moduleName, Visibility::Public);
1008 module.symbol->exports[extDecl->name_] = exportSet;
1009 } else {
1010 exportSet =
1011 std::dynamic_pointer_cast<OverloadSetSymbol>(exportIt->second);
1012 }
1013 if (!exportSet) {
1014 error(extDecl->span, "External function '" + extDecl->name_ +
1015 "' conflicts with an exported declaration.");
1016 } else {
1017 exportSet->addOverload(symbol);
1018 }
1019 }
1020 } else if (auto varDecl = dynamic_cast<VarDecl *>(child.get())) {
1021 if (!varDecl->isGlobal_) {
1022 continue;
1023 }
1024 auto type = mapType(*varDecl->type_);
1025 if (!type) {
1026 error(varDecl->span,
1027 "Unknown type: " + varDecl->type_->qualifiedName());
1028 type = std::make_shared<zir::PrimitiveType>(zir::TypeKind::Void);
1029 }
1030 auto symbol = std::make_shared<VariableSymbol>(
1031 varDecl->name_, type, false, false,
1032 mangleName(module.info->linkPath.empty() ? module.info->moduleId
1033 : module.info->linkPath,
1034 varDecl->name_),
1035 module.info->moduleName, varDecl->visibility_);
1036 if (!module.scope->declare(varDecl->name_, symbol)) {
1037 error(varDecl->span,
1038 "Variable '" + varDecl->name_ + "' already declared.");
1039 }
1040 module.symbol->members[varDecl->name_] = symbol;
1041 if (varDecl->visibility_ == Visibility::Public) {
1042 module.symbol->exports[varDecl->name_] = symbol;
1043 }
1044 } else if (auto constDecl = dynamic_cast<ConstDecl *>(child.get())) {
1045 auto type = mapType(*constDecl->type_);
1046 if (!type) {
1047 error(constDecl->span,
1048 "Unknown type: " + constDecl->type_->qualifiedName());
1049 type = std::make_shared<zir::PrimitiveType>(zir::TypeKind::Void);
1050 }
1051 auto symbol = std::make_shared<VariableSymbol>(
1052 constDecl->name_, type, true, false,
1053 mangleName(module.info->linkPath.empty() ? module.info->moduleId
1054 : module.info->linkPath,
1055 constDecl->name_),
1056 module.info->moduleName, constDecl->visibility_);
1057 if (!module.scope->declare(constDecl->name_, symbol)) {
1058 error(constDecl->span,
1059 "Identifier '" + constDecl->name_ + "' already declared.");
1060 }
1061 module.symbol->members[constDecl->name_] = symbol;
1062 if (constDecl->visibility_ == Visibility::Public) {
1063 module.symbol->exports[constDecl->name_] = symbol;
1064 }
1065 }
1066 }

1075Binder::resolveModuleMember(const std::string &moduleName,
1076 const std::string &memberName, SourceSpan span) {
1077 auto moduleSym = std::dynamic_pointer_cast<ModuleSymbol>(
1078 currentScope_->lookup(moduleName));
1079 if (!moduleSym) {
1080 error(span, "Undefined module: " + moduleName);

1087 }
1088

1092 "Module '" + moduleName + "' has no member '" + memberName + "'.");

1094 }
1095 error(span, "Member '" + memberName + "' of module '" + moduleName +
1096 "' is private.");
1097 return nullptr;

1100std::shared_ptr<Symbol>
1101Binder::resolveQualifiedSymbol(const std::vector<std::string> &parts,
1102 SourceSpan span, SymbolKind expectedKind,
1103 bool allowAnyKind) {
1104 if (parts.empty()) {
1105 return nullptr;
1106 }
1107
1108 if (parts.size() == 1) {
1109 auto symbol = lookupVisibleSymbol(parts.front());
1110 if (!symbol) {
1111 error(span, "Undefined identifier: " + parts.front());
1112 return nullptr;
1113 }
1114 if (!allowAnyKind && symbol->getKind() != expectedKind &&
1115 !(expectedKind == SymbolKind::Function &&
1116 symbol->getKind() == SymbolKind::OverloadSet)) {
1117 return nullptr;
1118 }
1119 return symbol;
1120 }
1121
1122 if (parts.size() != 2) {
1123 error(span, "Only single-level module qualification is supported.");
1124 return nullptr;
1125 }
1126
1127 auto symbol = resolveModuleMember(parts[0], parts[1], span);
1128 if (!symbol) {
1129 return nullptr;
1130 }
1131 if (!allowAnyKind && symbol->getKind() != expectedKind &&
1132 !(expectedKind == SymbolKind::Function &&
1133 symbol->getKind() == SymbolKind::OverloadSet)) {
1134 return nullptr;
1135 }
1136 return symbol;
1137}

1156

1160
1161 if (node.baseType_) {
1162 auto baseType = mapType(*node.baseType_);
1163 if (!baseType || baseType->getKind() != zir::TypeKind::Class) {
1164 error(node.baseType_->span, "Base type must be a class.");
1165 } else {

1178 }

1185 }

1190 modules_[currentModuleId_].info->moduleName, field->visibility_);

1199 }
1200
1201 currentClassStack_.pop_back();

1204void Binder::visit(FunDecl &node) {
1205 auto symbolIt = declaredFunctionSymbols_.find(&node);
1206 auto symbol =
1207 symbolIt == declaredFunctionSymbols_.end() ? nullptr : symbolIt->second;
1208 if (!symbol) {
1209 return;
1210 }
1211
1212 pushScope();
1213 auto oldFunction = currentFunction_;
1214 currentFunction_ = symbol;
1215 int oldUnsafeDepth = unsafeDepth_;
1216 if (node.isUnsafe_) {
1217 requireUnsafeEnabled(node.span, "'unsafe fun'");
1218 ++unsafeDepth_;
1219 }
1220
1221 for (const auto &param : symbol->parameters) {
1222 if (!currentScope_->declare(param->name, param)) {
1223 error(node.span, "Parameter '" + param->name + "' already declared.");
1224 }
1225 }
1226
1227 auto boundBody = bindBody(node.body_.get(), false);
1228
1229 popScope();
1230 currentFunction_ = oldFunction;
1231 unsafeDepth_ = oldUnsafeDepth;
1232
1233 bool hasReturn = false;
1234 if (boundBody) {
1235 if (boundBody->result)
1236 hasReturn = true;
1237 for (const auto &stmt : boundBody->statements) {
1238 if (dynamic_cast<BoundReturnStatement *>(stmt.get())) {
1239 hasReturn = true;
1240 break;
1241 }
1242 }
1243 }
1244
1245 if (!hasReturn && symbol->linkName == "main" &&
1246 symbol->returnType->isInteger()) {
1247 auto intType = std::make_shared<zir::PrimitiveType>(zir::TypeKind::Int);
1248 auto lit = std::make_unique<BoundLiteral>("0", intType);
1249 boundBody->statements.push_back(
1250 std::make_unique<BoundReturnStatement>(std::move(lit)));
1251 hasReturn = true;
1252 }
1253
1254 if (!hasReturn && symbol->returnType->getKind() != zir::TypeKind::Void) {
1255 auto kind = symbol->returnType->getKind();
1256 if (symbol->returnType->isInteger() || kind == zir::TypeKind::Float ||
1257 kind == zir::TypeKind::Bool) {
1258 std::string litVal = "0";
1259 if (kind == zir::TypeKind::Float)
1260 litVal = "0.0";
1261 else if (kind == zir::TypeKind::Bool)
1262 litVal = "false";
1263 auto lit = std::make_unique<BoundLiteral>(litVal, symbol->returnType);
1264 boundBody->statements.push_back(
1265 std::make_unique<BoundReturnStatement>(std::move(lit)));
1266 }
1267
1268 _diag.report(node.span, zap::DiagnosticLevel::Warning,
1269 "Function '" + node.name_ +
1270 "' has non-void return type but no return on some paths.");
1271 }
1272
1273 boundRoot_->functions.push_back(
1274 std::make_unique<BoundFunctionDeclaration>(symbol, std::move(boundBody)));
1275}

1296 auto savedBlock = std::move(currentBlock_);

1298 pushScope();

1300
1301 if (body) {
1302 body->accept(*this);
1303 }
1304
1305 auto boundBody = std::make_unique<BoundBlock>();
1306 if (currentBlock_) {
1307 boundBody = std::move(currentBlock_);
1308 }
1309
1310 if (createScope) {
1311 popScope();
1312 }
1313

1318void Binder::visit(BodyNode &node) {
1319 currentBlock_ = std::make_unique<BoundBlock>();
1320
1321 for (const auto &stmt : node.statements) {
1322 stmt->accept(*this);
1323 if (!statementStack_.empty()) {
1324 currentBlock_->statements.push_back(std::move(statementStack_.top()));
1325 statementStack_.pop();
1326 } else if (!expressionStack_.empty()) {
1327 auto expr = std::move(expressionStack_.top());
1328 expressionStack_.pop();
1329 currentBlock_->statements.push_back(
1330 std::make_unique<BoundExpressionStatement>(std::move(expr)));
1331 }
1332 }
1333
1334 if (node.result) {
1335 node.result->accept(*this);
1336 if (!expressionStack_.empty()) {
1337 currentBlock_->result = std::move(expressionStack_.top());
1338 expressionStack_.pop();
1339 }
1340 }

1363void Binder::visit(VarDecl &node) {
1364 auto existing = currentScope_->lookupLocal(node.name_);
1365 std::shared_ptr<VariableSymbol> symbol;
1366 if (existing) {
1367 symbol = std::dynamic_pointer_cast<VariableSymbol>(existing);
1368 }
1369
1370 auto type = mapType(*node.type_);
1371 if (!type) {
1372 error(node.span, "Unknown type: " + node.type_->qualifiedName());
1373 type = std::make_shared<zir::PrimitiveType>(zir::TypeKind::Void);
1374 }
1375
1376 std::unique_ptr<BoundExpression> initializer = nullptr;
1377 if (node.initializer_) {
1378 initializer = bindExpressionWithExpected(node.initializer_.get(), type);
1379 if (initializer) {
1380 if (!canConvert(initializer->type, type)) {
1381 error(node.initializer_->span, "Cannot assign expression of type '" +
1382 initializer->type->toString() +
1383 "' to variable of type '" +
1384 type->toString() + "'");
1385 } else {
1386 initializer = wrapInCast(std::move(initializer), type);
1387 }
1388 }
1389 }
1390
1391 if (!symbol) {
1392 symbol = std::make_shared<VariableSymbol>(
1393 node.name_, type, false, false,
1394 node.isGlobal_ ? mangleName(currentModuleLinkPath(), node.name_)
1395 : node.name_,
1396 modules_[currentModuleId_].info->moduleName, node.visibility_);
1397 if (!currentScope_->declare(node.name_, symbol)) {
1398 error(node.span, "Variable '" + node.name_ + "' already declared.");
1399 }
1400 }
1401
1402 auto boundDecl = std::make_unique<BoundVariableDeclaration>(
1403 symbol, std::move(initializer));
1404
1405 if (currentBlock_ && !node.isGlobal_) {
1406 statementStack_.push(std::move(boundDecl));
1407 } else {
1408 boundRoot_->globals.push_back(std::move(boundDecl));
1409 }

1493void Binder::visit(BinExpr &node) {
1494 std::unique_ptr<BoundExpression> left;
1495 std::unique_ptr<BoundExpression> right;
1496
1497 bool leftIsNullLiteral =
1498 dynamic_cast<ConstNull *>(node.left_.get()) != nullptr;
1499 bool rightIsNullLiteral =
1500 dynamic_cast<ConstNull *>(node.right_.get()) != nullptr;
1501
1502 if (leftIsNullLiteral && !rightIsNullLiteral) {
1503 right = bindExpressionWithExpected(node.right_.get(), nullptr);
1504 if (!right)
1505 return;
1506 left = bindExpressionWithExpected(node.left_.get(), right->type);
1507 if (!left)
1508 return;
1509 } else if (rightIsNullLiteral && !leftIsNullLiteral) {
1510 left = bindExpressionWithExpected(node.left_.get(), nullptr);
1511 if (!left)
1512 return;
1513 right = bindExpressionWithExpected(node.right_.get(), left->type);
1514 if (!right)
1515 return;
1516 } else {
1517 node.left_->accept(*this);
1518 if (expressionStack_.empty())
1519 return;
1520 left = std::move(expressionStack_.top());
1521 expressionStack_.pop();
1522
1523 node.right_->accept(*this);
1524 if (expressionStack_.empty())
1525 return;
1526 right = std::move(expressionStack_.top());
1527 expressionStack_.pop();
1528 }
1529
1530 auto leftType = left->type;
1531 auto rightType = right->type;
1532 std::shared_ptr<zir::Type> resultType = leftType;
1533
1534 if (node.op_ == "~") {
1535 bool leftOk =
1536 isStringType(leftType) || leftType->getKind() == zir::TypeKind::Char;
1537 bool rightOk =
1538 isStringType(rightType) || rightType->getKind() == zir::TypeKind::Char;
1539 bool hasString = isStringType(leftType) || isStringType(rightType);
1540
1541 if (!leftOk || !rightOk || !hasString) {
1542 error(SourceSpan::merge(node.left_->span, node.right_->span),
1543 "Concatenation requires String and/or Char operands with at least "
1544 "one String, got '" +
1545 leftType->toString() + "' and '" + rightType->toString() + "'");
1546 }
1547 resultType = std::make_shared<zir::RecordType>("String", "String");
1548 } else if ((node.op_ == "+" || node.op_ == "-") &&
1549 (isPointerType(leftType) || isPointerType(rightType))) {
1550 requireUnsafeEnabled(node.span, "pointer arithmetic");
1551 requireUnsafeContext(node.span, "pointer arithmetic");
1552
1553 if (node.op_ == "+" && isPointerType(leftType) && rightType->isInteger()) {
1554 resultType = leftType;
1555 } else if (node.op_ == "+" && leftType->isInteger() &&
1556 isPointerType(rightType)) {
1557 std::swap(left, right);
1558 std::swap(leftType, rightType);
1559 resultType = leftType;
1560 } else if (node.op_ == "-" && isPointerType(leftType) &&
1561 rightType->isInteger()) {
1562 resultType = leftType;
1563 } else if (node.op_ == "-" && isPointerType(leftType) &&
1564 isPointerType(rightType)) {
1565 if (leftType->toString() != rightType->toString()) {
1566 error(SourceSpan::merge(node.left_->span, node.right_->span),
1567 "Pointer subtraction requires operands of the same type.");
1568 }
1569 resultType = std::make_shared<zir::PrimitiveType>(zir::TypeKind::Int);
1570 } else {
1571 error(SourceSpan::merge(node.left_->span, node.right_->span),
1572 "Invalid pointer arithmetic between '" + leftType->toString() +
1573 "' and '" + rightType->toString() + "'");
1574 }
1575 } else if (node.op_ == "+" || node.op_ == "-" || node.op_ == "*" ||
1576 node.op_ == "/" || node.op_ == "%" || node.op_ == "^") {
1577 if (!isNumeric(leftType) || !isNumeric(rightType)) {
1578 error(SourceSpan::merge(node.left_->span, node.right_->span),
1579 "Operator '" + node.op_ + "' cannot be applied to '" +
1580 leftType->toString() + "' and '" + rightType->toString() + "'");
1581 } else {
1582 resultType = getPromotedType(leftType, rightType);
1583 left = wrapInCast(std::move(left), resultType);
1584 right = wrapInCast(std::move(right), resultType);
1585 }
1586 } else if (node.op_ == "==" || node.op_ == "!=" || node.op_ == "<" ||
1587 node.op_ == "<=" || node.op_ == ">" || node.op_ == ">=") {
1588 bool classOrNullComparison =
1589 (leftType->getKind() == zir::TypeKind::Class &&
1590 isNullType(rightType)) ||
1591 (rightType->getKind() == zir::TypeKind::Class && isNullType(leftType));
1592 if (!classOrNullComparison &&
1593 (isPointerType(leftType) || isPointerType(rightType) ||
1594 isNullType(leftType) || isNullType(rightType))) {
1595 requireUnsafeEnabled(node.span, "pointer comparisons");
1596 requireUnsafeContext(node.span, "pointer comparisons");
1597 }
1598
1599 // Reject comparisons of struct types
1600 if (leftType->getKind() == zir::TypeKind::Record ||
1601 rightType->getKind() == zir::TypeKind::Record) {
1602 error(SourceSpan::merge(node.left_->span, node.right_->span),
1603 "Cannot compare struct types '" + leftType->toString() + "' and '" +
1604 rightType->toString() + "'");
1605 }
1606
1607 if (!canConvert(leftType, rightType) && !canConvert(rightType, leftType)) {
1608 error(SourceSpan::merge(node.left_->span, node.right_->span),
1609 "Cannot compare '" + leftType->toString() + "' and '" +
1610 rightType->toString() + "'");
1611 } else if (isNullType(leftType) && isPointerType(rightType)) {
1612 left = wrapInCast(std::move(left), rightType);
1613 } else if (isNullType(rightType) && isPointerType(leftType)) {
1614 right = wrapInCast(std::move(right), leftType);
1615 } else if (isNullType(leftType) &&
1616 rightType->getKind() == zir::TypeKind::Class) {
1617 left = wrapInCast(std::move(left), rightType);
1618 } else if (isNullType(rightType) &&
1619 leftType->getKind() == zir::TypeKind::Class) {
1620 right = wrapInCast(std::move(right), leftType);
1621 }
1622 resultType = std::make_shared<zir::PrimitiveType>(zir::TypeKind::Bool);
1623 } else if (node.op_ == "&&" || node.op_ == "||") {
1624 if (leftType->getKind() != zir::TypeKind::Bool ||
1625 rightType->getKind() != zir::TypeKind::Bool) {
1626 error(SourceSpan::merge(node.left_->span, node.right_->span),
1627 "Logical operator '" + node.op_ + "' requires Bool operands.");
1628 }
1629 resultType = std::make_shared<zir::PrimitiveType>(zir::TypeKind::Bool);
1630 }
1631
1632 expressionStack_.push(std::make_unique<BoundBinaryExpression>(
1633 std::move(left), node.op_, std::move(right), resultType));
1634}

1636void Binder::visit(TernaryExpr &node) {
1637 node.condition_->accept(*this);
1638 if (expressionStack_.empty())
1639 return;
1640 auto condition = std::move(expressionStack_.top());
1641 expressionStack_.pop();
1642
1643 if (condition->type->getKind() != zir::TypeKind::Bool) {
1644 error(node.condition_->span, "Ternary condition must be Bool, got '" +
1645 condition->type->toString() + "'");
1646 }
1647
1648 node.thenExpr_->accept(*this);
1649 if (expressionStack_.empty())
1650 return;
1651 auto thenExpr = std::move(expressionStack_.top());
1652 expressionStack_.pop();
1653
1654 node.elseExpr_->accept(*this);
1655 if (expressionStack_.empty())
1656 return;
1657 auto elseExpr = std::move(expressionStack_.top());
1658 expressionStack_.pop();
1659
1660 if (!canConvert(thenExpr->type, elseExpr->type) &&
1661 !canConvert(elseExpr->type, thenExpr->type)) {
1662 error(SourceSpan::merge(node.thenExpr_->span, node.elseExpr_->span),
1663 "Ternary branches must be compatible, got '" +
1664 thenExpr->type->toString() + "' and '" +
1665 elseExpr->type->toString() + "'");
1666 }
1667
1668 auto resultType = canConvert(thenExpr->type, elseExpr->type) ? elseExpr->type
1669 : thenExpr->type;
1670 thenExpr = wrapInCast(std::move(thenExpr), resultType);
1671 elseExpr = wrapInCast(std::move(elseExpr), resultType);
1672
1673 expressionStack_.push(std::make_unique<BoundTernaryExpression>(
1674 std::move(condition), std::move(thenExpr), std::move(elseExpr),
1675 resultType));
1676}

1715 if (expressionStack_.empty())
1716 return;
1717

1719 expressionStack_.pop();
1720
1721 auto targetType = mapType(*node.type_);
1722 if (!targetType) {
1723 error(node.type_->span, "Unknown type: " + node.type_->qualifiedName());
1724 return;
1725 }
1726
1727 requireUnsafeEnabled(node.span, "explicit casts");
1728 requireUnsafeContext(node.span, "explicit casts");
1729
1730 bool castAllowed = false;
1731 if (isNumeric(expr->type) && isNumeric(targetType))
1732 castAllowed = true;
1733 else if ((isPointerType(expr->type) || isNullType(expr->type)) &&
1734 isPointerType(targetType))
1735 castAllowed = true;
1736 else if (isStringType(expr->type) && isPointerType(targetType))
1737 castAllowed = true;
1738 else if (isPointerType(expr->type) && targetType->isInteger())
1739 castAllowed = true;
1740 else if (expr->type->isInteger() && isPointerType(targetType))
1741 castAllowed = true;
1742

1745 targetType->toString() + "'");

1772void Binder::visit(AssignNode &node) {
1773 node.target_->accept(*this);
1774 if (expressionStack_.empty())
1775 return;
1776 auto target = std::move(expressionStack_.top());
1777 expressionStack_.pop();
1778
1779 bool isLValue =
1780 dynamic_cast<BoundVariableExpression *>(target.get()) ||
1781 dynamic_cast<BoundIndexAccess *>(target.get()) ||
1782 dynamic_cast<BoundMemberAccess *>(target.get()) ||
1783 (dynamic_cast<BoundUnaryExpression *>(target.get()) &&
1784 static_cast<BoundUnaryExpression *>(target.get())->op == "*");
1785
1786 if (!isLValue) {
1787 error(node.span, "Target of assignment must be an l-value.");
1788 return;
1789 }
1790
1791 if (auto varExpr = dynamic_cast<BoundVariableExpression *>(target.get())) {
1792 if (varExpr->symbol->is_const) {
1793 error(node.span,
1794 "Cannot assign to constant '" + varExpr->symbol->name + "'.");
1795 return;
1796 }
1797 } else if (auto indexExpr = dynamic_cast<BoundIndexAccess *>(target.get())) {
1798 if (isStringType(indexExpr->left->type)) {
1799 error(node.span, "Cannot assign through String index access.");
1800 return;
1801 }
1802 }
1803
1804 auto expr = bindExpressionWithExpected(node.expr_.get(), target->type);
1805 if (!expr)
1806 return;
1807
1808 if (!canConvert(expr->type, target->type)) {
1809 error(node.span, "Cannot assign expression of type '" +
1810 expr->type->toString() + "' to type '" +
1811 target->type->toString() + "'");
1812 } else {
1813 expr = wrapInCast(std::move(expr), target->type);
1814 }
1815
1816 statementStack_.push(
1817 std::make_unique<BoundAssignment>(std::move(target), std::move(expr)));
1818}

1823 return;

1825 expressionStack_.pop();
1826

1828 !isVariadicViewType(left->type) && !isStringType(left->type)) {
1829 error(node.span,

1835 if (expressionStack_.empty())
1836 return;
1837 auto index = std::move(expressionStack_.top());
1838 expressionStack_.pop();
1839
1840 if (!index->type->isInteger()) {
1841 error(node.span, "Array index must be an integer, but got '" +
1842 index->type->toString() + "'");
1843 }

1846 if (left->type->getKind() == zir::TypeKind::Array) {
1847 auto arrayType = std::static_pointer_cast<zir::ArrayType>(left->type);
1848 elementType = arrayType->getBaseType();
1849 } else if (isVariadicViewType(left->type)) {
1850 auto recordType = std::static_pointer_cast<zir::RecordType>(left->type);
1851 const auto &fields = recordType->getFields();
1852 if (fields.empty() || fields[0].type->getKind() != zir::TypeKind::Pointer) {
1853 error(node.span, "Internal error: invalid variadic view layout.");
1854 return;
1855 }
1856 elementType = std::static_pointer_cast<zir::PointerType>(fields[0].type)
1857 ->getBaseType();
1858 } else {
1859 elementType = std::make_shared<zir::PrimitiveType>(zir::TypeKind::Char);
1860 }
1861
1862 expressionStack_.push(std::make_unique<BoundIndexAccess>(
1863 std::move(left), std::move(index), elementType));
1864}

1866void Binder::visit(MemberAccessNode &node) {
1867 node.left_->accept(*this);
1868 if (expressionStack_.empty())
1869 return;
1870
1871 auto left = std::move(expressionStack_.top());
1872 expressionStack_.pop();
1873
1874 if (auto moduleRef = dynamic_cast<BoundModuleReference *>(left.get())) {
1875 auto memberIt = moduleRef->symbol->members.find(node.member_);
1876 if (memberIt == moduleRef->symbol->members.end()) {
1877 error(node.span, "Module '" + moduleRef->symbol->name +
1878 "' has no member '" + node.member_ + "'");
1879 return;
1880 }
1881 if (memberIt->second->visibility != Visibility::Public) {
1882 error(node.span, "Member '" + node.member_ + "' of module '" +
1883 moduleRef->symbol->name + "' is private.");
1884 return;
1885 }
1886
1887 if (auto varSymbol =
1888 std::dynamic_pointer_cast<VariableSymbol>(memberIt->second)) {
1889 expressionStack_.push(
1890 std::make_unique<BoundVariableExpression>(varSymbol));
1891 return;
1892 }
1893 if (auto typeSymbol =
1894 std::dynamic_pointer_cast<TypeSymbol>(memberIt->second)) {
1895 expressionStack_.push(
1896 std::make_unique<BoundLiteral>("", typeSymbol->type));
1897 return;
1898 }
1899 if (auto nestedModule =
1900 std::dynamic_pointer_cast<ModuleSymbol>(memberIt->second)) {
1901 expressionStack_.push(
1902 std::make_unique<BoundModuleReference>(nestedModule));
1903 return;
1904 }
1905
1906 error(node.span, "'" + node.member_ + "' is not a value or type.");
1907 return;
1908 }
1909
1910 if (left->type->getKind() == zir::TypeKind::Enum) {
1911 auto enumType = std::static_pointer_cast<zir::EnumType>(left->type);
1912 int value = enumType->getVariantIndex(node.member_);
1913 if (value != -1) {
1914 expressionStack_.push(
1915 std::make_unique<BoundLiteral>(std::to_string(value), enumType));
1916 return;
1917 }
1918 } else if (left->type->getKind() == zir::TypeKind::Record) {
1919 auto recordType = std::static_pointer_cast<zir::RecordType>(left->type);
1920 for (const auto &field : recordType->getFields()) {
1921 if (field.name == node.member_) {
1922 expressionStack_.push(std::make_unique<BoundMemberAccess>(
1923 std::move(left), node.member_, field.type));
1924 return;
1925 }
1926 }
1927 } else if (left->type->getKind() == zir::TypeKind::Class) {
1928 auto classType = std::static_pointer_cast<zir::ClassType>(left->type);
1929 if (classType->isWeak()) {
1930 error(node.span, "Weak references cannot be accessed directly.");
1931 return;
1932 }
1933 auto infoIt = classInfos_.find(classType->getName());
1934 if (infoIt != classInfos_.end()) {
1935 auto fieldIt = infoIt->second.fields.find(node.member_);
1936 if (fieldIt != infoIt->second.fields.end()) {
1937 auto fieldVis = fieldIt->second->visibility;
1938 bool allowed =
1939 fieldVis == Visibility::Public ||
1940 (!currentClassStack_.empty() &&
1941 currentClassStack_.back() == classType->getName()) ||
1942 (fieldVis == Visibility::Protected && !currentClassStack_.empty());
1943 if (!allowed) {
1944 error(node.span, "Field '" + node.member_ + "' is not accessible.");
1945 return;
1946 }
1947 expressionStack_.push(std::make_unique<BoundMemberAccess>(
1948 std::move(left), node.member_, fieldIt->second->type));
1949 return;
1950 }
1951 }
1952 }
1953
1954 error(node.span, "Member '" + node.member_ + "' not found in type '" +
1955 left->type->toString() + "'");
1956}

1958void Binder::visit(FunCall &node) {
1959 if (bindWeakBuiltinCall(node)) {
1960 return;
1961 }
1962
1963 if (auto member = dynamic_cast<MemberAccessNode *>(node.callee_.get())) {
1964 member->left_->accept(*this);
1965 if (expressionStack_.empty()) {
1966 return;
1967 }
1968 auto selfExpr = std::move(expressionStack_.top());
1969 expressionStack_.pop();
1970 if (selfExpr->type->getKind() != zir::TypeKind::Class) {
1971 // Not a class method call. Fall through to the normal qualified
1972 // function/module call resolution path below.
1973 } else {
1974 auto classType = std::static_pointer_cast<zir::ClassType>(selfExpr->type);
1975 if (classType->isWeak()) {
1976 error(node.span,
1977 "Weak references cannot be used to call methods directly.");
1978 return;
1979 }
1980 auto infoIt = classInfos_.find(classType->getName());
1981 if (infoIt == classInfos_.end()) {
1982 error(node.span, "Unknown class type: " + classType->getName());
1983 return;
1984 }
1985 auto methodIt = infoIt->second.methods.find(member->member_);
1986 if (methodIt == infoIt->second.methods.end()) {
1987 error(node.span, "Class '" + classType->getName() +
1988 "' has no method '" + member->member_ + "'.");
1989 return;
1990 }
1991 auto funcSymbol =
1992 std::dynamic_pointer_cast<FunctionSymbol>(methodIt->second);
1993 if (!funcSymbol) {
1994 error(node.span, "'" + member->member_ + "' is not a method.");
1995 return;
1996 }
1997 bool methodAllowed =
1998 funcSymbol->visibility == Visibility::Public ||
1999 (!currentClassStack_.empty() &&
2000 currentClassStack_.back() == classType->getName()) ||

2654std::optional<int64_t>
2655Binder::evaluateConstantInt(const BoundExpression *expr) {
2656 if (auto lit = dynamic_cast<const BoundLiteral *>(expr)) {
2657 try {
2658 return std::stoll(lit->value);
2659 } catch (...) {
2660 return std::nullopt;
2661 }
2662 }
2663
2664 if (auto unary = dynamic_cast<const BoundUnaryExpression *>(expr)) {
2665 auto inner = evaluateConstantInt(unary->expr.get());
2666 if (!inner)
2667 return std::nullopt;
2668 if (unary->op == "-")
2669 return -*inner;
2670 return inner;
2671 }
2672
2673 if (auto binary = dynamic_cast<const BoundBinaryExpression *>(expr)) {
2674 auto left = evaluateConstantInt(binary->left.get());
2675 auto right = evaluateConstantInt(binary->right.get());
2676 if (!left || !right)
2677 return std::nullopt;
2678
2679 if (binary->op == "+")
2680 return *left + *right;
2681 if (binary->op == "-")
2682 return *left - *right;
2683 if (binary->op == "*")
2684 return *left * *right;
2685 if (binary->op == "/")
2686 return *right == 0 ? std::nullopt
2687 : std::optional<int64_t>(*left / *right);
2688 if (binary->op == "%")
2689 return *right == 0 ? std::nullopt
2690 : std::optional<int64_t>(*left % *right);
2691 }

2693 return std::nullopt;
