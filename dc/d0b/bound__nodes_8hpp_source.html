121class BoundBlock : public BoundStatement {
122public:
123 std::vector<std::unique_ptr<BoundStatement>> statements;
124 std::unique_ptr<BoundExpression> result;
125 void accept(BoundVisitor &v) override { v.visit(*this); }
126 std::unique_ptr<BoundStatement> cloneStatement() const override {
127 auto cloned = std::make_unique<BoundBlock>();
128 for (const auto &stmt : statements)
129 cloned->statements.push_back(stmt->cloneStatement());
130 if (result)
131 cloned->result = result->clone();
132 return cloned;
133 }
134 std::unique_ptr<BoundBlock> cloneBlock() const {
135 auto res = cloneStatement();
136 return std::unique_ptr<BoundBlock>(
137 static_cast<BoundBlock *>(res.release()));
138 }
