378bool Binder::validateGenericConstraints(
379 const std::vector<GenericConstraint> &constraints,
380 std::unordered_map<std::string, std::shared_ptr<zir::Type>> &bindings,
381 std::string *failureReason) {
382 for (const auto &constraint : constraints) {
383 auto boundIt = bindings.find(constraint.parameterName);
384 if (boundIt == bindings.end() || !boundIt->second) {
385 if (failureReason) {
386 *failureReason = "missing binding for constrained type parameter '" +
387 constraint.parameterName + "'";
388 }
389 return false;
390 }
391 if (!constraint.boundType) {
392 continue;
393 }
394
395 activeGenericBindingsStack_.push_back(bindings);
396 auto requiredType = mapType(*constraint.boundType);
397 activeGenericBindingsStack_.pop_back();
398 if (!requiredType) {
399 if (failureReason) {
400 *failureReason =
401 "unknown constraint type for '" + constraint.parameterName + "'";
402 }
403 return false;
404 }
405 if (!canConvert(boundIt->second, requiredType)) {
406 if (failureReason) {
407 *failureReason =
408 "type parameter '" + constraint.parameterName + "' with type '" +
409 boundIt->second->toString() + "' does not satisfy constraint '" +
410 constraint.parameterName + ": " + requiredType->toString() + "'";
411 }
412 return false;
413 }
414 }
415 return true;
416}

678 std::static_pointer_cast<zir::ClassType>(instantiatedType);
679 auto &classInfo = classInfos_[instantiatedClassType->getName()];
680 auto oldScope = currentScope_;
681 auto oldModuleId = currentModuleId_;
682 currentScope_ = moduleIt->second.scope;
683 currentModuleId_ = moduleIt->second.info->moduleId;
684
685 for (const auto &methodDecl : classDecl->methods_) {
686 if (methodDecl->isUnsafe_) {
687 ++unsafeTypeContextDepth_;
688 }
689
690 std::unordered_map<std::string, std::shared_ptr<zir::Type>>
691 methodGenericBindings;
692 for (const auto &genericParam : methodDecl->genericParams_) {
693 if (genericParam) {
694 auto placeholder = std::make_shared<zir::RecordType>(
695 genericParam->typeName, genericParam->typeName);
696 methodGenericBindings[genericParam->typeName] = placeholder;
697 }
698 }
699
700 std::vector<std::shared_ptr<VariableSymbol>> params;
701 if (!methodDecl->isStatic_) {
702 params.push_back(std::make_shared<VariableSymbol>(
703 "self", instantiatedClassType, false, false, "self",
704 moduleIt->second.info->moduleName, Visibility::Private));
705 }
706
707 activeGenericBindingsStack_.push_back(genericBindings);
708 if (!methodGenericBindings.empty()) {
709 activeGenericBindingsStack_.push_back(methodGenericBindings);
710 }
711 for (size_t i = 0; i < methodDecl->params_.size(); ++i) {
712 const auto &p = methodDecl->params_[i];
713 auto mappedType = mapType(*p->type);
714 if (!mappedType) {
715 error(p->span, "Unknown type: " + p->type->qualifiedName());
716 mappedType =
717 std::make_shared<zir::PrimitiveType>(zir::TypeKind::Void);
718 }
719 params.push_back(std::make_shared<VariableSymbol>(
720 p->name, mappedType, false, p->isRef, p->name,
721 moduleIt->second.info->moduleName, Visibility::Private));
722 }
723
724 std::shared_ptr<zir::Type> retType;
725 bool isCtor = methodDecl->name_ == "init";
726 bool isDtor = methodDecl->name_ == "deinit";
727 if (isDtor && (!methodDecl->params_.empty() || methodDecl->returnType_)) {
728 error(methodDecl->span,
729 "Destructor 'deinit' cannot have parameters or a return type.");
730 }
731 if (isCtor || isDtor) {
732 retType = std::make_shared<zir::PrimitiveType>(zir::TypeKind::Void);
733 } else if (methodDecl->returnType_) {
734 retType = mapType(*methodDecl->returnType_);
735 } else {
736 retType = std::make_shared<zir::PrimitiveType>(zir::TypeKind::Void);
737 }
738 if (!methodGenericBindings.empty()) {
739 activeGenericBindingsStack_.pop_back();
740 }
741 activeGenericBindingsStack_.pop_back();
742
743 if (!retType) {
744 retType = std::make_shared<zir::PrimitiveType>(zir::TypeKind::Void);
745 }
746
747 if (methodDecl->isUnsafe_) {
748 --unsafeTypeContextDepth_;
749 }
750
751 auto methodSymbol = std::make_shared<FunctionSymbol>(
752 methodDecl->name_, std::move(params), std::move(retType), "",
753 moduleIt->second.info->moduleName, methodDecl->visibility_,
754 methodDecl->isUnsafe_);
755 for (const auto &genericParam : methodDecl->genericParams_) {
756 if (genericParam) {
757 methodSymbol->genericParameterNames.push_back(genericParam->typeName);
758 }
759 }
760 methodSymbol->isMethod = !methodDecl->isStatic_;
761 methodSymbol->isStatic = methodDecl->isStatic_;
762 methodSymbol->isConstructor = isCtor;
763 methodSymbol->isDestructor = isDtor;
764 methodSymbol->ownerTypeName = instantiatedClassType->getName();
765 if (methodSymbol->isMethod && !methodSymbol->isStatic &&
766 !methodSymbol->isConstructor && !methodSymbol->isDestructor) {
767 methodSymbol->vtableSlot =
768 findOverriddenVtableSlot(classInfo, *methodSymbol);
769 if (methodSymbol->vtableSlot < 0) {
770 methodSymbol->vtableSlot = classInfo.nextVirtualSlot++;
771 }
772 }
773 methodSymbol->linkName = mangleName(
774 moduleIt->second.info->linkPath.empty()
775 ? moduleIt->second.info->moduleId
776 : moduleIt->second.info->linkPath,
777 instantiatedClassType->getCodegenName() + "$" + methodDecl->name_ +
778 "$" + sanitizeTypeName(functionSignatureKey(*methodSymbol)));
779 functionDeclarationNodes_[methodSymbol.get()] = methodDecl.get();
780 functionDeclarationModuleIds_[methodSymbol.get()] =
781 moduleIt->second.info->moduleId;
782 functionGenericParamNames_[methodSymbol.get()] =
783 methodSymbol->genericParameterNames;
784 addClassMethodOverload(classInfo, methodSymbol);
785 if (isCtor) {
786 if (!classInfo.constructor) {
787 classInfo.constructor = methodSymbol;
788 }
789 } else if (isDtor) {
790 classInfo.destructor = methodSymbol;
791 }
792
793 if (!methodSymbol->genericParameterNames.empty() &&
794 !methodSymbol->isGenericInstantiation) {
795 continue;
796 }
797
798 auto oldScope = currentScope_;
799 auto oldFunction = currentFunction_;
800 auto oldModuleId = currentModuleId_;
801 auto oldClassStack = currentClassStack_;
802 int oldUnsafeDepth = unsafeDepth_;
803
804 currentModuleId_ = moduleIt->second.info->moduleId;
805 currentScope_ = moduleIt->second.scope;
806 currentFunction_ = methodSymbol;
807 currentClassStack_.push_back(instantiatedClassType->getName());
808 if (methodSymbol->isUnsafe) {
809 ++unsafeDepth_;
810 }
811
812 pushScope();
813 for (const auto &param : methodSymbol->parameters) {
814 if (!currentScope_->declare(param->name, param)) {
815 error(methodDecl->span,
816 "Parameter '" + param->name + "' already declared.");
817 }
818 }
819
820 activeGenericBindingsStack_.push_back(genericBindings);
821 auto boundBody = bindBody(methodDecl->body_.get(), false);
822 activeGenericBindingsStack_.pop_back();
823 popScope();
824
825 currentScope_ = oldScope;
826 currentFunction_ = oldFunction;
827 currentModuleId_ = oldModuleId;
828 currentClassStack_ = oldClassStack;
829 unsafeDepth_ = oldUnsafeDepth;
830
831 bool hasReturn = blockAlwaysReturns(boundBody.get());
832
833 if (!hasReturn &&
834 methodSymbol->returnType->getKind() != zir::TypeKind::Void) {
835 auto kind = methodSymbol->returnType->getKind();
836 if (methodSymbol->returnType->isInteger() ||
837 kind == zir::TypeKind::Float || kind == zir::TypeKind::Bool) {
838 std::string litVal = "0";
839 if (kind == zir::TypeKind::Float)
840 litVal = "0.0";
841 else if (kind == zir::TypeKind::Bool)
842 litVal = "false";
843 auto lit =
844 std::make_unique<BoundLiteral>(litVal, methodSymbol->returnType);
845 boundBody->statements.push_back(
846 std::make_unique<BoundReturnStatement>(std::move(lit)));
847 }
848
849 _diag.report(methodDecl->span, zap::DiagnosticLevel::Warning,
850 "Function '" + methodDecl->name_ +
851 "' has non-void return type but no return on some "
852 "paths.");
853 }
854
855 boundRoot_->functions.push_back(
856 std::make_unique<BoundFunctionDeclaration>(methodSymbol,
857 std::move(boundBody)));
858 }
859
860 currentScope_ = oldScope;
861 currentModuleId_ = oldModuleId;
862
863 auto boundRecord = std::make_unique<BoundRecordDeclaration>();
864 boundRecord->type = instantiatedClassType;
865 boundRoot_->records.push_back(std::move(boundRecord));
866 }
867
868 return instantiatedSymbol;
869}

878
879std::unordered_map<std::string, std::shared_ptr<zir::Type>>
880Binder::buildGenericBindings(
881 const FunctionSymbol &function,
882 const std::vector<std::unique_ptr<BoundExpression>> &arguments,
883 const std::vector<std::unique_ptr<TypeNode>> &explicitTypeArgs,
884 SourceSpan callSpan, std::string *failureReason) {
885 (void)callSpan;
886 std::unordered_map<std::string, std::shared_ptr<zir::Type>> bindings;
887
888 if (function.genericParameterNames.empty()) {
889 return bindings;
890 }
891
892 if (explicitTypeArgs.size() > function.genericParameterNames.size()) {
893 if (failureReason) {
894 *failureReason = "explicit generic argument count mismatch";
895 }
896 return {};
897 }
898
899 for (size_t i = 0; i < explicitTypeArgs.size(); ++i) {
900 if (!explicitTypeArgs[i]) {
901 continue;
902 }
903 auto mapped = mapType(*explicitTypeArgs[i]);
904 if (!mapped) {
905 if (failureReason) {
906 *failureReason = "unknown generic type argument";
907 }
908 return {};
909 }
910 bindings[function.genericParameterNames[i]] = mapped;
911 }
912
913 auto declIt = functionDeclarationNodes_.find(&function);
914 const auto *decl =
915 declIt == functionDeclarationNodes_.end() ? nullptr : declIt->second;
916
917 std::function<bool(const std::shared_ptr<zir::Type> &,
918 const std::shared_ptr<zir::Type> &)>
919 inferFrom = [&](const std::shared_ptr<zir::Type> &paramType,
920 const std::shared_ptr<zir::Type> &argType) -> bool {
921 if (!paramType || !argType) {
922 return true;
923 }
924
925 if (paramType->getKind() == zir::TypeKind::Record) {
926 auto rec = std::static_pointer_cast<zir::RecordType>(paramType);
927
928 if (isVariadicViewType(paramType)) {
929 const auto &fields = rec->getFields();
930 if (fields.size() < 2 ||
931 fields[0].type->getKind() != zir::TypeKind::Pointer) {
932 return false;
933 }
934 auto dataPtr =
935 std::static_pointer_cast<zir::PointerType>(fields[0].type);
936
937 if (argType->getKind() == zir::TypeKind::Array) {
938 auto aa = std::static_pointer_cast<zir::ArrayType>(argType);
939 return inferFrom(dataPtr->getBaseType(), aa->getBaseType());
940 }
941
942 if (argType->getKind() == zir::TypeKind::Record &&
943 isVariadicViewType(argType)) {
944 auto argRec = std::static_pointer_cast<zir::RecordType>(argType);
945 const auto &argFields = argRec->getFields();
946 if (argFields.size() < 2 ||
947 argFields[0].type->getKind() != zir::TypeKind::Pointer) {
948 return false;
949 }
950 auto argDataPtr =
951 std::static_pointer_cast<zir::PointerType>(argFields[0].type);
952 return inferFrom(dataPtr->getBaseType(), argDataPtr->getBaseType());
953 }
954 }
955
956 if (rec->isGenericInstance() &&
957 argType->getKind() == zir::TypeKind::Record) {
958 auto argRecord = std::static_pointer_cast<zir::RecordType>(argType);
959 if (!argRecord->isGenericInstance() ||
960 rec->getGenericBaseName() != argRecord->getGenericBaseName() ||
961 rec->getGenericArguments().size() !=
962 argRecord->getGenericArguments().size()) {
963 return false;
964 }
965 for (size_t i = 0; i < rec->getGenericArguments().size(); ++i) {
966 if (!inferFrom(rec->getGenericArguments()[i],
967 argRecord->getGenericArguments()[i])) {
968 return false;
969 }
970 }
971 return true;
972 }
973 auto paramName = rec->getName();
974 std::string normalizedParamName =
975 (!paramName.empty() && paramName[0] == '%') ? paramName.substr(1)
976 : paramName;
977 auto isGenericName = std::find(function.genericParameterNames.begin(),
978 function.genericParameterNames.end(),
979 normalizedParamName) !=
980 function.genericParameterNames.end();
981 if (isGenericName) {
982 auto it = bindings.find(normalizedParamName);
983 if (it == bindings.end()) {
984 bindings[normalizedParamName] = argType;
985 return true;
986 }
987 return it->second->toString() == argType->toString();
988 }
989 return true;
990 }
991
992 if (paramType->getKind() == zir::TypeKind::Class) {
993 auto cls = std::static_pointer_cast<zir::ClassType>(paramType);
994 if (cls->isGenericInstance() &&
995 argType->getKind() == zir::TypeKind::Class) {
996 auto argClass = std::static_pointer_cast<zir::ClassType>(argType);
997 std::shared_ptr<zir::ClassType> matchingArgClass = argClass;
998 while (matchingArgClass &&
999 (!matchingArgClass->isGenericInstance() ||
1000 cls->getGenericBaseName() !=
1001 matchingArgClass->getGenericBaseName())) {
1002 matchingArgClass = matchingArgClass->getBase();
1003 }
1004 if (!matchingArgClass ||
1005 matchingArgClass->getGenericArguments().size() !=
1006 cls->getGenericArguments().size()) {
1007 return false;
1008 }
1009 for (size_t i = 0; i < cls->getGenericArguments().size(); ++i) {
1010 if (!inferFrom(cls->getGenericArguments()[i],
1011 matchingArgClass->getGenericArguments()[i])) {
1012 return false;
1013 }
1014 }
1015 return true;
1016 }
1017 auto paramName = cls->getName();
1018 std::string normalizedParamName =
1019 (!paramName.empty() && paramName[0] == '%') ? paramName.substr(1)
1020 : paramName;
1021 auto isGenericName = std::find(function.genericParameterNames.begin(),
1022 function.genericParameterNames.end(),
1023 normalizedParamName) !=
1024 function.genericParameterNames.end();
1025 if (isGenericName) {
1026 auto it = bindings.find(normalizedParamName);
1027 if (it == bindings.end()) {
1028 bindings[normalizedParamName] = argType;
1029 return true;
1030 }
1031 return it->second->toString() == argType->toString();
1032 }
1033 return true;
1034 }
1035
1036 if (paramType->getKind() == zir::TypeKind::Pointer &&
1037 argType->getKind() == zir::TypeKind::Pointer) {
1038 auto pp = std::static_pointer_cast<zir::PointerType>(paramType);
1039 auto ap = std::static_pointer_cast<zir::PointerType>(argType);
1040 return inferFrom(pp->getBaseType(), ap->getBaseType());
1041 }
1042
1043 if (paramType->getKind() == zir::TypeKind::Array &&
1044 argType->getKind() == zir::TypeKind::Array) {
1045 auto pa = std::static_pointer_cast<zir::ArrayType>(paramType);
1046 auto aa = std::static_pointer_cast<zir::ArrayType>(argType);
1047 if (pa->getSize() != aa->getSize()) {
1048 return false;
1049 }
1050 return inferFrom(pa->getBaseType(), aa->getBaseType());
1051 }
1052
1053 return true;
1054 };
1055
1056 size_t fixedCount = function.fixedParameterCount();
1057 for (size_t i = 0; i < arguments.size() && i < fixedCount; ++i) {
1058 if (!inferFrom(function.parameters[i]->type, arguments[i]->type)) {
1059 if (failureReason) {
1060 *failureReason = "conflicting generic type inference";
1061 }
1062 return {};
1063 }
1064 }
1065
1066 for (const auto &name : function.genericParameterNames) {
1067 if (bindings.find(name) == bindings.end()) {
1068 bool filledFromDefault = false;
1069 if (decl) {
1070 auto paramIndex = static_cast<size_t>(std::distance(
1071 function.genericParameterNames.begin(),
1072 std::find(function.genericParameterNames.begin(),
1073 function.genericParameterNames.end(), name)));
1074 if (paramIndex < decl->genericParams_.size() &&
1075 decl->genericParams_[paramIndex] &&
1076 decl->genericParams_[paramIndex]->defaultType) {
1077 activeGenericBindingsStack_.push_back(bindings);
1078 auto mapped = mapType(*decl->genericParams_[paramIndex]->defaultType);
1079 activeGenericBindingsStack_.pop_back();
1080 if (mapped) {
1081 bindings[name] = mapped;
1082 filledFromDefault = true;
1083 }
1084 }
1085 }
1086 if (!filledFromDefault) {
1087 if (failureReason) {
1088 *failureReason = "cannot infer generic type parameter '" + name + "'";
1089 }
1090 return {};
1091 }
1092 }
1093 }
1094
1095 if (decl) {
1096 if (!validateGenericConstraints(decl->genericConstraints_, bindings,
1097 failureReason)) {
1098 return {};
1099 }
1100 }
1101
1102 return bindings;
1103}

1157int Binder::conversionCost(std::shared_ptr<zir::Type> from,
1158 std::shared_ptr<zir::Type> to) const {
1159 if (!from || !to) {
1160 return 1000;
1161 }
1162 if (from->toString() == to->toString()) {
1163 return 0;
1164 }
1165 if (isNullType(from) && isPointerType(to)) {
1166 return 3;
1167 }
1168 if (!canConvert(from, to)) {
1169 return 1000;
1170 }
1171 if (isStringType(from) && isStringType(to)) {
1172 return zap::text::isStringViewType(to) ? 0 : 1;
1173 }
1174
1175 if (from->getKind() == zir::TypeKind::Enum && to->isInteger()) {
1176 return 1;
1177 }
1178
1179 if (from->isFloatingPoint() && to->isFloatingPoint()) {
1180 return typeBitWidth(to) >= typeBitWidth(from) ? 1 : 5;
1181 }
1182
1183 if (isSignedIntegerType(from) && isSignedIntegerType(to)) {
1184 return typeBitWidth(to) >= typeBitWidth(from) ? 1 : 4;
1185 }
1186
1187 if (isUnsignedIntegerType(from) && isUnsignedIntegerType(to)) {
1188 return typeBitWidth(to) >= typeBitWidth(from) ? 1 : 4;
1189 }
1190
1191 if (from->isInteger() && to->isFloatingPoint()) {
1192 return typeBitWidth(to) >= typeBitWidth(from) ? 2 : 5;
1193 }
1194
1195 if (from->isFloatingPoint() && to->isInteger()) {
1196 return 6;
1197 }
1198
1199 if (from->isInteger() && to->isInteger()) {
1200 return 5;
1201 }
1202
1203 return 7;
1204}

1206std::string Binder::describeConversion(std::shared_ptr<zir::Type> from,
1207 std::shared_ptr<zir::Type> to) const {
1208 if (!from || !to) {
1209 return "invalid conversion";
1210 }
1211 int cost = conversionCost(from, to);
1212 if (cost == 0) {
1213 return "exact match";
1214 }
1215 if (isNullType(from) &&
1216 (isPointerType(to) || to->getKind() == zir::TypeKind::Class)) {
1217 return "null-to-reference conversion";
1218 }
1219 if (from->isFloatingPoint() && to->isFloatingPoint()) {
1220 return typeBitWidth(to) >= typeBitWidth(from) ? "floating widening"
1221 : "floating narrowing";
1222 }
1223 if (isSignedIntegerType(from) && isSignedIntegerType(to)) {
1224 return typeBitWidth(to) >= typeBitWidth(from) ? "signed widening"
1225 : "signed narrowing";
1226 }
1227 if (isUnsignedIntegerType(from) && isUnsignedIntegerType(to)) {
1228 return typeBitWidth(to) >= typeBitWidth(from) ? "unsigned widening"
1229 : "unsigned narrowing";
1230 }
1231 if (from->isInteger() && to->isFloatingPoint()) {
1232 return "integer-to-float conversion";
1233 }
1234 if (from->isFloatingPoint() && to->isInteger()) {
1235 return "float-to-integer conversion";
1236 }
1237 if (from->isInteger() && to->isInteger()) {
1238 return "signedness-changing integer conversion";
1239 }
1240 return cost >= 1000 ? "not convertible" : "implicit conversion";
1241}

1258void Binder::validateAndApplyTypeAttributes(
1259 const TopLevel &node, const std::shared_ptr<TypeSymbol> &symbol,
1260 bool allowErrorAttribute) {
1261 if (!symbol) {
1262 return;
1263 }
1264
1265 warnUnknownAttributes(node);
1266
1267 bool seenError = false;
1268 bool seenRepr = false;
1269 bool seenPacked = false;
1270
1271 for (const auto &attr : node.attributes_) {
1272 if (attr.name == "error") {
1273 if (!allowErrorAttribute) {
1274 error(attr.span, "attribute 'error' cannot be applied to this type");
1275 continue;
1276 }
1277 if (seenError) {
1278 error(attr.span, "duplicate attribute 'error'");
1279 continue;
1280 }
1281 if (attr.hasArguments()) {
1282 error(attr.span, "attribute 'error' does not accept arguments");
1283 continue;
1284 }
1285 seenError = true;
1286 symbol->isErrorType = true;
1287 continue;
1288 }
1289
1290 if (attr.name == "repr") {
1291 if (seenRepr) {
1292 error(attr.span, "duplicate attribute 'repr'");
1293 continue;
1294 }
1295 seenRepr = true;
1296
1297 if (attr.arguments.size() != 1 ||
1298 attr.arguments[0].kind != AttributeArgumentKind::Positional) {
1299 error(
1300 attr.span,
1301 "attribute 'repr' expects exactly one positional string argument");
1302 continue;
1303 }
1304
1305 auto *str = dynamic_cast<ConstString *>(attr.arguments[0].value.get());
1306 if (!str) {
1307 error(attr.span, "attribute 'repr' expects a string literal argument");
1308 continue;
1309 }
1310
1311 if (str->value_ != "C") {
1312 error(attr.span,
1313 "invalid argument for attribute 'repr': expected \"C\"");
1314 continue;
1315 }
1316
1317 symbol->hasReprC = true;
1318 symbol->reprValue = str->value_;
1319 continue;
1320 }
1321
1322 if (attr.name == "packed") {
1323 if (seenPacked) {
1324 error(attr.span, "duplicate attribute 'packed'");
1325 continue;
1326 }
1327 if (attr.hasArguments()) {
1328 error(attr.span, "attribute 'packed' does not accept arguments");
1329 continue;
1330 }
1331 auto recordType =
1332 std::dynamic_pointer_cast<zir::RecordType>(symbol->type);
1333 if (!recordType || symbol->type->getKind() != zir::TypeKind::Record) {
1334 error(attr.span,
1335 "attribute 'packed' can only be applied to record types");
1336 continue;
1337 }
1338 seenPacked = true;
1339 symbol->isPacked = true;
1340 recordType->isPacked = true;
1341 continue;
1342 }
1343
1344 if (attr.name == "extern" || attr.name == "noMangle") {
1345 error(attr.span, "attribute '" + attr.name +
1346 "' cannot be applied to type declarations");
1347 continue;
1348 }
1349 }
1350}
1351
1352void Binder::validateAndApplyFunctionAttributes(
1353 const TopLevel &node, const std::shared_ptr<FunctionSymbol> &symbol,
1354 bool isExternalDeclaration) {
1355 (void)isExternalDeclaration;
1356 if (!symbol) {
1357 return;
1358 }
1359
1360 warnUnknownAttributes(node);
1361
1362 bool seenExtern = false;
1363 bool seenNoMangle = false;
1364
1365 for (const auto &attr : node.attributes_) {
1366 if (attr.name == "extern") {
1367 if (seenExtern) {
1368 error(attr.span, "duplicate attribute 'extern'");
1369 continue;
1370 }
1371 seenExtern = true;
1372
1373 if (attr.arguments.size() != 1 ||

1420std::shared_ptr<zir::Type> Binder::mapType(const TypeNode &typeNode) {
1421 // Cache simple types when not in a generic instantiation context.
1422 // Skip cache for array types with size expressions (they evaluate AST nodes)
1423 // and for failable types (they may emit errors on first call).
1424 bool canCache = activeGenericBindingsStack_.empty() && !typeNode.isVarArgs &&
1425 !typeNode.isFailable &&
1426 !(typeNode.isArray && typeNode.arraySize);
1427 if (canCache) {
1428 auto it = mapTypeCache_.find(&typeNode);
1429 if (it != mapTypeCache_.end())
1430 return it->second;
1431 }
1432
1433 auto doMap = [&]() -> std::shared_ptr<zir::Type> {
1434 if (typeNode.isVarArgs) {
1435 if (!typeNode.baseType)
1436 return nullptr;
1437 return mapType(*typeNode.baseType);
1438 }
1439
1440 if (typeNode.isFailable) {
1441 if (!typeNode.baseType || !typeNode.errorType) {
1442 error(typeNode.span, "Invalid failable type declaration.");
1443 return nullptr;
1444 }
1445
1446 auto valueType = mapType(*typeNode.baseType);
1447 auto errorType = mapType(*typeNode.errorType);
1448
1449 if (!valueType || !errorType) {
1450 return nullptr;
1451 }
1452
1453 if (!typeNode.errorType->qualifiers.empty() ||
1454 !typeNode.errorType->typeName.empty()) {
1455 std::vector<std::string> errParts = typeNode.errorType->qualifiers;
1456 errParts.push_back(typeNode.errorType->typeName);
1457 auto errSymbol = resolveQualifiedSymbol(
1458 errParts, typeNode.errorType->span, SymbolKind::Type);
1459 if (errSymbol && errSymbol->getKind() == SymbolKind::Type) {
1460 auto errTypeSymbol = std::static_pointer_cast<TypeSymbol>(errSymbol);
1461 if (!errTypeSymbol->isErrorType) {
1462 error(typeNode.errorType->span,
1463 "Type '" + typeNode.errorType->qualifiedName() +
1464 "' used as failable error type must be annotated with "
1465 "@error.");
1466 }
1467 }
1468 }
1469
1470 return makeFailableType(valueType, errorType);
1471 }
1472
1473 if (!activeGenericBindingsStack_.empty() && typeNode.qualifiers.empty() &&
1474 typeNode.genericArgs.empty()) {
1475 const auto &bindings = activeGenericBindingsStack_.back();
1476 auto it = bindings.find(typeNode.typeName);
1477 if (it != bindings.end()) {
1478 auto mapped = it->second;
1479 if (typeNode.isWeak) {
1480 if (!mapped || mapped->getKind() != zir::TypeKind::Class) {
1481 error(typeNode.span, "'weak' can only be used with class types.");
1482 return nullptr;
1483 }
1484 auto weakType = std::make_shared<zir::ClassType>(
1485 *std::static_pointer_cast<zir::ClassType>(mapped));
1486 weakType->setWeak(true);
1487 return weakType;
1488 }
1489 return mapped;
1490 }
1491 }
1492
1493 if (typeNode.isArray) {
1494 if (!typeNode.baseType)
1495 return nullptr;
1496 auto base = mapType(*typeNode.baseType);
1497
1498 if (!typeNode.arraySize) {
1499 return makeVariadicViewType(base);
1500 }
1501
1502 size_t size = 0;
1503 typeNode.arraySize->accept(*this);
1504 if (!expressionStack_.empty()) {
1505 auto boundSize = std::move(expressionStack_.top());
1506 expressionStack_.pop();
1507 auto evaluated = evaluateConstantInt(boundSize.get());
1508 if (evaluated) {
1509 size = static_cast<size_t>(*evaluated);
1510 } else {
1511 error(typeNode.span,
1512 "Array size must be a constant integer expression.");
1513 }
1514 }
1515
1516 return std::make_shared<zir::ArrayType>(std::move(base), size);
1517 }
1518
1519 if (typeNode.isPointer) {
1520 if (!typeNode.baseType)
1521 return nullptr;
1522 auto base = mapType(*typeNode.baseType);
1523 return std::make_shared<zir::PointerType>(std::move(base));
1524 }
1525
1526 if (typeNode.isFunPtr) {
1527 std::vector<std::shared_ptr<zir::Type>> params;
1528 for (const auto &p : typeNode.funPtrParams) {
1529 auto mapped = mapType(*p);
1530 if (!mapped)
1531 return nullptr;
1532 params.push_back(std::move(mapped));
1533 }
1534 auto ret =
1535 typeNode.funPtrReturn
1536 ? mapType(*typeNode.funPtrReturn)
1537 : std::make_shared<zir::PrimitiveType>(zir::TypeKind::Void);
1538 if (!ret)
1539 return nullptr;
1540 return std::make_shared<zir::FunctionPointerType>(std::move(params),
1541 std::move(ret));
1542 }
1543
1544 std::vector<std::string> parts = typeNode.qualifiers;
1545 parts.push_back(typeNode.typeName);
1546 auto symbol =
1547 resolveQualifiedSymbol(parts, typeNode.span, SymbolKind::Type);
1548 if (symbol && symbol->getKind() == SymbolKind::Type) {
1549 auto typeSymbol = std::static_pointer_cast<TypeSymbol>(symbol);
1550 if (!typeSymbol->genericParameterNames.empty()) {
1551 typeSymbol = instantiateGenericTypeSymbol(typeSymbol, typeNode);
1552 if (!typeSymbol) {
1553 return nullptr;
1554 }
1555 } else if (!typeNode.genericArgs.empty()) {
1556 error(typeNode.span,
1557 "Type '" + typeNode.qualifiedName() + "' is not generic.");
1558 return nullptr;
1559 }
1560 if (typeSymbol->isUnsafe) {
1561 requireUnsafeContext(typeNode.span, "unsafe struct types");
1562 }
1563 if (typeNode.isWeak) {
1564 if (!typeSymbol->isClass ||
1565 typeSymbol->type->getKind() != zir::TypeKind::Class) {
1566 error(typeNode.span, "'weak' can only be used with class types.");
1567 return nullptr;
1568 }
1569 auto weakType = std::make_shared<zir::ClassType>(
1570 *std::static_pointer_cast<zir::ClassType>(typeSymbol->type));
1571 weakType->setWeak(true);
1572 return weakType;
1573 }
1574
1575 auto resolvedType = typeSymbol->type;
1576 if (typeNode.errorType && !typeSymbol->isErrorType) {
1577 error(typeNode.span,
1578 "Type '" + typeNode.qualifiedName() +
1579 "' used as failable error type must be annotated with "
1580 "@error.");
1581 }
1582 return resolvedType;
1583 }
1584
1585 return nullptr;
1586 }; // end doMap lambda
1587
1588 auto result = doMap();
1589 if (canCache)
1590 mapTypeCache_[&typeNode] = result;
1591 return result;
1592}

1622bool Binder::canConvert(std::shared_ptr<zir::Type> from,
1623 std::shared_ptr<zir::Type> to) const {
1624 if (!from || !to)
1625 return false;
1626 if (from->getKind() == to->getKind() && from->toString() == to->toString())
1627 return true;
1628
1629 auto key = std::make_pair(from.get(), to.get());
1630 auto cacheIt = canConvertCache_.find(key);
1631 if (cacheIt != canConvertCache_.end())
1632 return cacheIt->second;
1633
1634 auto &cached = canConvertCache_[key];
1635 if (isNullType(from) &&
1636 (isPointerType(to) || to->getKind() == zir::TypeKind::Class))
1637 return cached = true;
1638 if (isStringType(from) && isStringType(to))
1639 return cached = true;
1640 if (isStringType(from) && isPointerType(to)) {
1641 auto ptrType = std::static_pointer_cast<zir::PointerType>(to);
1642 return cached = (ptrType &&
1643 ptrType->getBaseType()->getKind() == zir::TypeKind::Char);
1644 }
1645 if (from->getKind() == zir::TypeKind::Class &&
1646 to->getKind() == zir::TypeKind::Class) {
1647 auto fromClass = std::static_pointer_cast<zir::ClassType>(from);
1648 auto toClass = std::static_pointer_cast<zir::ClassType>(to);
1649 if (fromClass->isWeak() && !toClass->isWeak())
1650 return cached = false;
1651 for (auto current = fromClass; current; current = current->getBase()) {
1652 if (current->getName() == toClass->getName())
1653 return cached = true;
1654 }
1655 }
1656 if (from->getKind() == zir::TypeKind::Array && isVariadicViewType(to)) {
1657 auto arrayType = std::static_pointer_cast<zir::ArrayType>(from);
1658 auto viewType = std::static_pointer_cast<zir::RecordType>(to);
1659 if (!viewType->getFields().empty() &&
1660 viewType->getFields()[0].type->getKind() == zir::TypeKind::Pointer) {
1661 auto dataType = std::static_pointer_cast<zir::PointerType>(
1662 viewType->getFields()[0].type);
1663 return cached = (arrayType->getBaseType()->toString() ==
1664 dataType->getBaseType()->toString());
1665 }
1666 }
1667 if (isFailableType(from) && isFailableType(to)) {
1668 auto fromValueType = failableValueType(from);
1669 auto fromErrorType = failableErrorType(from);
1670 auto toValueType = failableValueType(to);
1671 auto toErrorType = failableErrorType(to);
1672 return cached = (canConvert(fromValueType, toValueType) &&
1673 canConvert(fromErrorType, toErrorType));
1674 }
1675 if (isNumeric(from) && isNumeric(to))
1676 return cached = true;
1677 if (from->getKind() == zir::TypeKind::Enum &&
1678 to->getKind() == zir::TypeKind::Int)
1679 return cached = true;
1680 return cached = false;
1681}

1683std::shared_ptr<zir::Type>
1684Binder::getPromotedType(std::shared_ptr<zir::Type> t1,
1685 std::shared_ptr<zir::Type> t2) {
1686 if (t1->toString() == t2->toString())
1687 return t1;
1688
1689 if (t1->isFloatingPoint() || t2->isFloatingPoint()) {
1690 if (t1->getKind() == zir::TypeKind::Float64 ||
1691 t2->getKind() == zir::TypeKind::Float64) {
1692 return std::make_shared<zir::PrimitiveType>(zir::TypeKind::Float64);
1693 }
1694 return std::make_shared<zir::PrimitiveType>(zir::TypeKind::Float);
1695 }
1696
1697 // Integer promotion: preserve unsignedness when both operands are unsigned,
1698 // and choose width-aware integer kinds instead of always forcing Int64.
1699 if (isUnsignedIntegerType(t1) && isUnsignedIntegerType(t2)) {
1700 int width = std::max(typeBitWidth(t1), typeBitWidth(t2));
1701 if (width <= 8)
1702 return std::make_shared<zir::PrimitiveType>(zir::TypeKind::UInt8);
1703 if (width <= 16)
1704 return std::make_shared<zir::PrimitiveType>(zir::TypeKind::UInt16);
1705 if (width <= 32)
1706 return std::make_shared<zir::PrimitiveType>(zir::TypeKind::UInt);
1707 return std::make_shared<zir::PrimitiveType>(zir::TypeKind::UInt64);
1708 }
1709
1710 // Mixed signed/unsigned or both signed:
1711 // - preserve unsignedness when the unsigned operand width is >= signed width
1712 // (e.g. Int + UInt -> UInt, Int16 + UInt16 -> UInt16, Int + UInt64 ->
1713 // UInt64)
1714 // - otherwise use signed, width-aware promotion.
1715 if ((isUnsignedIntegerType(t1) && isSignedIntegerType(t2)) ||
1716 (isSignedIntegerType(t1) && isUnsignedIntegerType(t2))) {
1717 auto unsignedType = isUnsignedIntegerType(t1) ? t1 : t2;
1718 auto signedType = isUnsignedIntegerType(t1) ? t2 : t1;
1719
1720 int unsignedWidth = typeBitWidth(unsignedType);
1721 int signedWidth = typeBitWidth(signedType);
1722
1723 if (unsignedWidth >= signedWidth) {
1724 if (unsignedWidth <= 8)
1725 return std::make_shared<zir::PrimitiveType>(zir::TypeKind::UInt8);
1726 if (unsignedWidth <= 16)
1727 return std::make_shared<zir::PrimitiveType>(zir::TypeKind::UInt16);
1728 if (unsignedWidth <= 32)
1729 return std::make_shared<zir::PrimitiveType>(zir::TypeKind::UInt);
1730 return std::make_shared<zir::PrimitiveType>(zir::TypeKind::UInt64);
1731 }
1732
1733 int width = std::max(unsignedWidth, signedWidth);
1734 if (width <= 8)
1735 return std::make_shared<zir::PrimitiveType>(zir::TypeKind::Int8);
1736 if (width <= 16)
1737 return std::make_shared<zir::PrimitiveType>(zir::TypeKind::Int16);
1738 if (width <= 32)
1739 return std::make_shared<zir::PrimitiveType>(zir::TypeKind::Int);
1740 return std::make_shared<zir::PrimitiveType>(zir::TypeKind::Int64);
1741 }
1742
1743 // Both signed: keep signed result, width-aware.
1744 int width = std::max(typeBitWidth(t1), typeBitWidth(t2));
1745 if (width <= 8)
1746 return std::make_shared<zir::PrimitiveType>(zir::TypeKind::Int8);
1747 if (width <= 16)
1748 return std::make_shared<zir::PrimitiveType>(zir::TypeKind::Int16);
1749 if (width <= 32)
1750 return std::make_shared<zir::PrimitiveType>(zir::TypeKind::Int);
1751 return std::make_shared<zir::PrimitiveType>(zir::TypeKind::Int64);
1752}
