34 void BoundIRGenerator::visit(sema::BoundFunctionDeclaration &node)
35 {
36 auto symbol = node.symbol;
37 auto func =
38 std::make_unique<Function>(symbol->name, symbol->returnType->toString());
39 currentFunction_ = func.get();
40
41 auto entryBlock = std::make_unique<BasicBlock>("entry");
42 currentBlock_ = entryBlock.get();
43 currentFunction_->addBlock(std::move(entryBlock));
44
45 for (const auto &paramSymbol : symbol->parameters)
46 {
47 auto arg = std::make_shared<Argument>(paramSymbol->name, paramSymbol->type);
48 currentFunction_->arguments.push_back(arg);
49
50 // Create an alloca for the parameter to make it mutable (standard LLVM
51 // practice)
52 auto allocaReg =
53 createRegister(std::make_shared<PointerType>(paramSymbol->type));
54 currentBlock_->addInstruction(
55 std::make_unique<AllocaInst>(allocaReg, paramSymbol->type));
56 currentBlock_->addInstruction(std::make_unique<StoreInst>(arg, allocaReg));
57
58 symbolMap_[paramSymbol] = allocaReg;
59 }
60
61 if (node.body)
62 {
63 node.body->accept(*this);
64 }
65
66 if (currentBlock_ &&
67 (currentBlock_->instructions.empty() ||
68 currentBlock_->instructions.back()->getOpCode() != OpCode::Ret))
69 {
70 if (symbol->returnType->getKind() == TypeKind::Void)
71 {
72 currentBlock_->addInstruction(std::make_unique<ReturnInst>());
73 }
74 else
75 {
76 auto dummy = std::make_shared<Constant>("0", symbol->returnType);
77 currentBlock_->addInstruction(std::make_unique<ReturnInst>(dummy));
78 }
79 }
80
81 module_->addFunction(std::move(func));
82 currentFunction_ = nullptr;
83 currentBlock_ = nullptr;
84 symbolMap_.clear();
85 }

182 void BoundIRGenerator::visit(sema::BoundBinaryExpression &node)
183 {
184 if (node.op == "&&")
185 {
186 auto rhsLabel = createBlockLabel("and.rhs");
187 auto mergeLabel = createBlockLabel("and.merge");
188
189 node.left->accept(*this);
190 auto leftVal = valueStack_.top();
191 valueStack_.pop();
192 std::string leftBlockLabel = currentBlock_->label;
193
194 currentBlock_->addInstruction(
195 std::make_unique<CondBranchInst>(leftVal, rhsLabel, mergeLabel));
196
197 auto rhsBlock = std::make_unique<BasicBlock>(rhsLabel);
198 auto *rhsBlockPtr = rhsBlock.get();
199 currentFunction_->addBlock(std::move(rhsBlock));
200 currentBlock_ = rhsBlockPtr;
201
202 node.right->accept(*this);
203 auto rightVal = valueStack_.top();
204 valueStack_.pop();
205 std::string actualRhsBlockLabel = currentBlock_->label;
206 currentBlock_->addInstruction(std::make_unique<BranchInst>(mergeLabel));
207
208 auto mergeBlock = std::make_unique<BasicBlock>(mergeLabel);
209 auto *mergeBlockPtr = mergeBlock.get();
210 currentFunction_->addBlock(std::move(mergeBlock));
211 currentBlock_ = mergeBlockPtr;
212
213 auto res = createRegister(node.type);
214 std::vector<std::pair<std::string, std::shared_ptr<Value>>> incoming;
215 incoming.push_back({leftBlockLabel, std::make_shared<Constant>("false", node.type)});
216 incoming.push_back({actualRhsBlockLabel, rightVal});
217
218 currentBlock_->addInstruction(std::make_unique<PhiInst>(res, incoming));
219 valueStack_.push(res);
220 return;
221 }
222
223 if (node.op == "||")
224 {
225 auto rhsLabel = createBlockLabel("or.rhs");
226 auto mergeLabel = createBlockLabel("or.merge");
227
228 node.left->accept(*this);
229 auto leftVal = valueStack_.top();
230 valueStack_.pop();
231 std::string leftBlockLabel = currentBlock_->label;
232
233 currentBlock_->addInstruction(
234 std::make_unique<CondBranchInst>(leftVal, mergeLabel, rhsLabel));
235
236 auto rhsBlock = std::make_unique<BasicBlock>(rhsLabel);
237 auto *rhsBlockPtr = rhsBlock.get();
238 currentFunction_->addBlock(std::move(rhsBlock));
239 currentBlock_ = rhsBlockPtr;
240
241 node.right->accept(*this);
242 auto rightVal = valueStack_.top();
243 valueStack_.pop();
244 std::string actualRhsBlockLabel = currentBlock_->label;
245 currentBlock_->addInstruction(std::make_unique<BranchInst>(mergeLabel));
246
247 auto mergeBlock = std::make_unique<BasicBlock>(mergeLabel);
248 auto *mergeBlockPtr = mergeBlock.get();
249 currentFunction_->addBlock(std::move(mergeBlock));
250 currentBlock_ = mergeBlockPtr;
251
252 auto res = createRegister(node.type);
253 std::vector<std::pair<std::string, std::shared_ptr<Value>>> incoming;
254 incoming.push_back({leftBlockLabel, std::make_shared<Constant>("true", node.type)});
255 incoming.push_back({actualRhsBlockLabel, rightVal});
256
257 currentBlock_->addInstruction(std::make_unique<PhiInst>(res, incoming));
258 valueStack_.push(res);
259 return;
260 }
261
262 node.left->accept(*this);
263 auto left = valueStack_.top();
264 valueStack_.pop();
265
266 node.right->accept(*this);
267 auto right = valueStack_.top();
268 valueStack_.pop();
269
270 auto reg = createRegister(node.type);
271 if (node.op == "==" || node.op == "!=" || node.op == "<" ||
272 node.op == ">" || node.op == "<=" || node.op == ">=")
273 {
274 std::string pred;
275 if (node.op == "==") pred = "eq";
276 else if (node.op == "!=") pred = "ne";
277 else if (node.op == "<") pred = "slt";
278 else if (node.op == ">") pred = "sgt";
279 else if (node.op == "<=") pred = "sle";
280 else if (node.op == ">=") pred = "sge";
281
282 currentBlock_->addInstruction(
283 std::make_unique<CmpInst>(pred, reg, left, right));
284 }
285 else
286 {
287 OpCode op;
288 if (node.op == "+")
289 op = OpCode::Add;
290 else if (node.op == "-")
291 op = OpCode::Sub;
292 else if (node.op == "*")
293 op = OpCode::Mul;
294 else if (node.op == "/")
295 op = OpCode::Div;
296 else
297 op = OpCode::Add;
298
299 currentBlock_->addInstruction(
300 std::make_unique<BinaryInst>(op, reg, left, right));
301 }
302 valueStack_.push(reg);
303 }

352 void BoundIRGenerator::visit(sema::BoundMemberAccess &node)
353 {
354 node.left->accept(*this);
355 auto left = std::move(valueStack_.top());
356 valueStack_.pop();
357
358 if (left->getType()->getKind() == zir::TypeKind::Enum)
359 {
360 auto enumType = std::static_pointer_cast<zir::EnumType>(left->getType());
361 int value = enumType->getVariantIndex(node.member);
362 if (value != -1)
363 {
364 valueStack_.push(std::make_shared<Constant>(
365 std::to_string(value),
366 std::make_shared<zir::PrimitiveType>(zir::TypeKind::Int)));
367 return;
368 }
369 }
370 else if (left->getType()->getKind() == zir::TypeKind::Record)
371 {
372 auto recordType = std::static_pointer_cast<zir::RecordType>(left->getType());
373 int fieldIndex = -1;
374 const auto &fields = recordType->getFields();
375 for (size_t i = 0; i < fields.size(); ++i)
376 {
377 if (fields[i].name == node.member)
378 {
379 fieldIndex = static_cast<int>(i);
380 break;
381 }
382 }
383
384 if (fieldIndex != -1)
385 {
386 auto result = createRegister(fields[fieldIndex].type);
387 currentBlock_->addInstruction(std::make_unique<GetElementPtrInst>(result, left, fieldIndex));
388 valueStack_.push(result);
389 return;
390 }
391 }
392
393 throw std::runtime_error("Member '" + node.member + "' not found in type '" + left->getTypeName() + "'");
394 }

396 void BoundIRGenerator::visit(sema::BoundStructLiteral &node)
397 {
398 auto recordType = std::static_pointer_cast<zir::RecordType>(node.type);
399 auto allocaReg = createRegister(std::make_shared<PointerType>(recordType));
400 currentBlock_->addInstruction(std::make_unique<AllocaInst>(allocaReg, recordType));
401
402 for (const auto &fieldInit : node.fields)
403 {
404 // Find field index
405 int fieldIndex = -1;
406 const auto &fields = recordType->getFields();
407 for (size_t i = 0; i < fields.size(); ++i)
408 {
409 if (fields[i].name == fieldInit.first)
410 {
411 fieldIndex = static_cast<int>(i);
412 break;
413 }
414 }
415
416 fieldInit.second->accept(*this);
417 auto val = std::move(valueStack_.top());
418 valueStack_.pop();
419
420 auto fieldAddr = createRegister(std::make_shared<PointerType>(fields[fieldIndex].type));
421 currentBlock_->addInstruction(std::make_unique<GetElementPtrInst>(fieldAddr, allocaReg, fieldIndex));
422 currentBlock_->addInstruction(std::make_unique<StoreInst>(val, fieldAddr));
423 }
424
425 auto result = createRegister(recordType);
426 currentBlock_->addInstruction(std::make_unique<LoadInst>(result, allocaReg));

430 void BoundIRGenerator::visit(sema::BoundIfExpression &node)
431 {
432 auto trueLabel = createBlockLabel("if.then");
433 auto falseLabel = node.elseBody ? createBlockLabel("if.else") : "";
434 auto mergeLabel = createBlockLabel("if.merge");
435
436 node.condition->accept(*this);
437 auto cond = valueStack_.top();
438 valueStack_.pop();
439
440 if (node.elseBody)
441 {
442 currentBlock_->addInstruction(
443 std::make_unique<CondBranchInst>(cond, trueLabel, falseLabel));
444 }
445 else
446 {
447 currentBlock_->addInstruction(
448 std::make_unique<CondBranchInst>(cond, trueLabel, mergeLabel));
449 }
450
451 auto thenBlock = std::make_unique<BasicBlock>(trueLabel);
452 auto *thenBlockPtr = thenBlock.get();
453 currentFunction_->addBlock(std::move(thenBlock));
454 currentBlock_ = thenBlockPtr;
455
456 if (node.thenBody)
457 node.thenBody->accept(*this);
458
459 std::shared_ptr<Value> thenVal = nullptr;
460 if (!valueStack_.empty())
461 {
462 thenVal = valueStack_.top();
463 valueStack_.pop();
464 }
465 std::string actualThenLabel = currentBlock_->label;
466
467 if (currentBlock_->instructions.empty() ||
468 currentBlock_->instructions.back()->getOpCode() != OpCode::Ret)
469 {
470 currentBlock_->addInstruction(std::make_unique<BranchInst>(mergeLabel));
471 }
472
473 std::shared_ptr<Value> elseVal = nullptr;
474 std::string actualElseLabel = "";
475 if (node.elseBody)
476 {
477 auto elseBlock = std::make_unique<BasicBlock>(falseLabel);
478 auto *elseBlockPtr = elseBlock.get();
479 currentFunction_->addBlock(std::move(elseBlock));
480 currentBlock_ = elseBlockPtr;
481 node.elseBody->accept(*this);
482
483 if (!valueStack_.empty())
484 {
485 elseVal = valueStack_.top();
486 valueStack_.pop();
487 }
488 actualElseLabel = currentBlock_->label;
489
490 if (currentBlock_->instructions.empty() ||
491 currentBlock_->instructions.back()->getOpCode() != OpCode::Ret)
492 {
493 currentBlock_->addInstruction(std::make_unique<BranchInst>(mergeLabel));
494 }
495 }
496
497 auto mergeBlock = std::make_unique<BasicBlock>(mergeLabel);
498 auto *mergeBlockPtr = mergeBlock.get();
499 currentFunction_->addBlock(std::move(mergeBlock));
500 currentBlock_ = mergeBlockPtr;
501
502 if (node.type->getKind() != TypeKind::Void)
503 {
504 auto res = createRegister(node.type);
505 std::vector<std::pair<std::string, std::shared_ptr<Value>>> incoming;
506
507 auto finalThenVal =
508 thenVal ? thenVal : std::make_shared<Constant>("undef", node.type);
509 incoming.push_back({actualThenLabel, finalThenVal});
510
511 if (node.elseBody)
512 {
513 auto finalElseVal =
514 elseVal ? elseVal : std::make_shared<Constant>("undef", node.type);
515 incoming.push_back({actualElseLabel, finalElseVal});
516 }
517 currentBlock_->addInstruction(std::make_unique<PhiInst>(res, incoming));
518 valueStack_.push(res);
519 }
