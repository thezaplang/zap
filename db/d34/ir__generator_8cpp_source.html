27void BoundIRGenerator::visit(sema::BoundFunctionDeclaration &node) {
28 auto symbol = node.symbol;
29 auto func = std::make_unique<Function>(symbol->linkName,
30 symbol->returnType->toString());
31 currentFunction_ = func.get();
32
33 auto entryBlock = std::make_unique<BasicBlock>("entry");
34 currentBlock_ = entryBlock.get();
35 currentFunction_->addBlock(std::move(entryBlock));
36
37 for (const auto &paramSymbol : symbol->parameters) {
38 auto arg = std::make_shared<Argument>(paramSymbol->name, paramSymbol->type);
39 currentFunction_->arguments.push_back(arg);
40
41 auto allocaReg =
42 createRegister(std::make_shared<PointerType>(paramSymbol->type));
43 currentBlock_->addInstruction(
44 std::make_unique<AllocaInst>(allocaReg, paramSymbol->type));
45 currentBlock_->addInstruction(std::make_unique<StoreInst>(arg, allocaReg));
46
47 symbolMap_[paramSymbol] = allocaReg;
48 }
49
50 if (node.body) {
51 node.body->accept(*this);
52 }
53
54 if (currentBlock_ &&
55 (currentBlock_->instructions.empty() ||
56 currentBlock_->instructions.back()->getOpCode() != OpCode::Ret)) {
57 if (symbol->returnType->getKind() == TypeKind::Void) {
58 currentBlock_->addInstruction(std::make_unique<ReturnInst>());
59 } else {
60 auto dummy = std::make_shared<Constant>("0", symbol->returnType);
61 currentBlock_->addInstruction(std::make_unique<ReturnInst>(dummy));
62 }
63 }
64
65 module_->addFunction(std::move(func));
66 currentFunction_ = nullptr;
67 currentBlock_ = nullptr;
68 symbolMap_.clear();
69}

152void BoundIRGenerator::visit(sema::BoundBinaryExpression &node) {
153 if (node.op == "&&") {
154 auto rhsLabel = createBlockLabel("and.rhs");
155 auto mergeLabel = createBlockLabel("and.merge");
156
157 node.left->accept(*this);
158 auto leftVal = valueStack_.top();
159 valueStack_.pop();
160 std::string leftBlockLabel = currentBlock_->label;
161
162 currentBlock_->addInstruction(
163 std::make_unique<CondBranchInst>(leftVal, rhsLabel, mergeLabel));
164
165 auto rhsBlock = std::make_unique<BasicBlock>(rhsLabel);
166 auto *rhsBlockPtr = rhsBlock.get();
167 currentFunction_->addBlock(std::move(rhsBlock));
168 currentBlock_ = rhsBlockPtr;
169
170 node.right->accept(*this);
171 auto rightVal = valueStack_.top();
172 valueStack_.pop();
173 std::string actualRhsBlockLabel = currentBlock_->label;
174 currentBlock_->addInstruction(std::make_unique<BranchInst>(mergeLabel));
175
176 auto mergeBlock = std::make_unique<BasicBlock>(mergeLabel);
177 auto *mergeBlockPtr = mergeBlock.get();
178 currentFunction_->addBlock(std::move(mergeBlock));
179 currentBlock_ = mergeBlockPtr;
180
181 auto res = createRegister(node.type);
182 std::vector<std::pair<std::string, std::shared_ptr<Value>>> incoming;
183 incoming.push_back(
184 {leftBlockLabel, std::make_shared<Constant>("false", node.type)});
185 incoming.push_back({actualRhsBlockLabel, rightVal});
186
187 currentBlock_->addInstruction(std::make_unique<PhiInst>(res, incoming));
188 valueStack_.push(res);
189 return;
190 }
191
192 if (node.op == "||") {
193 auto rhsLabel = createBlockLabel("or.rhs");
194 auto mergeLabel = createBlockLabel("or.merge");
195
196 node.left->accept(*this);
197 auto leftVal = valueStack_.top();
198 valueStack_.pop();
199 std::string leftBlockLabel = currentBlock_->label;
200
201 currentBlock_->addInstruction(
202 std::make_unique<CondBranchInst>(leftVal, mergeLabel, rhsLabel));
203
204 auto rhsBlock = std::make_unique<BasicBlock>(rhsLabel);
205 auto *rhsBlockPtr = rhsBlock.get();
206 currentFunction_->addBlock(std::move(rhsBlock));
207 currentBlock_ = rhsBlockPtr;
208
209 node.right->accept(*this);
210 auto rightVal = valueStack_.top();
211 valueStack_.pop();
212 std::string actualRhsBlockLabel = currentBlock_->label;
213 currentBlock_->addInstruction(std::make_unique<BranchInst>(mergeLabel));
214
215 auto mergeBlock = std::make_unique<BasicBlock>(mergeLabel);
216 auto *mergeBlockPtr = mergeBlock.get();
217 currentFunction_->addBlock(std::move(mergeBlock));
218 currentBlock_ = mergeBlockPtr;
219
220 auto res = createRegister(node.type);
221 std::vector<std::pair<std::string, std::shared_ptr<Value>>> incoming;
222 incoming.push_back(
223 {leftBlockLabel, std::make_shared<Constant>("true", node.type)});
224 incoming.push_back({actualRhsBlockLabel, rightVal});
225
226 currentBlock_->addInstruction(std::make_unique<PhiInst>(res, incoming));
227 valueStack_.push(res);
228 return;
229 }
230
231 node.left->accept(*this);
232 auto left = valueStack_.top();
233 valueStack_.pop();
234
235 node.right->accept(*this);
236 auto right = valueStack_.top();
237 valueStack_.pop();
238
239 auto reg = createRegister(node.type);
240 bool isUnsigned = node.left->type->isUnsigned();
241 if (node.op == "==" || node.op == "!=" || node.op == "<" ||
242 (node.op == ">") || (node.op == "<=") || (node.op == ">=")) {
243 std::string pred;
244 if (node.op == "==")
245 pred = "eq";
246 else if (node.op == "!=")
247 pred = "ne";
248 else if (node.op == "<")
249 pred = isUnsigned ? "ult" : "slt";
250 else if (node.op == ">")
251 pred = isUnsigned ? "ugt" : "sgt";
252 else if (node.op == "<=")
253 pred = isUnsigned ? "ule" : "sle";
254 else if (node.op == ">=")
255 pred = isUnsigned ? "uge" : "sge";
256
257 currentBlock_->addInstruction(
258 std::make_unique<CmpInst>(pred, reg, left, right));
259 } else {
260 OpCode op;
261 if (node.op == "+")
262 op = OpCode::Add;
263 else if (node.op == "-")
264 op = OpCode::Sub;
265 else if (node.op == "*")
266 op = OpCode::Mul;
267 else if (node.op == "/")
268 op = isUnsigned ? OpCode::UDiv : OpCode::SDiv;
269 else if (node.op == "%")
270 op = isUnsigned ? OpCode::URem : OpCode::SRem;
271 else
272 op = OpCode::Add;
273
274 currentBlock_->addInstruction(
275 std::make_unique<BinaryInst>(op, reg, left, right));
276 }
277 valueStack_.push(reg);
278}

366void BoundIRGenerator::visit(sema::BoundMemberAccess &node) {
367 node.left->accept(*this);
368 auto left = std::move(valueStack_.top());
369 valueStack_.pop();
370
371 if (left->getType()->getKind() == zir::TypeKind::Enum) {
372 auto enumType = std::static_pointer_cast<zir::EnumType>(left->getType());
373 int value = enumType->getVariantIndex(node.member);
374 if (value != -1) {
375 valueStack_.push(std::make_shared<Constant>(
376 std::to_string(value),
377 std::make_shared<zir::PrimitiveType>(zir::TypeKind::Int)));
378 return;
379 }
380 } else if (left->getType()->getKind() == zir::TypeKind::Record) {
381 auto recordType =
382 std::static_pointer_cast<zir::RecordType>(left->getType());
383 int fieldIndex = -1;
384 const auto &fields = recordType->getFields();
385 for (size_t i = 0; i < fields.size(); ++i) {
386 if (fields[i].name == node.member) {
387 fieldIndex = static_cast<int>(i);
388 break;
389 }
390 }
391
392 if (fieldIndex != -1) {
393 auto result = createRegister(fields[fieldIndex].type);
394 currentBlock_->addInstruction(
395 std::make_unique<GetElementPtrInst>(result, left, fieldIndex));
396 valueStack_.push(result);
397 return;
398 }
399 }
400
401 throw std::runtime_error("Member '" + node.member + "' not found in type '" +
402 left->getTypeName() + "'");
403}

405void BoundIRGenerator::visit(sema::BoundStructLiteral &node) {
406 auto recordType = std::static_pointer_cast<zir::RecordType>(node.type);
407 auto allocaReg = createRegister(std::make_shared<PointerType>(recordType));
408 currentBlock_->addInstruction(
409 std::make_unique<AllocaInst>(allocaReg, recordType));
410
411 for (const auto &fieldInit : node.fields) {
412 // Find field index
413 int fieldIndex = -1;
414 const auto &fields = recordType->getFields();
415 for (size_t i = 0; i < fields.size(); ++i) {
416 if (fields[i].name == fieldInit.first) {
417 fieldIndex = static_cast<int>(i);
418 break;
419 }
420 }
421
422 fieldInit.second->accept(*this);
423 auto val = std::move(valueStack_.top());
424 valueStack_.pop();
425
426 auto fieldAddr =
427 createRegister(std::make_shared<PointerType>(fields[fieldIndex].type));
428 currentBlock_->addInstruction(
429 std::make_unique<GetElementPtrInst>(fieldAddr, allocaReg, fieldIndex));
430 currentBlock_->addInstruction(std::make_unique<StoreInst>(val, fieldAddr));
431 }
432
433 auto result = createRegister(recordType);
434 currentBlock_->addInstruction(std::make_unique<LoadInst>(result, allocaReg));
435 valueStack_.push(result);
436}

443void BoundIRGenerator::visit(sema::BoundIfStatement &node) {
444 auto trueLabel = createBlockLabel("if.then");
445 auto falseLabel = node.elseBody ? createBlockLabel("if.else") : "";
446 auto mergeLabel = createBlockLabel("if.merge");
447
448 node.condition->accept(*this);
449 auto cond = valueStack_.top();
450 valueStack_.pop();
451
452 if (node.elseBody) {
453 currentBlock_->addInstruction(
454 std::make_unique<CondBranchInst>(cond, trueLabel, falseLabel));
455 } else {
456 currentBlock_->addInstruction(
457 std::make_unique<CondBranchInst>(cond, trueLabel, mergeLabel));
458 }
459
460 auto thenBlock = std::make_unique<BasicBlock>(trueLabel);
461 auto *thenBlockPtr = thenBlock.get();
462 currentFunction_->addBlock(std::move(thenBlock));
463 currentBlock_ = thenBlockPtr;
464
465 if (node.thenBody)
466 node.thenBody->accept(*this);
467
468 std::string actualThenLabel = currentBlock_->label;
469
470 if (currentBlock_->instructions.empty() ||
471 currentBlock_->instructions.back()->getOpCode() != OpCode::Ret) {
472 currentBlock_->addInstruction(std::make_unique<BranchInst>(mergeLabel));
473 }
474
475 std::string actualElseLabel = "";
476 if (node.elseBody) {
477 auto elseBlock = std::make_unique<BasicBlock>(falseLabel);
478 auto *elseBlockPtr = elseBlock.get();
479 currentFunction_->addBlock(std::move(elseBlock));
480 currentBlock_ = elseBlockPtr;
481 node.elseBody->accept(*this);
482
483 actualElseLabel = currentBlock_->label;
484
485 if (currentBlock_->instructions.empty() ||
486 currentBlock_->instructions.back()->getOpCode() != OpCode::Ret) {
487 currentBlock_->addInstruction(std::make_unique<BranchInst>(mergeLabel));
488 }
489 }
490
491 auto mergeBlock = std::make_unique<BasicBlock>(mergeLabel);
492 auto *mergeBlockPtr = mergeBlock.get();
493 currentFunction_->addBlock(std::move(mergeBlock));
494 currentBlock_ = mergeBlockPtr;
495}
