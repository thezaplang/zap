307 zap_arc_collecting = 0;

312 size_t ws_count = 0;

314 if (!zap_arc_ws_push(&zap_arc_ws, &ws_count, &zap_arc_ws_cap, &zap_arc_map,

369 if (!header->alive || !header->metadata) {

386 if (!header->alive || reachable[i]) {

391 }
392 for (size_t i = 0; i < ws_count; ++i) {

397 }
