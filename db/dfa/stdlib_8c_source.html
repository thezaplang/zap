859 char *buffer = zap_copy_path(path);
860 if (!buffer) {

862 }

865 free(buffer);
866 if (!file) {
867 return (zap_string_t){.ptr = "", .len = 0};
868 }
869

871 fclose(file);
872 return (zap_string_t){.ptr = "", .len = 0};

877 fclose(file);

893 fclose(file);

895 zap_string_release_ptr(content);
896 return (zap_string_t){.ptr = "", .len = 0};
