18std::optional<std::string>
19Workspace::sourceForPath(const std::filesystem::path &path) const {
20 std::string key = std::filesystem::weakly_canonical(path).string();
21 auto uriIt = uriByCanonicalPath_.find(key);
22 if (uriIt != uriByCanonicalPath_.end()) {
23 auto docIt = documentsByUri_.find(uriIt->second);
24 if (docIt != documentsByUri_.end()) {
25 return docIt->second.text;
26 }
27 }
28
29 std::error_code ec;
30 auto writeTime = std::filesystem::last_write_time(path, ec);
31 if (!ec) {
32 auto cacheIt = fileContentCache_.find(key);
33 if (cacheIt != fileContentCache_.end() &&
34 cacheIt->second.lastWriteTime == writeTime) {
35 return cacheIt->second.content;
36 }
37 }
38
39 std::string content;
40 if (!readSourceFile(path, content)) {
41 return std::nullopt;
42 }
43 if (!ec) {
44 fileContentCache_[key] = CachedFile{writeTime, content};
45 }
46 return content;
47}

49bool Workspace::loadModuleGraph(
50 const std::filesystem::path &modulePath,
51 std::map<std::string, std::unique_ptr<sema::ModuleInfo>> &modules,
52 std::set<std::string> &visiting, AnalysisResult &result,
53 const std::string &entryUri, const zap::args::ImportMap &importMap,
54 bool allowEntryErrors) const {
55 std::filesystem::path canonicalPath =
56 std::filesystem::weakly_canonical(modulePath);
57 std::string moduleId = canonicalPath.string();
58 std::string entryModuleId =
59 std::filesystem::weakly_canonical(documentsByUri_.at(entryUri).path)
60 .string();
61 bool isEntryModule = moduleId == entryModuleId;
62 if (modules.find(moduleId) != modules.end()) {
63 return true;
64 }
65 if (visiting.count(moduleId)) {
66 return false;
67 }
68
69 visiting.insert(moduleId);
70
71 auto source = sourceForPath(canonicalPath);
72 if (!source) {
73 visiting.erase(moduleId);
74 return false;
75 }
76
77 zap::DiagnosticEngine diagnostics(*source, canonicalPath.string());
78 Lexer lex(diagnostics);
79 auto tokens = lex.tokenize(*source);
80 zap::Parser parser(tokens, diagnostics);
81 auto ast = parser.parse();
82
83 auto uriIt = uriByCanonicalPath_.find(moduleId);
84 if (uriIt != uriByCanonicalPath_.end()) {
85 result.diagnosticsByUri[uriIt->second] = diagnostics.diagnostics();
86 } else if (moduleId == std::filesystem::weakly_canonical(
87 documentsByUri_.at(entryUri).path)
88 .string()) {
89 result.diagnosticsByUri[entryUri] = diagnostics.diagnostics();
90 }
91
92 if (!ast ||
93 (diagnostics.hadErrors() && !(allowEntryErrors && isEntryModule))) {
94 visiting.erase(moduleId);
95 return false;
96 }
97
98 auto module = std::make_unique<sema::ModuleInfo>();
99 module->moduleId = moduleId;
100 module->moduleName = canonicalPath.stem().string();
101 module->linkPath = zap::frontend::computeLogicalModulePath(
102 canonicalPath, runtimePaths(), importMap);
103 module->sourceName = canonicalPath.string();
104 module->sourceText = *source;
105 module->root = std::move(ast);
106
107 zap::frontend::injectImplicitPreludeImportIfNeeded(*module, true);
108
109 for (const auto &child : module->root->children) {
110 auto importNode = dynamic_cast<ImportNode *>(child.get());
111 if (!importNode) {
112 continue;
113 }
114
115 std::vector<std::filesystem::path> importTargets;
116 if (!zap::frontend::resolveImportTargets(canonicalPath, *importNode,
117 importTargets, importMap,
118 runtimePaths())) {
119 continue;
120 }
121
122 module->imports.push_back(
123 zap::frontend::makeResolvedImport(*importNode, importTargets));
124 }
125
126 for (const auto &import : module->imports) {
127 for (const auto &targetId : import.targetModuleIds) {
128 if (!loadModuleGraph(targetId, modules, visiting, result, entryUri,
129 importMap, allowEntryErrors)) {
130 visiting.erase(moduleId);
131 return false;
132 }
133 }
134 }
135
136 visiting.erase(moduleId);
137 modules[moduleId] = std::move(module);
138 return true;
139}

146void Workspace::open(const std::string &uri, std::filesystem::path path,
147 std::string text, int64_t version) {
148 std::filesystem::path canonicalPath = std::filesystem::weakly_canonical(path);
149 std::string canonicalKey = canonicalPath.string();
150
151 auto existingUriIt = uriByCanonicalPath_.find(canonicalKey);
152 if (existingUriIt != uriByCanonicalPath_.end() &&
153 existingUriIt->second != uri) {
154 documentsByUri_.erase(existingUriIt->second);
155 existingUriIt->second = uri;
156 } else {
157 uriByCanonicalPath_[canonicalKey] = uri;
158 }
159
160 documentsByUri_[uri] =
161 DocumentState{uri, std::move(canonicalPath), std::move(text), version};
162}

175 auto it = documentsByUri_.find(uri);
176 if (it == documentsByUri_.end()) {
177 return;
178 }
179
180 std::string canonicalKey = it->second.path.string();
181 auto pathIt = uriByCanonicalPath_.find(canonicalKey);
182 if (pathIt != uriByCanonicalPath_.end() && pathIt->second == uri) {
183 uriByCanonicalPath_.erase(pathIt);
184 }
185
186 documentsByUri_.erase(it);

193std::optional<ProjectState>
194Workspace::loadProject(const std::string &uri, bool allowEntryErrors) const {
195 auto docIt = documentsByUri_.find(uri);
196 if (docIt == documentsByUri_.end()) {
197 return std::nullopt;
198 }
199
200 auto flags = findAndReadFlags(docIt->second.path);
201
202 ProjectState state;
203 std::set<std::string> visiting;
204 if (!loadModuleGraph(docIt->second.path, state.moduleMap, visiting,
205 state.analysis, uri, flags.importMap,
206 allowEntryErrors)) {
207 if (state.analysis.diagnosticsByUri.find(uri) ==
208 state.analysis.diagnosticsByUri.end()) {
209 state.analysis.diagnosticsByUri[uri] = {};
210 }
211 return state;
212 }
213
214 for (const auto &[moduleId, _] : state.moduleMap) {
215 auto uriIt = uriByCanonicalPath_.find(moduleId);
216 if (uriIt != uriByCanonicalPath_.end()) {
217 state.uriByModuleId[moduleId] = uriIt->second;
218 } else {
219 state.uriByModuleId[moduleId] = pathToUri(moduleId);
220 }
221 }
222
223 auto entrySource = sourceForPath(docIt->second.path);
224 if (entrySource) {
225 auto entryId =
226 std::filesystem::weakly_canonical(docIt->second.path).string();
227 auto entryModuleIt = state.moduleMap.find(entryId);
228 if (entryModuleIt != state.moduleMap.end()) {
229 entryModuleIt->second->isEntry = true;
230 }
231
232 zap::DiagnosticEngine diagnostics(*entrySource,
233 docIt->second.path.string());
234 std::vector<sema::ModuleInfo *> modules;
235 modules.reserve(state.moduleMap.size());
236 for (auto &[_, modulePtr] : state.moduleMap) {
237 diagnostics.registerSource(modulePtr->sourceName, modulePtr->sourceText);
238 modules.push_back(modulePtr.get());
239 }
240
241 sema::Binder binder(diagnostics, true, &state.semanticInfo);
242 auto boundAst = binder.bind(std::move(modules));
243 (void)boundAst;
244 }
245 return state;
246}

248AnalysisResult Workspace::analyze(const std::string &uri) const {
249 AnalysisResult result;
250 auto docIt = documentsByUri_.find(uri);
251 if (docIt == documentsByUri_.end()) {
252 return result;
253 }
254
255 auto flags = findAndReadFlags(docIt->second.path);
256
257 std::map<std::string, std::unique_ptr<sema::ModuleInfo>> moduleMap;
258 std::set<std::string> visiting;
259 if (!loadModuleGraph(docIt->second.path, moduleMap, visiting, result, uri,
260 flags.importMap, false)) {
261 if (result.diagnosticsByUri.find(uri) == result.diagnosticsByUri.end()) {
262 result.diagnosticsByUri[uri] = {};
263 }
264 return result;
265 }
266
267 std::string entryId =
268 std::filesystem::weakly_canonical(docIt->second.path).string();
269 auto entrySource = sourceForPath(docIt->second.path);
270 if (!entrySource) {
271 result.diagnosticsByUri[uri] = {};
272 return result;
273 }
274
275 auto entryModuleIt = moduleMap.find(entryId);
276 if (entryModuleIt == moduleMap.end()) {
277 return result;
278 }
279
280 entryModuleIt->second->isEntry = true;
281
282 zap::DiagnosticEngine diagnostics(*entrySource, docIt->second.path.string());
283 std::vector<sema::ModuleInfo> modules;
284 modules.reserve(moduleMap.size());
285 for (auto &[_, modulePtr] : moduleMap) {
286 diagnostics.registerSource(modulePtr->sourceName, modulePtr->sourceText);
287 modules.push_back(std::move(*modulePtr));
288 }
289
290 sema::Binder binder(diagnostics);
291 auto boundAst = binder.bind(modules);
292 (void)boundAst;
293
294 result.diagnosticsByUri[uri] = diagnostics.diagnostics();
295 return result;
296}
