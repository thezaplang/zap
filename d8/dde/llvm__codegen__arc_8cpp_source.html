58llvm::Value *
59LLVMCodeGen::emitStringRetainIfNeeded(llvm::Value *value,
60 const std::shared_ptr<zir::Type> &type) {
61 if (!isOwnedStringType(type)) {
62 return value;
63 }
64 auto *stringTy = toLLVMType(*type);
65 if (value->getType() != stringTy && isStringLikeStruct(value->getType()) &&
66 isStringLikeStruct(stringTy)) {
67 auto *ptr = builder_.CreateExtractValue(value, {0}, "str.cvt.ptr");
68 auto *len = builder_.CreateExtractValue(value, {1}, "str.cvt.len");
69 llvm::Value *converted = llvm::UndefValue::get(stringTy);
70 converted =
71 builder_.CreateInsertValue(converted, ptr, {0}, "str.cvt.ptr.i");
72 converted =
73 builder_.CreateInsertValue(converted, len, {1}, "str.cvt.len.i");
74 value = converted;
75 }
76 auto *fnTy = llvm::FunctionType::get(stringTy, {stringTy}, false);
77 auto callee = module_->getOrInsertFunction("zap_string_retain", fnTy);
78 return builder_.CreateCall(fnTy, callee.getCallee(), {value}, "str.retain");
79}

136void LLVMCodeGen::emitStoreWithStringArc(llvm::Value *addr, llvm::Value *value,
137 const std::shared_ptr<zir::Type> &type,
138 bool valueIsOwned,
139 bool skipReleaseOld) {
140 if (!skipReleaseOld) {
141 auto *oldValue = builder_.CreateLoad(toLLVMType(*type), addr, "str.old");
142 emitStringReleaseIfNeeded(oldValue, type);

144

147 builder_.CreateStore(storedValue, addr);
