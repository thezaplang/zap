94std::string computeLogicalModulePath(const std::filesystem::path &canonicalPath,
95 const RuntimePaths &paths,
96 const ImportMap &importMap) {
97 auto stdRoot = std::filesystem::weakly_canonical(stdlibRootPath(paths));
98 auto coreRoot = std::filesystem::weakly_canonical(coreRootPath(paths));
99 auto cwdRoot =
100 std::filesystem::weakly_canonical(std::filesystem::current_path());
101
102 auto buildRelative = [&](const std::filesystem::path &root,
103 const std::string &prefix =
104 "") -> std::optional<std::string> {
105 auto rootText = root.generic_string();
106 auto pathText = canonicalPath.generic_string();
107 if (pathText == rootText || pathText.rfind(rootText + "/", 0) != 0) {
108 return std::nullopt;
109 }
110
111 auto rel = std::filesystem::relative(canonicalPath, root);
112 auto stripped = stripSourceExtension(rel);
113 if (prefix.empty()) {
114 return stripped;
115 }
116 return prefix + "/" + stripped;
117 };
118
119 for (const auto &[alias, target] : importMap) {
120 std::error_code ec;
121 auto root = std::filesystem::weakly_canonical(target, ec);
122 if (ec) {
123 continue;
124 }
125 if (auto logical = buildRelative(root, alias)) {
126 return *logical;
127 }
128 }
129
130 if (auto logical = buildRelative(coreRoot, "core")) {
131 if (*logical == "core/core") {
132 return "core";
133 }
134 return *logical;
135 }
136 if (auto logical = buildRelative(stdRoot, "std")) {
137 return *logical;
138 }
139 if (auto logical = buildRelative(cwdRoot)) {
140 return *logical;
141 }
142 return stripSourceExtension(canonicalPath.filename());
143}

183bool resolveImportTargets(const std::filesystem::path &modulePath,
184 const ImportNode &importNode,
185 std::vector<std::filesystem::path> &targets,
186 const ImportMap &importMap, const RuntimePaths &paths,
187 std::string *errorMessage) {
188 std::filesystem::path resolvedPath;
189 const std::string &importPath = importNode.path;
190
191 bool resolvedViaMap = false;
192 for (const auto &[alias, target] : importMap) {
193 if (importPathMatchesAlias(importPath, alias)) {
194 std::string rest = importPath.substr(alias.size());
195 if (!rest.empty() && rest[0] == '/') {
196 rest = rest.substr(1);
197 }
198 resolvedPath = std::filesystem::path(target) / rest;
199 resolvedViaMap = true;
200 break;
201 }
202 }
203
204 if (!resolvedViaMap) {
205 if (importPath == "core") {
206 resolvedPath = coreRootPath(paths) / "core";
207 } else if (importPath.rfind("core/", 0) == 0) {
208 resolvedPath =
209 coreRootPath(paths) / importPath.substr(std::string("core/").size());
210 } else if (importPath.rfind("std/", 0) == 0) {
211 resolvedPath =
212 stdlibRootPath(paths) / importPath.substr(std::string("std/").size());
213 } else {
214 resolvedPath = modulePath.parent_path() / importPath;
215 }
216 }
217 resolvedPath = resolvedPath.lexically_normal();
218
219 auto tryFile = [&](const std::filesystem::path &path) -> bool {
220 if (std::filesystem::exists(path) &&
221 std::filesystem::is_regular_file(path)) {
222 targets.push_back(std::filesystem::weakly_canonical(path));
223 return true;
224 }
225 return false;
226 };
227
228 if (tryFile(resolvedPath)) {
229 return true;
230 }
231 if (resolvedPath.extension() != ".zp" &&
232 tryFile(resolvedPath.string() + ".zp")) {
233 return true;
234 }
235
236 if (std::filesystem::exists(resolvedPath) &&
237 std::filesystem::is_directory(resolvedPath)) {
238 for (const auto &entry :
239 std::filesystem::directory_iterator(resolvedPath)) {
240 if (entry.is_regular_file() && entry.path().extension() == ".zp") {
241 targets.push_back(std::filesystem::weakly_canonical(entry.path()));
242 }
243 }
244 std::sort(targets.begin(), targets.end());
245 return !targets.empty();
246 }
247
248 setImportError(errorMessage, resolvedPath);
249 return false;
250}
