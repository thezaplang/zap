105ParseResult parse(const std::vector<std::string_view> &cmdline,
106 CmdlineArgs &args) {
107 bool ok = true;
108 std::vector<ArgVal> argsvec;
109
110 for (size_t i = 1; i < cmdline.size(); i++) {
111 std::string_view original(cmdline[i]);
112
113 if (original[0] == '-') {
114 constexpr std::string_view targetPrefix = "--target=";
115 if (original.rfind(targetPrefix, 0) == 0) {
116 argsvec.emplace_back(original, original.substr(targetPrefix.size()),
117 ArgTypes::Target);
118 continue;
119 }
120
121 if (auto it = arg_map.find(original); it != arg_map.end()) {
122 const ArgConf &conf = it->second;
123
124 if (conf.getKind() == ArgConf::ArgKind::Flag) {
125 argsvec.emplace_back(original, std::string_view(), conf.getType());
126 } else {
127 if (i + 1 == cmdline.size()) {
128 reportError("missing value for flag '", it->first, "'");
129 ok = false;
130 break;
131 }
132 argsvec.emplace_back(original, cmdline[i + 1], conf.getType());
133 i++;
134 }
135 } else {
136 const ArgConf *maybeConf = nullptr;
137 size_t offset = 0;
138#define ZAP_FLAG(ID, STR, MSG, KIND, ...) \
139 if constexpr (ArgConf::ArgKind::KIND == ArgConf::ArgKind::Joined || \
140 ArgConf::ArgKind::KIND == ArgConf::ArgKind::JoinedSeparate) { \
141 if (original.size() >= (sizeof(STR) - 1)) { \
142 auto it = arg_map.find(original.substr(0, (sizeof(STR) - 1))); \
143 if (it != arg_map.end()) { \
144 maybeConf = &it->second; \
145 offset = sizeof(STR) - 1; \
146 goto match_found; \
147 } \
148 } \
149 }
150#include "flags.inc"
151#undef ZAP_FLAG
152 reportError("unknown flag '", original, "' encountered");
153 ok = false;
154 match_found:
155 argsvec.emplace_back(original, original.substr(offset),
156 maybeConf->getType());
157 }
158 } else {
159 args.inputs.emplace_back(original);
160 }
161 }
162
163 if (!ok)
164 return ParseResult::Failed;
165
166 ArgHolder holder(argsvec);
167
168 if (holder.has(ArgTypes::Help)) {
169 printHelp();
170 return ParseResult::SkipCompilation;
171 }
172 if (holder.has(ArgTypes::Version)) {
173 printVersion();
174 return ParseResult::SkipCompilation;
175 }
176 if (holder.has(ArgTypes::PrintStdlibPath)) {
177 args.printStdlibPath = true;
178 return ParseResult::Success;
179 }
180 if (holder.has(ArgTypes::PrintCorePath)) {
181 args.printCorePath = true;
182 return ParseResult::Success;
183 }
184
185 args.output.implicit = !holder.has(ArgTypes::Output);
186 args.freestanding = holder.has(ArgTypes::Freestanding);
187 args.incStdlib = !args.freestanding && !holder.has(ArgTypes::NoStdlib);
188 args.incPrelude = !args.freestanding && !holder.has(ArgTypes::NoPrelude);
189
190 bool emitS = holder.has(ArgTypes::CompileOnlyS);
191 bool compileOnly = holder.has(ArgTypes::CompileOnly);
192 bool emitLLVM = holder.has(ArgTypes::EmitLLVM);
193 bool emitZIR = holder.has(ArgTypes::EmitZIR);
194
195 if (holder.has(ArgTypes::OptLevel)) {
196 std::string_view optL = holder.get(ArgTypes::OptLevel)->optional;
197
198 if (optL == "0") {
199 args.optLevel = OptLevel::O0;
200 } else if (optL == "1") {
201 args.optLevel = OptLevel::O1;
202 } else if (optL == "2") {
203 args.optLevel = OptLevel::O2;
204 } else if (optL == "3") {
205 args.optLevel = OptLevel::O3;
206 } else {
207 reportError("unknown optimization value '", optL, "' encountered");
208 ok = false;
209 }
210 } else
211 args.optLevel = OptLevel::O0;
212
213 if (!ok)
214 return ParseResult::Failed;
215
216 if (holder.has(ArgTypes::Target)) {
217 std::string_view target = holder.get(ArgTypes::Target)->optional;
218 if (target.empty()) {
219 reportError("--target requires a target triple");
220 return ParseResult::Failed;
221 }
222 args.targetTriple = std::string(target);
223 }
224
225 for (const ArgVal *arg : holder.getAll(ArgTypes::LinkDir)) {
226 std::string val = "-L";
227 val += arg->optional;
228
229 args.linkerArgs.emplace_back(std::move(val));
230 }
231
232 for (const ArgVal *arg : holder.getAll(ArgTypes::LinkLib)) {
233 std::string val = "-l";
234 val += arg->optional;
235
236 args.linkerArgs.emplace_back(std::move(val));
237 }
238
239 for (const ArgVal *arg : holder.getAll(ArgTypes::ImportMap)) {
240 std::string_view entry = arg->optional;
241 auto eq = entry.find('=');
242 if (eq == std::string_view::npos) {
243 reportError("--import-map requires format alias=path, got: ", entry);
244 ok = false;
245 continue;
246 }
247 args.importMap[std::string(entry.substr(0, eq))] =
248 std::string(entry.substr(eq + 1));
249 }
250
251 if (!ok)
252 return ParseResult::Failed;
253
254 if (emitZIR) {
255 args.output.type = OutputType::ZIR;
256 } else if (emitLLVM) {
257 args.output.type = emitS ? OutputType::TEXT_LLVM : OutputType::LLVM;
258 } else if (emitS) {
259 args.output.type = OutputType::ASM;
260 } else if (compileOnly) {
261 args.output.type = OutputType::OBJECT;
262 }
263
264 if ((emitLLVM || emitZIR) && !args.output.implicit && (emitLLVM == emitZIR)) {
265 reportError("cannot use -o when emitting multiple text outputs");
266 return ParseResult::Failed;
267 }
268
269 args.output.path = holder.has(ArgTypes::Output)
270 ? holder.get(ArgTypes::Output)->optional
271 : "a.out";
272
273 if (args.inputs.empty()) {
274 reportError("no input files");
275 return ParseResult::Failed;
276 }
277
278 return ParseResult::Success;
279}
